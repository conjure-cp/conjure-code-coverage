   42 parseModel :: Parser Model
   43 parseModel = inCompleteFile $ do
   44     let
   45         pLanguage :: Parser LanguageVersion
   46         pLanguage = do
   47             lexeme L_language
   48             pos1 <- getPosition
   49             l <- identifierText
   50             -- ESSENCE' is accepted, just for convenience
   51             unless (l `elem` ["Essence", "ESSENCE", "ESSENCE'"]) $ do
   52                 setPosition pos1
   53                 fail $ "language name has to be Essence, but given:" <+> pretty l
   54             pos2 <- getPosition
   55             is <- sepBy1 integer dot
   56             unless (is >= [1]) $ do
   57                 setPosition pos2
   58                 fail $ "language version expected to be at least 1.0, but given:" <+>
   59                             pretty (intercalate "." (map show is))
   60             return (LanguageVersion (Name l) (map fromInteger is))
   61     l  <- optional pLanguage
   62     xs <- many parseTopLevels
   63     return Model
   64         { mLanguage = fromMaybe def l
   65         , mStatements = concat xs
   66         , mInfo = def
   67         }

   70 parseIO :: MonadFail m => Parser a -> String -> m a
   71 parseIO p s =
   72     case runLexerAndParser (inCompleteFile p) "" (T.pack s) of
   73         Left err -> fail err
   74         Right x  -> return x

   78 translateQnName qnName = case qnName of

   81     _        -> qnName

   91 parseTopLevels :: Parser [Statement]
   92 parseTopLevels = do
   93     let one = satisfyL $ \case
   94                 L_find -> Just $ do
   95                     decls <- flip sepEndBy1 comma $ do
   96                         is <- commaSeparated parseNameOrMeta
   97                         j  <- colon >> parseDomain
   98                         return [ Declaration (FindOrGiven Find i j)
   99                                | i <- is ]
  100                     return $ concat decls
  101                 L_given -> Just $ do
  102                     decls <- commaSeparated $ do
  103                         is <- commaSeparated parseName
  104                         msum
  105                             [ do
  106                                 colon
  107                                 j <- parseDomain
  108                                 return [ Declaration (FindOrGiven Given i j)
  109                                        | i <- is ]
  110                             , do
  111                                 lexeme L_new
  112                                 msum
  113                                     [ do
  114                                         lexeme L_type
  115                                         lexeme L_enum
  116                                         modify (\ st -> st { enumDomains = is ++ enumDomains st } )
  117                                         return [ Declaration (GivenDomainDefnEnum i)
  118                                                | i <- is ]
  119                                     ]
  120                             ]
  121                     return $ concat decls
  122                 L_letting -> Just $ do
  123                     decls <- commaSeparated $ do
  124                         is <- commaSeparated parseName
  125                         lexeme L_be
  126                         msum
  127                             [ do
  128                                 lexeme L_new
  129                                 lexeme L_type
  130                                 msum
  131                                     [ do
  132                                         lexeme L_of
  133                                         lexeme $ LIdentifier "size"
  134                                         j <- parseExpr
  135                                         return [ Declaration (LettingDomainDefnUnnamed i j)
  136                                                | i <- is
  137                                                ]
  138                                     , do
  139                                         lexeme L_enum
  140                                         ys <- braces (commaSeparated parseName) <|> return []
  141                                         modify (\ st -> st { enumDomains = is ++ enumDomains st } )
  142                                         return [ Declaration (LettingDomainDefnEnum i ys)
  143                                                | i <- is
  144                                                ]
  145                                     ]
  146                             , do
  147                                 lexeme L_domain
  148                                 j <- parseDomain
  149                                 return [ Declaration (Letting i (Domain j))
  150                                        | i <- is
  151                                        ]
  152                             , do
  153                                 j <- parseExpr
  154                                 return [ Declaration (Letting i j)
  155                                        | i <- is
  156                                        ]
  157                             ]
  158                     return $ concat decls
  159                 L_where -> Just $ do
  160                     xs <- commaSeparated parseExpr
  161                     return [Where xs]
  162                 L_such -> Just $ do
  163                     lexeme L_that
  164                     xs <- commaSeparated parseExpr
  165                     return [SuchThat xs]
  166                 L_minimising -> Just $ do
  167                     x <- parseExpr
  168                     return [ Objective Minimising x ]
  169                 L_maximising -> Just $ do
  170                     x <- parseExpr
  171                     return [ Objective Maximising x ]
  172                 L_branching -> Just $ do
  173                     lexeme L_on
  174                     xs <- brackets $ commaSeparated parseSearchOrder
  175                     return [ SearchOrder xs ]
  176                 L_heuristic -> Just $ do
  177                     nm <- parseName
  178                     return [ SearchHeuristic nm ]
  179                 _ -> Nothing
  180     concat <$> some (one <?> "statement")

  182 parseSearchOrder :: Parser SearchOrder
  183 parseSearchOrder = msum [try pBranchingOn, pCut]
  184     where
  185         pBranchingOn = BranchingOn <$> parseName
  186         pCut = Cut <$> parseExpr

  188 parseRange :: Parser a -> Parser (Range a)
  189 parseRange p = msum [try pRange, pSingle] <?> "range"
  190     where
  191         pRange = do
  192             fr <- optional p
  193             dotdot
  194             to <- optional p
  195             return $ case (fr,to) of
  196                 (Nothing, Nothing) -> RangeOpen
  197                 (Just x , Nothing) -> RangeLowerBounded x
  198                 (Nothing, Just y ) -> RangeUpperBounded y
  199                 (Just x , Just y ) -> RangeBounded x y
  200         pSingle = do
  201             x <- p
  202             return (RangeSingle x)

  204 parseDomain :: Parser (Domain () Expression)
  205 parseDomain = (forgetRepr <$> parseDomainWithRepr) <?> "domain"

  208 parseDomainWithRepr = pDomainAtom

  225     where
  226 
  227         pDomainAtom = msum
  228             [ pBool, try pIntFromExpr, pInt, try pEnum, try pReference
  229             , pMatrix, try pTupleWithout, pTupleWith
  230             , pRecord, pVariant
  231             , pSet
  232             , pMSet
  233             , try pFunction', pFunction
  234             , pSequence
  235             , pRelation
  236             , pPartition
  237             , DomainMetaVar <$> parseMetaVariable, parens parseDomainWithRepr
  238             ]
  239 
  240         parseRepr = return NoRepresentation
  241         -- -- Parsing set {representation} of ... notation if needed
  242         -- parseRepr = msum [ braces parseReprInner
  243         --                  , return NoRepresentation
  244         --                  ]
  245         --
  246         -- parseReprInner = do
  247         --     pos    <- getPosition
  248         --     nm     <- identifierText
  249         --     inners <- fromMaybe [] <$> optional (brackets (commaSeparated parseReprInner))
  250         --     case textToRepresentation nm inners of
  251         --         Nothing -> do
  252         --             setPosition pos
  253         --             fail ("Not a valid representation:" <+> pretty nm)
  254         --         Just r  -> return r
  255 
  256         pBool = do
  257             lexeme L_bool
  258             -- parse and discard, compatibility with SR
  259             _ <- optional $ parens $ commaSeparated0 $ parseRange parseExpr
  260             return DomainBool
  261 
  262         pIntFromExpr = do
  263             lexeme L_int
  264             x <- parens parseExpr
  265             case (let ?typeCheckerMode = StronglyTyped in typeOf x) of
  266                 Just (TypeInt TagInt) -> return $ DomainInt TagInt [RangeSingle x]
  267                 _ -> return $ DomainIntE x
  268 
  269         pInt = do
  270             lexeme L_int
  271             mxs <- optional $ parens $ commaSeparated0 $ parseRange parseExpr
  272             let xs = fromMaybe [] mxs
  273             return $ DomainInt TagInt xs
  274 
  275         pReference = do
  276             r  <- identifierText
  277             return $ DomainReference (Name r) Nothing
  278 
  279         pEnum = do
  280             r  <- identifierText
  281             xs <- optional $ parens $ commaSeparated0 $ parseRange parseExpr
  282             st <- get
  283             guard (Name r `elem` enumDomains st)
  284             return $ DomainEnum (Name r) xs Nothing
  285 
  286         pMatrix = do
  287             lexeme L_matrix
  288             lexeme L_indexed
  289             lexeme L_by
  290             xs <- brackets (commaSeparated parseDomain)
  291             lexeme L_of
  292             y  <- parseDomainWithRepr
  293             return $ foldr DomainMatrix y xs
  294 
  295         pTupleWith = do
  296             lexeme L_tuple
  297             xs <- parens $ commaSeparated0 parseDomainWithRepr
  298             return $ DomainTuple xs
  299 
  300         pTupleWithout = do
  301             xs <- parens $ countSepAtLeast 2 parseDomainWithRepr comma
  302             return $ DomainTuple xs
  303 
  304         pRecord = do
  305             lexeme L_record
  306             let one = do n <- parseName
  307                          lexeme L_Colon
  308                          d <- parseDomainWithRepr
  309                          return (n,d)
  310             xs <- braces $ commaSeparated0 one
  311             return $ DomainRecord xs
  312 
  313         pVariant = do
  314             lexeme L_variant
  315             let one = do n <- parseName
  316                          lexeme L_Colon
  317                          d <- parseDomainWithRepr
  318                          return (n,d)
  319             xs <- braces $ commaSeparated0 one
  320             return $ DomainVariant xs
  321 
  322         pSet = do
  323             lexeme L_set
  324             r <- parseRepr
  325             x <- parseSetAttr
  326             y <- lexeme L_of >> parseDomainWithRepr
  327             return $ DomainSet r x y
  328 
  329         pMSet = do
  330             lexeme L_mset
  331             r <- parseRepr
  332             x <- parseMSetAttr
  333             y <- lexeme L_of >> parseDomainWithRepr
  334             return $ DomainMSet r x y
  335 
  336         pFunction' = do
  337             lexeme L_function
  338             r <- parseRepr
  339             (y,z) <- arrowedPair parseDomainWithRepr
  340             return $ DomainFunction r def y z
  341 
  342         pFunction = do
  343             lexeme L_function
  344             r <- parseRepr
  345             x <- parseFunctionAttr
  346             (y,z) <- arrowedPair parseDomainWithRepr
  347             return $ DomainFunction r x y z
  348 
  349         pSequence = do
  350             lexeme L_sequence
  351             r <- parseRepr
  352             x <- parseSequenceAttr
  353             y <- lexeme L_of >> parseDomainWithRepr
  354             return $ DomainSequence r x y
  355 
  356         pRelation = do
  357             lexeme L_relation
  358             r  <- parseRepr
  359             pos <- getPosition
  360             x  <- parseRelationAttr
  361             lexeme L_of
  362             ys <- parens (parseDomainWithRepr `sepBy` lexeme L_Times)
  363             let RelationAttr _ (BinaryRelationAttrs binAttrs) = x
  364             when (length ys /= 2 && not (S.null binAttrs)) $ do
  365                 setPosition pos
  366                 fail $ "Only binary relations can have these attributes:" <+>
  367                             prettyList id "," (S.toList binAttrs)
  368             return $ DomainRelation r x ys
  369 
  370         pPartition = do
  371             lexeme L_partition
  372             r <- parseRepr
  373             x <- parsePartitionAttr
  374             lexeme L_from
  375             y <- parseDomainWithRepr
  376             return $ DomainPartition r x y

  378 parseAttributes :: Parser (DomainAttributes Expression)
  379 parseAttributes = do
  380     xs <- parens (commaSeparated0 parseAttribute) <|> return []
  381     return $ DomainAttributes xs
  382     where
  383         parseAttribute = msum [parseDontCare, try parseNameValue, parseDAName]
  384         parseNameValue = DANameValue <$> (Name <$> identifierText) <*> parseExpr
  385         parseDAName = DAName <$> (Name <$> identifierText)
  386         parseDontCare = do dotdot ; return DADotDot

  388 parseSetAttr :: Parser (SetAttr Expression)

  390     pos <- getPosition
  391     DomainAttributes attrs <- parseAttributes
  392     checkExtraAttributes pos "set" attrs

  398         [DANameValue "maxSize" a] -> return (SizeAttr_MaxSize a)

  400         as -> do
  401             setPosition pos
  402             fail ("incompatible attributes:" <+> stringToDoc (show as))

  551         [DANameValue "maxPartSize" a] -> return (SizeAttr_MaxSize a)
  552         [DANameValue "maxPartSize" b, DANameValue "minPartSize" a] -> return (SizeAttr_MinMaxSize a b)

  560 checkExtraAttributes :: SourcePos -> Doc -> [DomainAttribute a] -> [Name] -> Parser ()
  561 checkExtraAttributes pos ty attrs supported = do
  562     let extras = mapMaybe f attrs
  563     unless (null extras) $ do
  564         setPosition pos
  565         fail $ vcat [ "Unsupported attributes for" <+> ty <> ":" <+> prettyList id "," extras
  566                     , "Only these are supported:" <+> prettyList id "," supported
  567                     ]
  568     where
  569         f (DANameValue nm _) | nm `notElem` supported = Just nm
  570         f (DAName      nm  ) | nm `notElem` supported = Just nm
  571         f _ = Nothing
  572 
  573 filterAttrName :: Ord a => [Name] -> [DomainAttribute a] -> [DomainAttribute a]
  574 filterAttrName keep = sort . filter f
  575     where
  576         f (DANameValue nm _) | nm `elem` keep = True
  577         f (DAName      nm  ) | nm `elem` keep = True
  578         f _ = False
  579 
  580 filterSizey :: Ord a => [DomainAttribute a] -> [DomainAttribute a]
  581 filterSizey = filterAttrName ["size", "minSize", "maxSize"]
  582 
  583 filterJectivity :: Ord a => [DomainAttribute a] -> [DomainAttribute a]
  584 filterJectivity = filterAttrName ["injective", "surjective", "bijective"]

  613 parseExpr :: Parser Expression
  614 parseExpr =
  615     let
  616         mergeOp op = mkBinOp (lexemeText op)
  617 
  618         operatorsGrouped = operators
  619             |> sortBy  (\ (_,a) (_,b) -> compare a b )
  620             |> groupBy (\ (_,a) (_,b) -> a == b )
  621             |> reverse
  622 
  623         parseUnaryNegate = do
  624             lexeme L_Minus
  625             return $ \ x -> mkOp "negate" [x]
  626 
  627         parseUnaryNot = do
  628             lexeme L_ExclamationMark
  629             return $ \ x -> mkOp "not" [x]
  630 
  631     in
  632         makeExprParser (parseAtomicExpr <?> "expression")
  633             [ [ case descr of
  634                     BinaryOp op FLeft             -> InfixL $ do lexeme op
  635                                                                  return $ mergeOp op
  636                     BinaryOp op FNone             -> InfixN $ do lexeme op
  637                                                                  return $ mergeOp op
  638                     BinaryOp op FRight            -> InfixR $ do lexeme op
  639                                                                  return $ mergeOp op
  640                     UnaryPrefix L_Minus           -> Prefix $ foldr1 (.) <$> some parseUnaryNegate
  641                     UnaryPrefix L_ExclamationMark -> Prefix $ foldr1 (.) <$> some parseUnaryNot
  642                     UnaryPrefix l                 -> bug ("Unknown UnaryPrefix" <+> pretty (show l))
  643               | (descr, _) <- operatorsInGroup
  644               ]
  645             | operatorsInGroup <- operatorsGrouped
  646             ]
  647 
  648 parseAtomicExpr :: Parser Expression
  649 parseAtomicExpr = do
  650     let
  651         prefixes = do
  652             fs <- some $ msum parsePrefixes
  653             return $ foldr1 (.) fs
  654         postfixes = do
  655             fs <- some $ msum parsePostfixes
  656             return $ foldr1 (.) (reverse fs)
  657         withPrefix  x = try x <|> do f <- prefixes; i <- x; return $ f i
  658         withPostfix x = do i <- x; mf <- optional postfixes; return $ case mf of Nothing -> i
  659                                                                                  Just f  -> f i
  660     withPrefix (withPostfix parseAtomicExprNoPrePost) <?> "expression"

  663 parseAtomicExprNoPrePost :: Parser Expression
  664 parseAtomicExprNoPrePost = msum $ map try $ concat
  665     [ [parseQuantifiedExpr]
  666     , parseOthers
  667     , [metaVarInE <$> parseMetaVariable]
  668     , [parseAAC]
  669     , [parseReference]
  670     , [parseLiteral]
  671     , [parseDomainAsExpr]
  672     , [parseWithLocals]
  673     , [parseComprehension]
  674     , [parens parseExpr]

  677 parseComprehension :: Parser Expression
  678 parseComprehension = brackets $ do
  679     x   <- parseExpr
  680     lexeme L_Bar
  681     gens <- commaSeparated (letting <|> try generator <|> condition)
  682     return (Comprehension x (concat gens))
  683     where
  684         generator :: Parser [GeneratorOrCondition]
  685         generator = do
  686             pats <- commaSeparated parseAbstractPattern
  687             msum
  688                 [ do
  689                     lexeme L_Colon
  690                     domain <- parseDomain
  691                     return [Generator (GenDomainNoRepr pat domain) | pat <- pats]
  692                 , do
  693                     lexeme L_LeftArrow
  694                     expr <- parseExpr
  695                     return [Generator (GenInExpr       pat expr)   | pat <- pats]
  696                 ]
  697         condition :: Parser [GeneratorOrCondition]
  698         condition = return . Condition <$> parseExpr
  699         letting :: Parser [GeneratorOrCondition]
  700         letting = do
  701             lexeme L_letting
  702             pat <- parseAbstractPattern
  703             lexeme L_be
  704             x <- parseExpr
  705             return [ComprehensionLetting pat x]

  710 parsePrefixes :: [Parser (Expression -> Expression)]
  711 parsePrefixes = [parseUnaryMinus, parseUnaryNot]
  712     where
  713         parseUnaryMinus = do
  714             lexeme L_Minus
  715             return $ \ x -> mkOp "negate" [x]
  716         parseUnaryNot = do
  717             lexeme L_ExclamationMark
  718             return $ \ x -> mkOp "not" [x]

  720 parsePostfixes :: [Parser (Expression -> Expression)]
  721 parsePostfixes = [parseIndexed,parseFactorial,parseFuncApply]
  722     where
  723         parseIndexed :: Parser (Expression -> Expression)
  724         parseIndexed = do
  725             let
  726                 pIndexer = try pRList <|> (do i <- parseExpr ; return $ \ m -> Op (MkOpIndexing (OpIndexing m i)))
  727                 pRList   = do
  728                     i <- optional parseExpr
  729                     dotdot
  730                     j <- optional parseExpr
  731                     return $ \ m -> Op (MkOpSlicing (OpSlicing m i j))
  732             is <- brackets $ commaSeparated pIndexer
  733             return $ \ x -> foldl (\ m f -> f m ) x is
  734         parseFactorial :: Parser (Expression -> Expression)
  735         parseFactorial = do
  736             lexeme L_ExclamationMark
  737             return $ \ x -> mkOp "factorial" [x]
  738         parseFuncApply :: Parser (Expression -> Expression)
  739         parseFuncApply = parens $ do
  740             xs <- commaSeparated parseExpr
  741             let underscore = Reference "_" Nothing
  742             let ys = [ if underscore == x then Nothing else Just x | x <- xs ]
  743             return $ \ x -> Op $ MkOpRelationProj $ OpRelationProj x ys

  745 parseAAC :: Parser Expression
  746 parseAAC = do
  747     let
  748         isAttr (LIdentifier txt) | Just _ <- Name txt `lookup` allSupportedAttributes = True
  749         isAttr _ = False
  750     LIdentifier attr <- satisfyT isAttr
  751     let n = fromMaybe (bug "parseAAC") (lookup (Name attr) allSupportedAttributes)
  752     args <- parens $ countSep (n+1) parseExpr comma
  753     case (n, args) of
  754         (0, [e  ]) -> return $ Op $ MkOpAttributeAsConstraint $ OpAttributeAsConstraint e
  755                                         (fromString (textToString attr)) Nothing
  756         (1, [e,v]) -> return $ Op $ MkOpAttributeAsConstraint $ OpAttributeAsConstraint e
  757                                         (fromString (textToString attr)) (Just v)
  758         _ -> fail "parseAAC"

  760 parseOthers :: [Parser Expression]
  761 parseOthers = [ parseFunctional l
  762               | l <- functionals
  763               ] ++ [parseTyped, parseTwoBars]
  764     where
  765 
  766         parseTwoBars :: Parser Expression
  767         parseTwoBars = do
  768             x <- between (lexeme L_Bar) (lexeme L_Bar) parseExpr
  769             return (mkOp "twoBars" [x])
  770 
  771         parseTyped :: Parser Expression
  772         parseTyped = parens $ do
  773             x  <- parseExpr
  774             lexeme L_Colon
  775             d  <- betweenTicks parseDomain
  776             ty <- let ?typeCheckerMode = StronglyTyped in typeOfDomain d
  777             return (Typed x ty)
  778 
  779         parseFunctional :: Lexeme -> Parser Expression
  780         parseFunctional l = do
  781             lexeme l
  782             xs <- parens $ commaSeparated parseExpr
  783             return $ case (l,xs) of
  784                 (L_image, [y,z]) -> Op $ MkOpImage $ OpImage y z
  785                 _ -> mkOp (fromString $ show $ lexemeFace l) xs

  787 parseWithLocals :: Parser Expression
  788 parseWithLocals = braces $ do
  789     i  <- parseExpr
  790     lexeme L_At
  791     js <- parseTopLevels
  792     let decls =
  793             [ Declaration (FindOrGiven LocalFind nm dom)
  794             | Declaration (FindOrGiven Find nm dom) <- js ]
  795     let cons = concat
  796             [ xs
  797             | SuchThat xs <- js
  798             ]
  799     let locals = if null decls
  800                     then DefinednessConstraints cons
  801                     else AuxiliaryVars (decls ++ [SuchThat cons])
  802     return (WithLocals i locals)

  813 parseQuantifiedExpr :: Parser Expression
  814 parseQuantifiedExpr = do
  815     qnName      <- parseQuantifiedName
  816     qnPats      <- commaSeparated parseAbstractPattern
  817     qnOver      <- msum [ Left  <$> (colon *> parseDomain)
  818                         , Right <$> do
  819                             lexeme L_in
  820                             over <- parseExpr
  821                             return (\ pat -> GenInExpr pat over )
  822                         , Right <$> do
  823                             lexeme L_subsetEq
  824                             over <- parseExpr
  825                             return (\ pat -> GenInExpr pat (Op $ MkOpPowerSet $ OpPowerSet over) )
  826                         ]
  827     qnGuard     <- optional (comma *> parseExpr)
  828     qnBody      <- dot *> parseExpr <?> "body of a quantified expression"
  829 
  830     let qnMap pat = case qnOver of
  831             Left dom -> GenDomainNoRepr pat dom
  832             Right op -> op pat
  833 
  834     return $ mkOp (translateQnName qnName)
  835            $ return
  836            $ Comprehension qnBody
  837            $ [ Generator (qnMap pat) | pat    <- qnPats    ] ++
  838              [ Condition g           | Just g <- [qnGuard] ]

  850 parseAbstractPattern :: Parser AbstractPattern
  851 parseAbstractPattern = label "pattern" $ msum
  852     [ AbstractPatternMetaVar <$> parseMetaVariable
  853     , Single <$> parseName
  854     , do
  855         void $ optional $ lexeme L_tuple
  856         xs <- parens $ commaSeparated parseAbstractPattern
  857         return (AbsPatTuple xs)
  858     , do
  859         xs <- brackets $ commaSeparated parseAbstractPattern
  860         return (AbsPatMatrix xs)
  861     , do
  862         xs <- braces $ commaSeparated parseAbstractPattern
  863         return (AbsPatSet xs)
  864     ]

  866 parseLiteral :: Parser Expression
  867 parseLiteral = label "value" $ msum
  868     [ Constant <$> pBool
  869     , Constant <$> pInt
  870     , mkAbstractLiteral <$> pMatrix
  871     , mkAbstractLiteral <$> pTupleWith
  872     , mkAbstractLiteral <$> pTupleWithout
  873     , mkAbstractLiteral <$> pRecord
  874     , AbstractLiteral <$> pVariant
  875     , mkAbstractLiteral <$> pSet
  876     , mkAbstractLiteral <$> pMSet
  877     , mkAbstractLiteral <$> pFunction
  878     , mkAbstractLiteral <$> pSequence
  879     , mkAbstractLiteral <$> pRelation
  880     , mkAbstractLiteral <$> pPartition
  881     ]
  882     where
  883 
  884         -- convert x to a constant if possible
  885         -- might save us from evaluating it again and again later
  886         mkAbstractLiteral x =
  887             case e2c (AbstractLiteral x) of
  888                 Nothing -> AbstractLiteral x
  889                 Just c  -> Constant c
  890 
  891         pBool = do
  892             x <- False <$ lexeme L_false
  893                  <|>
  894                  True  <$ lexeme L_true
  895             return (ConstantBool x)
  896 
  897         pInt = ConstantInt TagInt . fromInteger <$> integer
  898 
  899         pMatrix = do
  900             lexeme L_OpenBracket
  901             xs <- commaSeparated0 parseExpr
  902             msum
  903                 [ do
  904                     let r = mkDomainIntB 1 (fromInt (genericLength xs))
  905                     lexeme L_CloseBracket
  906                     return (AbsLitMatrix r xs)
  907                 , do
  908                     lexeme L_SemiColon
  909                     r <- parseDomain
  910                     lexeme L_CloseBracket
  911                     return (AbsLitMatrix r xs)
  912                 ]
  913 
  914         pTupleWith = do
  915             lexeme L_tuple
  916             xs <- parens $ commaSeparated0 parseExpr
  917             return (AbsLitTuple xs)
  918 
  919         pTupleWithout = do
  920             xs <- parens $ countSepAtLeast 2 parseExpr comma
  921             return (AbsLitTuple xs)
  922 
  923         pRecord = do
  924             lexeme L_record
  925             let one = do n <- parseName
  926                          lexeme L_Eq
  927                          x <- parseExpr
  928                          return (n,x)
  929             xs <- braces $ commaSeparated0 one
  930             return $ AbsLitRecord xs
  931 
  932         pVariant = do
  933             lexeme L_variant
  934             let one = do n <- parseName
  935                          lexeme L_Eq
  936                          x <- parseExpr
  937                          return (n,x)
  938             (n,x) <- braces one
  939             return $ AbsLitVariant Nothing n x
  940 
  941         pSet = do
  942             xs <- braces (commaSeparated0 parseExpr)
  943             return (AbsLitSet xs)
  944 
  945         pMSet = do
  946             lexeme L_mset
  947             xs <- parens (commaSeparated0 parseExpr)
  948             return (AbsLitMSet xs)
  949 
  950         pFunction = do
  951             lexeme L_function
  952             xs <- parens (commaSeparated0 inner)
  953             return (AbsLitFunction xs)
  954             where
  955                 inner = arrowedPair parseExpr
  956 
  957         pSequence = do
  958             lexeme L_sequence
  959             xs <- parens (commaSeparated0 parseExpr)
  960             return (AbsLitSequence xs)
  961 
  962         pRelation = do
  963             lexeme L_relation
  964             xs <- parens (commaSeparated0 (pTupleWith <|> pTupleWithout))
  965             xsFiltered <- forM xs $ \case
  966                 -- Constant (ConstantAbstract (AbsLitTuple is)) -> ...
  967                 AbsLitTuple is -> return is
  968                 x -> fail ("Cannot parse as part of relation literal:" <+> vcat [pretty x, pretty (show x)])
  969             return (AbsLitRelation xsFiltered)
  970 
  971         pPartition = do
  972             lexeme L_partition
  973             xs <- parens (commaSeparated0 inner)
  974             return (AbsLitPartition xs)
  975             where
  976                 inner = braces (commaSeparated0 parseExpr)

  983 runLexerAndParser :: MonadFail m => Parser a -> String -> T.Text -> m a
  984 runLexerAndParser p file inp = do
  985     ls <- runLexer inp
  986     case runParser p file ls of
  987         Left (msg, line, col) ->
  988             let theLine = T.lines inp |> drop (line-1) |> take 1
  989             in  fail $ vcat
  990                     [ msg
  991                     , vcat (map pretty theLine)
  992                     , pretty $ replicate (col-1) ' ' ++ "^"
  993                     ]
  994         Right x -> return x

  996 runParser :: Parser a -> String -> [LexemePos] -> Either (Doc, Int, Int) a
  997 runParser p file ls = either modifyErr Right (P.runParser (evalStateT p (ParserState [])) file ls)
  998     where
  999         modifyErr :: ParseError -> Either (Doc, Int, Int) a
 1000         modifyErr e = Left $
 1001             let pos  = errorPos e
 1002             in  ( if file `isPrefixOf` show e
 1003                     then                       pretty (show e)
 1004                     else pretty file <> ":" <> pretty (show e)
 1005                 , sourceLine   pos
 1006                 , sourceColumn pos
 1007                 )

 1011     LIdentifier i <- satisfyT isIdentifier
 1012     return i

 1014           isIdentifier (LIdentifier "forAll") = False

 1019 satisfyT :: (Lexeme -> Bool) -> Parser Lexeme
 1020 satisfyT predicate = token nextPos testTok
 1021     where
 1022         testTok :: LexemePos -> Either [Message] Lexeme
 1023         testTok (LexemePos tok _ _) = if predicate tok then Right tok else Left [Unexpected (showToken tok)]
 1024         nextPos :: Int -> SourcePos -> LexemePos -> SourcePos
 1025         nextPos _ _ (LexemePos _ _ pos) = pos
 1026 
 1027 satisfyL :: forall a . (Lexeme -> Maybe (Parser a)) -> Parser a
 1028 satisfyL predicate = do
 1029     p <- token nextPos testTok
 1030     p
 1031     where
 1032         testTok :: LexemePos -> Either [Message] (Parser a)
 1033         testTok (LexemePos tok _ _) =
 1034             -- trace ("satisfyL: " ++ show pos ++ "\t" ++ show tok) $
 1035             case predicate tok of
 1036                 Nothing  -> Left [Unexpected (showToken tok)]
 1037                 Just res -> Right res
 1038 
 1039         nextPos :: Int -> SourcePos -> LexemePos -> SourcePos
 1040         nextPos _ _ (LexemePos _ _ pos) = pos

 1070 -- parses a specified number of elements separated by the given separator
 1071 countSep :: Int -> Parser a -> Parser sep -> Parser [a]
 1072 countSep 1 p _ = (:[]) <$> p
 1073 countSep i p separator | i > 1 = (:) <$> (p <* separator) <*> countSep (i-1) p separator
 1074 countSep _ _ _ = return []

 1076 -- parses at least a given number of elements separated by the given separator
 1077 countSepAtLeast :: Int -> Parser a -> Parser sep -> Parser [a]
 1078 countSepAtLeast i p separator = (++) <$> countSep i p separator <*> many (separator *> p)
