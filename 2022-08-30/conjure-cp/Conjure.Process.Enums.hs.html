   28 -- | The argument is a model before nameResolution.
   29 --   Only intended to work on problem specifications.
   30 removeEnumsFromModel ::
   31     MonadFail m =>
   32     MonadLog m =>
   33     MonadUserError m =>
   34     Model -> m Model
   35 removeEnumsFromModel =
   36     preCheckForNameReuse >=>
   37     removeEnumsFromModel_LettingEnums >=>
   38     removeEnumsFromModel_GivenEnums   >=>
   39     checkEnums
   40 
   41     where
   42 
   43         -- check if names defined as part of enumerated types are later used as names of top-level or quantified declarations
   44         preCheckForNameReuse model = do
   45             let enumNames = concat [ names | Declaration (LettingDomainDefnEnum _ names) <- mStatements model ]
   46             let redefinedTopLevel = [ name | Declaration (FindOrGiven _ name _) <- mStatements model, name `elem` enumNames ]
   47             let redefinedQuantified = [ name | Generator gen <- universeBi (mStatements model), name@Name{} <- universeBi gen, name `elem` enumNames ]
   48             let redefined = redefinedTopLevel ++ redefinedQuantified
   49             unless (null redefined) $ userErr1 $ vcat
   50                 [ "Members of an enum domain are later redefined as top-level or quantified variables."
   51                 , "Check:" <+> prettyList id "," redefined
   52                 ]
   53             return model
   54 
   55         removeEnumsFromModel_LettingEnums model = do
   56             (statements', ( enumDomainNames :: [(Name, Domain () Expression)]
   57                           , nameToIntMapping_ :: [(Name, (Name, Integer))]
   58                           )) <-
   59                 flip runStateT ([], []) $ forM (mStatements model) $ \ st ->
   60                     case st of
   61                         Declaration (LettingDomainDefnEnum ename@(Name enameText) names) -> do
   62                             namesBefore <- gets (map fst . snd)
   63                             let outDomain = mkDomainIntBTagged (TagEnum enameText)
   64                                                 (fromIntWithTag 1 (TagEnum enameText))
   65                                                 (fromIntWithTag (genericLength names) (TagEnum enameText))
   66                             case names `intersect` namesBefore of
   67                                 [] -> modify ( ( [(ename, outDomain)]
   68                                              , zip names (zip (cycle [ename]) allNats)
   69                                              ) `mappend` )
   70                                 repeated -> userErr1 $ vcat
   71                                     [ "Some members of this enum domain (" <> pretty ename <> ") seem to be defined"
   72                                     , "as part of other enum domains."
   73                                     , "Repeated:" <+> prettyList id "," repeated
   74                                     , "While working on domain:" <+> pretty st
   75                                     ]
   76                             return (Declaration (Letting ename (Domain outDomain)))
   77                         _ -> return st
   78 
   79             let nameToIntMapping = M.fromList nameToIntMapping_
   80 
   81             let
   82                 onX :: Monad m => Expression -> m Expression
   83                 onX (Reference nm Nothing)
   84                     | Just (Name ename, i) <- M.lookup nm nameToIntMapping
   85                     = return (fromIntWithTag i (TagEnum ename))
   86                 onX p = return p
   87 
   88                 onD :: MonadFail m => Domain () Expression -> m (Domain () Expression)
   89                 onD (DomainEnum nm@(Name nmText) (Just ranges) _)
   90                     | Just _ <- lookup nm enumDomainNames
   91                     = DomainInt (TagEnum nmText) <$> mapM (mapM (nameToX nameToIntMapping)) ranges
   92                 onD (DomainEnum nm Nothing _)
   93                     | Just d <- lookup nm enumDomainNames
   94                     = return (DomainReference nm (Just d))
   95                 onD (DomainReference nm Nothing)
   96                     | Just d <- lookup nm enumDomainNames
   97                     = return (DomainReference nm (Just d))
   98                 onD p = return p
   99 
  100             statements'' <- (transformBiM onD >=> transformBiM onX) statements'
  101             return model { mStatements = statements'' }
  102 
  103         removeEnumsFromModel_GivenEnums model = do
  104             (statements', enumDomainNames) <-
  105                 flip runStateT [] $ forM (mStatements model) $ \ st ->
  106                     case st of
  107                         Declaration (GivenDomainDefnEnum name@(Name nameText)) -> do
  108                             let nameS      = name `mappend` "_EnumSize"
  109                             let outDomainS = DomainInt (TagEnum nameText) []
  110                             let outDomain  = mkDomainIntBTagged (TagEnum nameText)
  111                                                 (fromIntWithTag 1 (TagEnum nameText))
  112                                                 (Reference nameS (Just (Alias (Domain outDomainS))))
  113                             modify ([(name, outDomain)] `mappend`)
  114                             return [ Declaration (FindOrGiven Given nameS         outDomainS)
  115                                    , Declaration (Letting           name  (Domain outDomain))
  116                                    ]
  117                         _ -> return [st]
  118 
  119             let
  120                 onD :: Domain () Expression -> Domain () Expression
  121                 onD (DomainEnum nm@(Name nmText) (Just ranges) _)
  122                     | Just _ <- lookup nm enumDomainNames
  123                     = DomainInt (TagEnum nmText) ranges
  124                 onD (DomainEnum nm Nothing _)
  125                     | Just d <- lookup nm enumDomainNames
  126                     = DomainReference nm (Just d)
  127                 onD (DomainReference nm Nothing)
  128                     | Just d <- lookup nm enumDomainNames
  129                     = DomainReference nm (Just d)
  130                 onD p = p
  131 
  132             let model' = model { mStatements = concat statements'
  133                                     |> transformBi onD
  134                                }
  135 
  136             logDebug $ "Recording enumGivens:" <+> prettyList id "," (map fst enumDomainNames)
  137 
  138             return model'
  139 
  140         checkEnums model = do
  141             let
  142                 leftovers :: [Domain () Expression]
  143                 leftovers = [ d | d@DomainEnum{} <- universeBi (mStatements model) ]
  144             unless (null leftovers) $ bug $ vcat
  145                 $ "Could not remove some enum domains:"
  146                 : map (nest 4 . pretty) leftovers
  147             return model

  150 removeEnumsFromParam
  151     :: (MonadFail m, MonadUserError m)
  152     => Model -> Model -> m (Model, Model)
  153 removeEnumsFromParam model param = do
  154     let allStatements = map (False,) (map Declaration (miEnumLettings (mInfo model)))
  155                      ++ map (True,)  (mStatements param)
  156 
  157     (statements', (enumDomainNames_, nameToIntMapping_)) <-
  158         flip runStateT ([], []) $ forM allStatements $ \ (keep,st) ->
  159             case st of
  160                 Declaration (LettingDomainDefnEnum ename@(Name enameText) names) -> do
  161                     namesBefore <- gets (map fst . snd)
  162                     let outDomain = mkDomainIntBTagged (TagEnum enameText)
  163                                         (fromIntWithTag 1 (TagEnum enameText))
  164                                         (fromIntWithTag (genericLength names) (TagEnum enameText))
  165                     case names `intersect` namesBefore of
  166                         [] -> modify ( ( [(ename, outDomain)]
  167                                      , zip names (zip (cycle [ename]) allNats)
  168                                      ) `mappend` )
  169                         repeated -> userErr1 $ vcat
  170                             [ "Some members of this enum domain (" <> pretty ename <> ") seem to be defined"
  171                             , "as part of other enum domains."
  172                             , "Repeated:" <+> prettyList id "," repeated
  173                             , "While working on domain:" <+> pretty st
  174                             ]
  175                     return (Just (Declaration (Letting ename (Domain outDomain))))
  176                 _ -> return (if keep then Just st else Nothing)
  177 
  178     let enumDomainNames = M.fromList enumDomainNames_
  179     let nameToIntMapping = M.fromList nameToIntMapping_
  180 
  181     let
  182         onX :: Monad m => Expression -> m Expression
  183         onX (Reference nm Nothing)
  184             | Just (Name ename, i) <- M.lookup nm nameToIntMapping
  185             = return (fromIntWithTag i (TagEnum ename))
  186         onX p = return p
  187 
  188         onD :: MonadFail m => Domain () Expression -> m (Domain () Expression)
  189         onD (DomainEnum nm@(Name nmText) (Just ranges) _)
  190             | Just _ <- M.lookup nm enumDomainNames
  191             = DomainInt (TagEnum nmText) <$> mapM (mapM (nameToX nameToIntMapping)) ranges
  192         onD (DomainEnum nm Nothing _)
  193             | Just d <- M.lookup nm enumDomainNames
  194             = return (DomainReference nm (Just d))
  195         onD (DomainReference nm Nothing)
  196             | Just d <- M.lookup nm enumDomainNames
  197             = return (DomainReference nm (Just d))
  198         onD p = return p
  199 
  200     let param' = param { mStatements = catMaybes statements' }
  201     let f = transformBiM onD >=> transformBiM onX
  202     (,) <$> f model <*> f param'

  206 --   Converting integers back to enum constants.

  209 addEnumsAndUnnamedsBack
  210     :: ( Pretty r, Pretty x )
  211     => [Name]                               -- unnamed types
  212     -> M.HashMap (Integer, Name) Constant   -- a lookup table for enums
  213     -> Domain r x                           -- the domain we are working on
  214     -> Constant                             -- the constant with ints in place of enums & unnameds
  215     -> Constant                             -- the constant with enums & unnameds again
  216 addEnumsAndUnnamedsBack unnameds ctxt = helper
  217 
  218     where
  219 
  220         helper domain constant = case (domain, constant) of
  221 
  222             (_, TypedConstant c _) -> helper domain c
  223 
  224             (_, c@ConstantUndefined{}) -> c
  225 
  226             (DomainBool  , c) -> c
  227             (DomainIntE{}, c) -> c
  228             (DomainInt{} , c) -> c
  229 
  230             (DomainEnum      ename _ _, ConstantInt _ i) ->
  231                 fromMaybe (bug $ "addEnumsAndUnnamedsBack 1:" <+> pretty (i, ename))
  232                           (M.lookup (i, ename) ctxt)
  233 
  234             (DomainReference ename _  , ConstantInt _ i) ->
  235                 if ename `elem` unnameds
  236                     then ConstantEnum ename [] (mconcat [ename, "_", Name (T.pack (show i))])
  237                     else bug $ "addEnumsAndUnnamedsBack Unnamed:" <++> vcat [ "domain  :" <+> pretty domain
  238                                                                             , "constant:" <+> pretty constant
  239                                                                             ]
  240 
  241             (DomainTuple ds, viewConstantTuple -> Just cs) ->
  242                 ConstantAbstract $ AbsLitTuple
  243                     [ helper d c
  244                     | (d,c) <- zip ds cs ]
  245 
  246             (DomainRecord (sortOn fst -> ds), viewConstantRecord -> Just cs) ->
  247                 ConstantAbstract $ AbsLitRecord
  248                     [ (n, helper d c)
  249                     | ((n,d),(_,c)) <- zip ds cs ]
  250 
  251             (DomainVariant ds, viewConstantVariant -> Just (t, n, c)) ->
  252                 case lookup n ds of
  253                     Nothing -> bug $ "addEnumsAndUnnamedsBack Variant:" <++> vcat [ "domain  :" <+> pretty domain
  254                                                                                   , "constant:" <+> pretty constant
  255                                                                                   ]
  256                     Just d  -> ConstantAbstract $ AbsLitVariant t n (helper d c)
  257 
  258             (DomainMatrix _ inner, viewConstantMatrix -> Just (index, vals)) ->
  259                 ConstantAbstract $ AbsLitMatrix index $ map (helper inner) vals
  260 
  261             (DomainSet _ _ inner, viewConstantSet -> Just vals) ->
  262                 ConstantAbstract $ AbsLitSet $ map (helper inner) vals
  263 
  264             (DomainMSet _ _ inner, viewConstantMSet -> Just vals) ->
  265                 ConstantAbstract $ AbsLitMSet $ map (helper inner) vals
  266 
  267             (DomainFunction _ _ fr to, viewConstantFunction -> Just vals) ->
  268                 ConstantAbstract $ AbsLitFunction
  269                     [ (helper fr a, helper to b)
  270                     | (a,b) <- vals ]
  271 
  272             (DomainSequence _ _ inner, viewConstantSequence -> Just vals) ->
  273                 ConstantAbstract $ AbsLitSequence $ map (helper inner) vals
  274 
  275             (DomainRelation _ _ inners, viewConstantRelation -> Just vals) ->
  276                 ConstantAbstract $ AbsLitRelation
  277                     [ [ helper d c | (d,c) <- zip inners line ]
  278                     | line <- vals ]
  279 
  280             (DomainPartition _ _ inner, viewConstantPartition -> Just vals) ->
  281                 ConstantAbstract $ AbsLitPartition
  282                     [ [ helper inner c | c <- line ]
  283                     | line <- vals ]
  284 
  285             _ -> bug ("addEnumsAndUnnamedsBack 3:" <++> vcat [ "domain  :" <+> pretty domain
  286                                                              , "constant:" <+> pretty constant
  287                                                              , "domain  :" <+> pretty (show domain)
  288                                                              , "constant:" <+> pretty (show constant)
  289                                                              ])
  290 
  291 -- first Name is the value, the second Name is the name of the enum domain
  292 nameToX :: MonadFail m => M.HashMap Name (Name, Integer) -> Expression -> m Expression
  293 nameToX nameToIntMapping (Reference nm _) = case M.lookup nm nameToIntMapping of
  294     Nothing -> fail (pretty nm <+> "is used in a domain, but it isn't a member of the enum domain.")
  295     Just (Name ename, i)  -> return (fromIntWithTag i (TagEnum ename))
  296     Just (ename, i) -> bug $ "nameToX, nm:" <+> vcat [pretty (show ename), pretty i]
  297 nameToX _ x = return x
