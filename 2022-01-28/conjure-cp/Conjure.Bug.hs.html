    4 module Conjure.Bug
    5     ( bug
    6     , bugFail, bugFailT
    7     ) where
    8 
    9 import Conjure.Prelude
   10 import Conjure.RepositoryVersion ( repositoryVersion )
   11 import Conjure.Language.Pretty
   12 
   13 #if __GLASGOW_HASKELL__ >= 800
   14 -- base
   15 import GHC.Stack ( HasCallStack )
   16 -- call this function instead of "error"
   17 bug :: HasCallStack => Doc -> a
   18 #else
   19 bug :: Doc -> a
   20 #endif
   21 bug message = error $ unlines
   22     [ "This should never happen, sorry!"
   23     , ""
   24     , "However, it did happen, so it must be a bug. Please report it to us!"
   25     , ""
   26     , "Conjure is actively maintained, we will get back to you as soon as possible."
   27     , "You can help us by providing a minimal failing example."
   28     , ""
   29     , "Also include the repository version for this build: " ++ repositoryVersion
   30     , ""
   31     , "Issue tracker: http://github.com/conjure-cp/conjure/issues"
   32     , "", "" , renderNormal message
   33     ]
   34 
   35 bugFail :: Doc -> Either Doc a -> a
   36 bugFail loc (Left err) = bug (vcat ["BUGFAIL at" <+> loc, err])
   37 bugFail _   (Right x) = x
   38 
   39 bugFailT :: Monad m => Doc -> ExceptT m a -> m a
   40 bugFailT loc comp = do
   41     res <- runExceptT comp
   42     case res of
   43         Left err -> bug (vcat ["BUGFAIL at" <+> loc, err])
   44         Right x  -> return x
   45 
   46 instance MonadFail IO where
   47     fail msg = bug (vcat ["IO Error", msg])
