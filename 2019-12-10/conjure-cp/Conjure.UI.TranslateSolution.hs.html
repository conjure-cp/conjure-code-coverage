   23 translateSolution ::
   24     MonadFail m =>
   25     MonadLog m =>
   26     NameGen m =>
   27     EnumerateDomain m =>
   28     (?typeCheckerMode :: TypeCheckerMode) =>
   29     Model ->      -- eprime model
   30     Model ->      -- essence param
   31     Model ->      -- eprime solution
   32     m Model       -- essence solution
   33 
   34 translateSolution eprimeModel essenceParam' eprimeSolution = do
   35 
   36     eprimeParam <- translateParameter eprimeModel essenceParam'
   37     (_, essenceParam) <- removeEnumsFromParam eprimeModel essenceParam'
   38 
   39     let eprimeLettingsForEnums =
   40             [ (nm, fromInt (genericLength vals))
   41             | nm1                                          <- eprimeModel |> mInfo |> miEnumGivens
   42             , Declaration (LettingDomainDefnEnum nm2 vals) <- essenceParam' |> mStatements
   43             , nm1 == nm2
   44             , let nm = nm1 `mappend` "_EnumSize"
   45             ]
   46 
   47     let eprimeLettings0 = extractLettings essenceParam ++
   48                           extractLettings eprimeParam ++
   49                           extractLettings eprimeSolution ++
   50                           extractLettings eprimeModel ++
   51                           (eprimeModel |> mInfo |> miLettings) ++
   52                           eprimeLettingsForEnums
   53     let essenceFindNames = eprimeModel |> mInfo |> miFinds
   54     let essenceFinds     = eprimeModel |> mInfo |> miRepresentations
   55                                        |> filter (\ (n,_) -> n `elem` essenceFindNames )
   56 
   57     -- the right hand sides of these lettings may be expressions (as opposed to constants)
   58     -- that will make evaluation unnecessarily slower
   59     let eprimeLettings =
   60             [ (name, maybe val Constant (e2c val))
   61             | (name, val) <- eprimeLettings0
   62             ]
   63 
   64     eprimeLettings' <- forM eprimeLettings $ \ (name, val) -> do
   65         constant <- instantiateExpression eprimeLettings val
   66         return (name, constant)
   67 
   68     essenceFinds' <- forM essenceFinds $ \ (name, dom) -> do
   69         constant <- instantiateDomain eprimeLettings dom
   70         return (name, constant)
   71 
   72     essenceLettings <- forM essenceFinds' $ \ (name, domain) -> do
   73         (_, constant) <- up eprimeLettings' (name, domain)
   74         let origDomain = eprimeModel
   75                 |> mInfo |> miOriginalDomains
   76                 |> lookup name
   77                 |> fromMaybe (bug ("Missing original domain for:" <+> pretty name))
   78         return (name, origDomain, constant)
   79 
   80     let
   81         intToEnumConstant :: M.HashMap (Integer, Name) Constant
   82         intToEnumConstant = M.fromList $ concat
   83             [ [ ((i,ename), ConstantEnum ename vals v)
   84               | (i,v) <- zip allNats vals
   85               ]
   86             | Declaration (LettingDomainDefnEnum ename vals)
   87                     <- mStatements essenceParam'
   88                     ++ eprimeModel |> mInfo |> miEnumLettings |> map Declaration
   89             ]
   90 
   91     let
   92         unnameds :: [(Name, Expression)]
   93         unnameds = eprimeModel |> mInfo |> miUnnameds
   94 
   95     unnamedsAsEnumDomains <- forM unnameds $ \ (n, s') -> do
   96         s <- instantiateExpression eprimeLettings s'
   97         case s of
   98             ConstantInt _ size -> return $
   99                 let nms = [ mconcat [n, "_", Name (T.pack (show i))]
  100                           | i <- [1 .. size]
  101                           ]
  102                 in  Declaration (LettingDomainDefnEnum n nms)
  103             _ -> fail $ vcat [ "Expecting an integer value for" <+> pretty n
  104                              , "But got:" <+> pretty s
  105                              ]
  106 
  107     let outStmts =
  108             unnamedsAsEnumDomains ++
  109             sortNub
  110                 [ Declaration (Letting n (Constant (normaliseConstant y)))
  111                 | (n, d, x) <- essenceLettings
  112                 , let y = addEnumsAndUnnamedsBack
  113                                 (map fst unnameds)
  114                                 intToEnumConstant
  115                                 d x
  116                 ]
  117 
  118     let undefs = [ u | u@ConstantUndefined{} <- universeBi outStmts ]
  119 
  120     if null undefs
  121         then return def { mStatements = outStmts }
  122         else bug $ vcat
  123             [ "Undefined values in the output:" <++> vcat (map pretty undefs)
  124             , ""
  125             , "Complete output would have been the following."
  126             , ""
  127             , pretty $ def { mStatements = outStmts }
  128             ]
