   70         structuralCons :: TypeOf_Structural m
   71         structuralCons f downX1 (DomainSequence Sequence_ExplicitBounded
   72                                         (SequenceAttr (SizeAttr_Size size) jectivityAttr) innerDomain) = do
   73             let
   74                 injectiveCons :: Expression -> m [Expression]
   75                 injectiveCons values = do
   76                     innerType <- typeOfDomain innerDomain
   77                     case innerType of
   78                       TypeInt _ -> do
   79                             return $ return $ -- list
   80                                 [essence| allDiff(&values) |]
   81 
   82                       _ ->  do
   83                             (iPat, i) <- quantifiedVar
   84                             (jPat, j) <- quantifiedVar
   85                             return $ return $ -- list
   86                                 [essence|
   87                                     and([ &values[&i] != &values[&j]
   88                                         | &iPat : int(1..&size)
   89                                         , &jPat : int(1..&size)
   90                                         , &i .< &j
   91                                         ])
   92                                 |]
   93 
   94                 surjectiveCons :: Expression -> m [Expression]
   95                 surjectiveCons values = do
   96                     (iPat, i) <- quantifiedVar
   97                     (jPat, j) <- quantifiedVar
   98                     return $ return $ -- list
   99                         [essence|
  100                             forAll &iPat : &innerDomain .
  101                                 exists &jPat : int(1..&size) .
  102                                     &values[&j] = &i
  103                         |]
  104 
  105                 jectivityCons :: Expression -> m [Expression]
  106                 jectivityCons values = case jectivityAttr of
  107                     JectivityAttr_None       -> return []
  108                     JectivityAttr_Injective  -> injectiveCons  values
  109                     JectivityAttr_Surjective -> surjectiveCons values
  110                     JectivityAttr_Bijective  -> (++) <$> injectiveCons  values
  111                                                      <*> surjectiveCons values
  112 
  113             let innerStructuralCons values = do
  114                     (iPat, i) <- quantifiedVarOverDomain [essenceDomain| int(1..&size) |]
  115                     let activeZone b = [essence| forAll &iPat : int(1..&size) . &b |]
  116 
  117                     -- preparing structural constraints for the inner guys
  118                     innerStructuralConsGen <- f innerDomain
  119 
  120                     let inLoop = [essence| &values[&i] |]
  121                     outs <- innerStructuralConsGen inLoop
  122                     return (map activeZone outs)
  123 
  124             return $ \ sequ -> do
  125                 refs <- downX1 sequ
  126                 case refs of
  127                     [_marker,values] ->
  128                         concat <$> sequence
  129                             [ jectivityCons       values
  130                             , innerStructuralCons values
  131                             ]
  132                     _ -> na "{structuralCons} ExplicitBounded"
  133         structuralCons f downX1 (DomainSequence Sequence_ExplicitBounded (SequenceAttr sizeAttr jectivityAttr) innerDomain) = do
  134             maxSize <- getMaxSize sizeAttr
  135             let injectiveCons marker values = do
  136                     (iPat, i) <- quantifiedVar
  137                     (jPat, j) <- quantifiedVar
  138                     return $ return $ -- list
  139                         [essence|
  140                             and([ &values[&i] != &values[&j]
  141                                 | &iPat : int(1..&maxSize)
  142                                 , &jPat : int(1..&maxSize)
  143                                 , &i .< &j
  144                                 , &i <= &marker
  145                                 , &j <= &marker
  146                                 ])
  147                         |]
  148 
  149             let surjectiveCons marker values = do
  150                     (iPat, i) <- quantifiedVar
  151                     (jPat, j) <- quantifiedVar
  152                     return $ return $ -- list
  153                         [essence|
  154                             forAll &iPat : &innerDomain .
  155                                 exists &jPat : int(1..&maxSize) .
  156                                     (&j <= &marker) /\ &values[&j] = &i
  157                         |]
  158 
  159             let jectivityCons marker values = case jectivityAttr of
  160                     JectivityAttr_None       -> return []
  161                     JectivityAttr_Injective  -> injectiveCons  marker values
  162                     JectivityAttr_Surjective -> surjectiveCons marker values
  163                     JectivityAttr_Bijective  -> (++) <$> injectiveCons  marker values
  164                                                      <*> surjectiveCons marker values
  165 
  166             let dontCareAfterMarker marker values = do
  167                     (iPat, i) <- quantifiedVar
  168                     return $ return $ -- list
  169                         [essence|
  170                             forAll &iPat : int(1..&maxSize) . &i > &marker ->
  171                                 dontCare(&values[&i])
  172                         |]
  173 
  174             let innerStructuralCons marker values = do
  175                     (iPat, i) <- quantifiedVarOverDomain [essenceDomain| int(1..&maxSize) |]
  176                     let activeZone b = [essence| forAll &iPat : int(1..&maxSize) . &i <= &marker -> &b |]
  177 
  178                     -- preparing structural constraints for the inner guys
  179                     innerStructuralConsGen <- f innerDomain
  180 
  181                     let inLoop = [essence| &values[&i] |]
  182                     outs <- innerStructuralConsGen inLoop
  183                     return (map activeZone outs)
  184 
  185             return $ \ sequ -> do
  186                 refs <- downX1 sequ
  187                 case refs of
  188                     [marker,values] ->
  189                         concat <$> sequence
  190                             [ dontCareAfterMarker marker values
  191                             , return (mkSizeCons sizeAttr marker)
  192                             , jectivityCons       marker values
  193                             , innerStructuralCons marker values
  194                             ]
  195                     _ -> na "{structuralCons} ExplicitBounded"
  196 
  197         structuralCons _ _ _ = na "{structuralCons} ExplicitBounded"

  199         downC :: TypeOf_DownC m
  200         downC ( name
  201               , domain@(DomainSequence _ (SequenceAttr (SizeAttr_Size size) _) innerDomain)
  202               , ConstantAbstract (AbsLitSequence constants)
  203               ) =
  204             return $ Just
  205                 [ ( nameMarker domain name
  206                   , DomainInt TagInt [RangeBounded size size]
  207                   , ConstantInt TagInt (genericLength constants)
  208                   )
  209                 , ( nameValues domain name
  210                   , DomainMatrix (DomainInt TagInt [RangeBounded 1 size]) innerDomain
  211                   , ConstantAbstract $ AbsLitMatrix (DomainInt TagInt [RangeBounded 1 size]) constants
  212                   )
  213                 ]
  214         downC ( name
  215               , domain@(DomainSequence _ (SequenceAttr sizeAttr _) innerDomain)
  216               , ConstantAbstract (AbsLitSequence constants)
  217               ) = do
  218             maxSize <- getMaxSize sizeAttr
  219             let indexDomain i = mkDomainIntB (fromInt i) maxSize
  220             maxSizeInt <-
  221                 case maxSize of
  222                     ConstantInt _ x -> return x
  223                     _ -> fail $ vcat
  224                             [ "Expecting an integer for the maxSize attribute."
  225                             , "But got:" <+> pretty maxSize
  226                             , "When working on:" <+> pretty name
  227                             , "With domain:" <+> pretty domain
  228                             ]
  229             z <- zeroVal innerDomain
  230             let zeroes = replicate (fromInteger (maxSizeInt - genericLength constants)) z
  231             return $ Just
  232                 [ ( nameMarker domain name
  233                   , defRepr (indexDomain 0)
  234                   , ConstantInt TagInt (genericLength constants)
  235                   )
  236                 , ( nameValues domain name
  237                   , DomainMatrix (indexDomain 1) innerDomain
  238                   , ConstantAbstract $ AbsLitMatrix (indexDomain 1) (constants ++ zeroes)
  239                   )
  240                 ]
  241         downC (name, domain, constant) = na $ vcat [ "{downC} ExplicitBounded"
  242                                                    , "name    :" <+> pretty name
  243                                                    , "domain  :" <+> pretty domain
  244                                                    , "constant:" <+> pretty constant
  245                                                    ]

  247         up :: TypeOf_Up m
  248         up ctxt (name, domain) =
  249             case (lookup (nameMarker domain name) ctxt, lookup (nameValues domain name) ctxt) of
  250                 (Just marker, Just constantMatrix) ->
  251                     case marker of
  252                         ConstantInt _ card ->
  253                             case viewConstantMatrix constantMatrix of
  254                                 Just (_, vals) ->
  255                                     return (name, ConstantAbstract (AbsLitSequence (genericTake card vals)))
  256                                 _ -> fail $ vcat
  257                                         [ "Expecting a matrix literal for:" <+> pretty (nameValues domain name)
  258                                         , "But got:" <+> pretty constantMatrix
  259                                         , "When working on:" <+> pretty name
  260                                         , "With domain:" <+> pretty domain
  261                                         ]
  262                         _ -> fail $ vcat
  263                                 [ "Expecting an integer literal for:" <+> pretty (nameMarker domain name)
  264                                 , "But got:" <+> pretty marker
  265                                 , "When working on:" <+> pretty name
  266                                 , "With domain:" <+> pretty domain
  267                                 ]
  268                 (Nothing, _) -> fail $ vcat $
  269                     [ "(in Sequence ExplicitBounded up 1)"
  270                     , "No value for:" <+> pretty (nameMarker domain name)
  271                     , "When working on:" <+> pretty name
  272                     , "With domain:" <+> pretty domain
  273                     ] ++
  274                     ("Bindings in context:" : prettyContext ctxt)
  275                 (_, Nothing) -> fail $ vcat $
  276                     [ "(in Sequence ExplicitBounded up 2)"
  277                     , "No value for:" <+> pretty (nameValues domain name)
  278                     , "When working on:" <+> pretty name
  279                     , "With domain:" <+> pretty domain
  280                     ] ++
  281                     ("Bindings in context:" : prettyContext ctxt)

  283         symmetryOrdering :: TypeOf_SymmetryOrdering m
  284         symmetryOrdering innerSO downX1 inp domain = do
  285             [marker, values] <- downX1 inp
  286             Just [_, (_, DomainMatrix index inner)] <- downD ("SO", domain)
  287             (iPat, i) <- quantifiedVar
  288             soValues <- innerSO downX1 [essence| &values[&i] |] inner
  289             return
  290                 [essence|
  291                     ( &marker
  292                     , [ &soValues
  293                       | &iPat : &index
  294                       ]
  295                     )
  296                 |]
