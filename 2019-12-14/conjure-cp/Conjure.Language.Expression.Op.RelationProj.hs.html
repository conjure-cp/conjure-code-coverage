   21 instance (TypeOf x, Pretty x) => TypeOf (OpRelationProj x) where
   22     typeOf p@(OpRelationProj r xs) = do
   23         tyR <- typeOf r
   24         case (tyR, xs) of
   25             (TypeRelation ts', _) -> do
   26                 let xs' = catMaybes xs
   27                 if length xs == length xs'
   28                     then do -- all Just's
   29                         let loop [] [] = return TypeBool
   30                             loop (i:is) (t:ts) = do
   31                                 tyI <- typeOf i
   32                                 if typesUnify [tyI,t]
   33                                     then loop is ts
   34                                     else raiseTypeError $ "(relation projection)" <+> pretty p
   35                             loop _ _ = raiseTypeError $ "(relation projection)" <+> pretty p
   36                         loop xs' ts'
   37                     else do
   38                         let loop [] [] = return []
   39                             loop (Nothing:is) (t:ts) = (t:) <$> loop is ts
   40                             loop (Just i :is) (t:ts) = do
   41                                 tyI <- typeOf i
   42                                 if typesUnify [tyI,t]
   43                                     then loop is ts
   44                                     else raiseTypeError $ "(relation projection)" <+> pretty p
   45                             loop _ _ = raiseTypeError $ "(relation projection)" <+> pretty p
   46                         TypeRelation <$> loop xs ts'
   47             _ -> raiseTypeError $ "(relation projection)" <+> pretty p

   52 instance Pretty x => Pretty (OpRelationProj x) where
