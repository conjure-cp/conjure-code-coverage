   24 -- | Refine (down) an expression (X), one level (1).
   25 downX1 ::
   26     MonadFail m =>
   27     NameGen m =>
   28     EnumerateDomain m =>
   29     (?typeCheckerMode :: TypeCheckerMode) =>
   30     Expression -> m [Expression]
   31 downX1 (Constant x) = onConstant x
   32 downX1 (AbstractLiteral x) = onAbstractLiteral x
   33 downX1 (Reference x (Just refTo)) = onReference x refTo
   34 downX1 (Op x) = onOp x
   35 downX1 (Comprehension body stmts) = do
   36     xs <- downX1 body
   37     return [Comprehension x stmts | x <- xs]
   38 downX1 x@WithLocals{} = fail ("downX1:" <++> pretty (show x))
   39 downX1 x = bug ("downX1:" <++> pretty (show x))

   42 -- | Refine (down) an expression (X), all the way.
   43 downX ::
   44     NameGen m =>
   45     EnumerateDomain m =>
   46     (?typeCheckerMode :: TypeCheckerMode) =>
   47     Expression -> m [Expression]
   48 downX x = do
   49     res <- runMaybeT $ downX1 x
   50     case res of
   51         Nothing -> return [x]
   52         Just [] -> return [x]
   53         Just xs -> concatMapM downX xs

   56 onConstant ::
   57     MonadFail m =>
   58     NameGen m =>
   59     EnumerateDomain m =>
   60     (?typeCheckerMode :: TypeCheckerMode) =>
   61     Constant -> m [Expression]
   62 onConstant (ConstantAbstract (AbsLitTuple xs)) = return (map Constant xs)
   63 onConstant (ConstantAbstract (AbsLitRecord xs)) = return (map (Constant . snd) xs)
   64 onConstant (ConstantAbstract (AbsLitVariant (Just t) n x))
   65     | Just i <- elemIndex n (map fst t)
   66     , let iExpr = fromInt (fromIntegral (i+1))
   67     = return $ iExpr : [ if n == n'
   68                             then Constant x
   69                             else ExpressionMetaVar "zeroVal for variant"
   70                        | (n',_) <- t ]
   71 onConstant (ConstantAbstract (AbsLitMatrix index xs)) = do
   72     yss <- mapM (downX1 . Constant) xs
   73     let indexX = fmap Constant index
   74     return [ AbstractLiteral (AbsLitMatrix indexX ys) | ys <- transpose yss ]
   75 onConstant (TypedConstant c _) = onConstant c
   76 onConstant x = bug ("downX1.onConstant:" <++> pretty (show x))

   78 onAbstractLiteral ::
   79     MonadFail m =>
   80     NameGen m =>
   81     EnumerateDomain m =>
   82     (?typeCheckerMode :: TypeCheckerMode) =>
   83     AbstractLiteral Expression -> m [Expression]
   84 onAbstractLiteral (AbsLitTuple xs) = return xs
   85 onAbstractLiteral (AbsLitRecord xs) = return (map snd xs)
   86 onAbstractLiteral (AbsLitVariant (Just t) n x)
   87     | Just i <- elemIndex n (map fst t)
   88     , let iExpr = fromInt (fromIntegral (i+1))
   89     = return $ iExpr : [ if n == n'
   90                             then x
   91                             else ExpressionMetaVar "zeroVal for variant"
   92                        | (n',_) <- t ]
   93 onAbstractLiteral (AbsLitMatrix index xs) = do
   94     yss <- mapM downX1 xs
   95     return [ AbstractLiteral (AbsLitMatrix index ys) | ys <- transpose yss ]
   96 onAbstractLiteral x = bug ("downX1.onAbstractLiteral:" <++> pretty (show x))

   98 onReference ::
   99     MonadFail m =>
  100     NameGen m =>
  101     EnumerateDomain m =>
  102     (?typeCheckerMode :: TypeCheckerMode) =>
  103     Name -> ReferenceTo -> m [Expression]
  104 onReference nm refTo =
  105     case refTo of
  106         Alias x                   -> downX1 x
  107         InComprehension{}         -> fail ("downX1.onReference.InComprehension:" <++> pretty (show nm))
  108         DeclNoRepr{}              -> fail ("downX1.onReference.DeclNoRepr:"      <++> pretty (show nm))
  109         DeclHasRepr forg _ domain -> downToX1 forg nm domain
  110         RecordField{}             -> fail ("downX1.onReference.RecordField:"     <++> pretty (show nm))
  111         VariantField{}            -> fail ("downX1.onReference.VariantField:"    <++> pretty (show nm))
  112 
  113 onOp ::
  114     MonadFail m =>
  115     NameGen m =>
  116     EnumerateDomain m =>
  117     (?typeCheckerMode :: TypeCheckerMode) =>
  118     Op Expression -> m [Expression]
  119 onOp p@(MkOpIndexing (OpIndexing m i)) = do
  120     ty <- typeOf m
  121     case ty of
  122         TypeMatrix{} -> return ()
  123         TypeList{}   -> return ()
  124         _ -> fail $ "[onOp, not a TypeMatrix or TypeList]" <+> vcat [pretty ty, pretty p]
  125     xs <- downX1 m
  126     let iIndexed x = Op (MkOpIndexing (OpIndexing x i))
  127     return (map iIndexed xs)
  128 onOp (MkOpImage (OpImage (match functionLiteral -> Just (_, xs)) a)) | length xs > 0 = do
  129     vals <- forM xs $ \ (_, value) -> do
  130         ys <- downX1 value
  131         return ys
  132     let keys = map fst xs
  133     let outs = map (zip keys) (transpose vals)
  134     return [ Op $ MkOpImage $ OpImage (AbstractLiteral (AbsLitFunction out)) a
  135            | out <- outs ]
  136 onOp op = fail ("downX1.onOp:" <++> pretty op)

  140 symmetryOrdering ::
  141     MonadFail m =>
  142     NameGen m =>
  143     EnumerateDomain m =>
  144     (?typeCheckerMode :: TypeCheckerMode) =>
  145     Expression -> m Expression
  146 symmetryOrdering inp =
  147     case inp of
  148         -- Constant x -> so_onConstant x
  149         -- AbstractLiteral x
  150         Reference _ (Just refTo) -> do
  151             case refTo of
  152                 Alias x                        -> symmetryOrdering x
  153                 InComprehension{}              -> bug ("symmetryOrdering.InComprehension:" <++> pretty (show inp))
  154                 DeclNoRepr{}                   -> bug ("symmetryOrdering.DeclNoRepr:"      <++> pretty (show inp))
  155                 DeclHasRepr _forg _name domain -> symmetryOrderingDispatch downX1 inp domain
  156                 RecordField{}                  -> bug ("symmetryOrdering.RecordField:"     <++> pretty (show inp))
  157                 VariantField{}                 -> bug ("symmetryOrdering.VariantField:"    <++> pretty (show inp))
  158         Op op -> case op of
  159             MkOpIndexing (OpIndexing m _) -> do
  160                 ty <- typeOf m
  161                 case ty of
  162                     TypeMatrix{} -> return ()
  163                     TypeList{}   -> return ()
  164                     _ -> bug $ "[symmetryOrdering.onOp, not a TypeMatrix or TypeList]" <+> vcat [pretty ty, pretty op]
  165                 mDom <- domainOfR m
  166                 case mDom of
  167                     DomainMatrix _ domainInner -> symmetryOrderingDispatch downX1 inp domainInner
  168                     _ -> bug ("symmetryOrdering, not DomainMatrix:" <++> pretty (show op))
  169             _ -> bug ("symmetryOrdering, unhandled Op:" <++> pretty (show op))

  171         --     xs <- downX1 body
  172         --     return [Comprehension x stmts | x <- xs]
  173         -- x@WithLocals{} -> bug ("downX1:" <++> pretty (show x))
  174         _ -> bug ("symmetryOrdering:" <++> pretty (show inp))
