   57 instance (Pretty a, Pretty b) => Pretty (a, b) where
   58     pretty (a, b) = prettyListDoc parens "," [pretty a, pretty b]

   60 instance (Pretty a, Pretty b, Pretty c) => Pretty (a, b, c) where

   63 instance Pretty a => Pretty (Maybe a) where

   69 (<++>) :: Doc -> Doc -> Doc
   70 a <++> b = hang a 4 b
   71 
   72 -- | For debugging output, truncates the second argument to 5 lines
   73 (<+->) :: Doc -> Doc -> Doc
   74 a <+-> b = a <+> (vcat $ map pretty $ take 5 $ lines $ renderWide $ b)
   75 
   76 prettyList :: Pretty a => (Doc -> Doc) -> Doc -> [a] -> Doc
   77 prettyList wrap punc = prettyListDoc wrap punc . map pretty
   78 
   79 prettyListDoc :: (Doc -> Doc) -> Doc -> [Doc] -> Doc
   80 prettyListDoc wrap punc = wrap . fsep . punctuate punc

   82 parensIf :: Bool -> Doc -> Doc
   83 parensIf = wrapIf parens
   84     where
   85         wrapIf :: (Doc -> Doc) -> Bool -> Doc -> Doc
   86         wrapIf wrap c = if c then wrap else id

  117 instance Pretty JSON.Value where
  118     pretty (Object x) = pretty x
  119     pretty (Array x) = pretty x
  120     pretty (String x) = "\"" <> pretty (T.unpack (T.replace "\"" "\\\"" x)) <> "\""
  121     pretty (Number x) = pretty x
  122     pretty (Bool False) = "false"
  123     pretty (Bool True) = "true"
  124     pretty Null = "null"
  125 
  126 instance Pretty JSON.Object where
  127     pretty = prBraces . fsep . punctuate "," . map f . sortBy (comp `on` fst) . M.toList
  128         where
  129             f (key, Array value)
  130                 | not (any (\ v -> case v of String t -> T.length t < 20 ; _ -> True ) value)
  131                 = pretty (show key) <> ":" <++> prettyArrayVCat value
  132             f (key, value) = pretty (show key) <> ":" <++> pretty value
  133 
  134             keyOrder :: M.HashMap Text Int
  135             keyOrder = M.fromList $
  136                 zip [ "finds", "givens", "enumGivens", "enumLettings", "unnameds"
  137                     , "strategyQ", "strategyA"
  138                     , "trailCompact", "trailVerbose", "trailRewrites"
  139                     , "nameGenState", "nbExtraGivens"
  140                     , "representations", "representationsTree"
  141                     , "originalDomains"
  142                     , "before", "after"
  143                     ] [1..]
  144 
  145             comp a b =
  146                 let preferred = compare <$> M.lookup a keyOrder
  147                                         <*> M.lookup b keyOrder
  148                 in  fromMaybe (compare a b) preferred
  149 
  150 instance Pretty JSON.Array where
  151     pretty = prBrackets . fsep . punctuate "," . map pretty . V.toList
  152 
  153 prettyArrayVCat :: V.Vector Value -> Doc
  154 prettyArrayVCat = prBrackets . vcat . punctuate "," . map pretty . V.toList

  156 instance Pretty Scientific where

  159 logDebugId :: (MonadLog m, Pretty a) => Doc -> a -> m a
  160 logDebugId msg a = logDebug (msg <++> pretty a) >> return a

  162 tracingPretty :: Pretty a => Doc -> a -> a
  163 tracingPretty s a = trace (renderWide $ "tracing" <+> s <> ": " <++> pretty a) a
