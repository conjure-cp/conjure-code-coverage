   21 resolveNamesMulti ::
   22     MonadLog m =>
   23     MonadUserError m =>
   24     NameGen m =>
   25     (?typeCheckerMode :: TypeCheckerMode) =>
   26     [Model] -> m [Model]
   27 resolveNamesMulti = flip evalStateT [] . go
   28     where
   29         go [] = return []
   30         go (m:ms) = (:) <$> resolveNames_ m <*> go ms

   32 resolveNames ::
   33     MonadLog m =>
   34     MonadUserError m =>
   35     NameGen m =>
   36     (?typeCheckerMode :: TypeCheckerMode) =>
   37     Model -> m Model
   38 resolveNames = flip evalStateT [] . resolveNames_
   39 
   40 resolveNames_ ::
   41     MonadLog m =>
   42     MonadState [(Name, ReferenceTo)] m =>
   43     MonadUserError m =>
   44     NameGen m =>
   45     (?typeCheckerMode :: TypeCheckerMode) =>
   46     Model -> m Model
   47 resolveNames_ model = failToUserError $ do
   48     statements <- mapM resolveStatement (mStatements model)
   49     mapM_ check (universeBi statements)
   50     return model { mStatements = toTaggedInt statements }
   51 
   52 -- this is for when a name will shadow an already existing name that is outside of this expression
   53 -- we rename the new names to avoid name shadowing
   54 shadowing ::
   55     MonadFail m =>
   56     MonadState [(Name, ReferenceTo)] m =>
   57     NameGen m =>
   58     Expression -> m Expression
   59 shadowing p@(Comprehension _ is) = do
   60     -- list of names originating from this comprehension
   61     let generators = concat
   62             [ names
   63             | Generator gen <- is
   64             , let pat = generatorPat gen
   65             , let names = [ n | n@Name{} <- universeBi pat ]
   66             ]
   67     ctxt <- gets id
   68     -- a subset of names originating from this comprehension that will shadow already existing names
   69     let shadows = [ g | g <- generators, g `elem` map fst ctxt ]
   70     shadowsNew <- forM shadows $ \ s -> do n <- nextName "shadow" ; return (s,n)
   71     let f n = fromMaybe n (lookup n shadowsNew)
   72     return (transformBi f p)
   73 shadowing p = return p

   76 resolveNamesX ::
   77     MonadFail m =>
   78     MonadUserError m =>
   79     NameGen m =>
   80     (?typeCheckerMode :: TypeCheckerMode) =>
   81     Expression -> m Expression
   82 resolveNamesX x = do
   83     x' <- evalStateT (resolveX x) []
   84     mapM_ check (universe x')
   85     return x'

   88 toTaggedInt :: Data a => a -> a
   89 toTaggedInt = transformBi f
   90     where
   91         f :: Type -> Type
   92         f (TypeEnum (Name nm)) = TypeInt (TagEnum nm)
   93         f ty = ty
   94

   96 check :: MonadFail m => Expression -> m ()
   97 check (Reference nm Nothing) = fail ("Undefined:" <+> pretty nm)
   98 check _ = return ()

  101 resolveStatement ::
  102     MonadFail m =>
  103     MonadState [(Name, ReferenceTo)] m =>
  104     MonadUserError m =>
  105     NameGen m =>
  106     (?typeCheckerMode :: TypeCheckerMode) =>
  107     Statement -> m Statement
  108 resolveStatement st =
  109     case st of
  110         Declaration decl ->
  111             case decl of
  112                 FindOrGiven forg nm dom       -> do
  113                     dom' <- resolveD dom
  114                     modify ((nm, DeclNoRepr forg nm dom' NoRegion) :)
  115                     return (Declaration (FindOrGiven forg nm dom'))
  116                 Letting nm x                  -> do
  117                     x' <- resolveX x
  118                     modify ((nm, Alias x') :)
  119                     return (Declaration (Letting nm x'))
  120                 LettingDomainDefnUnnamed nm x -> do
  121                     x' <- resolveX x
  122                     modify ((nm, Alias (Domain (DomainUnnamed nm x'))) :)
  123                     return (Declaration (LettingDomainDefnUnnamed nm x'))
  124                 LettingDomainDefnEnum (Name ename) nms -> do
  125                     modify ( [ (nm, Alias (Constant (ConstantInt (TagEnum ename) i)))
  126                              | (nm, i) <- zip nms [1..]
  127                              ] ++)
  128                     return st
  129                 LettingDomainDefnEnum{} -> bug "resolveStatement, Name"
  130                 GivenDomainDefnEnum{}       -> return st             -- ignoring
  131         SearchOrder xs -> SearchOrder <$> mapM resolveSearchOrder xs
  132         SearchHeuristic nm -> do
  133             let allowed = ["static", "sdf", "conflict", "srf", "ldf", "wdeg", "domoverwdeg"]
  134             if nm `elem` allowed
  135                 then return (SearchHeuristic nm)
  136                 else userErr1 $ vcat [ "Invalid heuristic:" <+> pretty nm
  137                                      , "Allowed values are:" <+> prettyList id "," allowed
  138                                      ]
  139         Where xs -> Where <$> mapM resolveX xs
  140         Objective obj x -> Objective obj <$> resolveX x
  141         SuchThat xs -> SuchThat <$> mapM resolveX xs

  144 resolveSearchOrder ::
  145     MonadFail m =>
  146     MonadState [(Name, ReferenceTo)] m =>
  147     MonadUserError m =>
  148     NameGen m =>
  149     (?typeCheckerMode :: TypeCheckerMode) =>
  150     SearchOrder -> m SearchOrder
  151 resolveSearchOrder (BranchingOn nm) = do
  152     ctxt <- gets id
  153     mval <- gets (lookup nm)
  154     case mval of
  155         Nothing -> userErr1 $ vcat $ ("Undefined reference:" <+> pretty nm)
  156                                    : ("Bindings in context:" : prettyContext ctxt)
  157         Just{}  -> return (BranchingOn nm)
  158 resolveSearchOrder (Cut x) =
  159     let f Find = CutFind
  160         f forg = forg
  161     in  Cut . transformBi f <$> resolveX x

  164 resolveX ::
  165     MonadFail m =>
  166     MonadState [(Name, ReferenceTo)] m =>
  167     MonadUserError m =>
  168     NameGen m =>
  169     (?typeCheckerMode :: TypeCheckerMode) =>
  170     Expression -> m Expression
  171 resolveX (Reference nm Nothing) = do
  172     ctxt <- gets id
  173     mval <- gets (lookup nm)
  174     case mval of
  175         Nothing -> userErr1 $ vcat $ ("Undefined reference:" <+> pretty nm)
  176                                    : ("Bindings in context:" : prettyContext ctxt)
  177         Just r  -> return (Reference nm (Just r))
  178 
  179 resolveX p@(Reference nm (Just refto)) = do             -- this is for re-resolving
  180     mval <- gets (lookup nm)
  181     case mval of
  182         Nothing -> return p                             -- hence, do not fail if not in the context
  183         Just DeclNoRepr{}                               -- if the newly found guy doesn't have a repr
  184             | DeclHasRepr{} <- refto                    -- but the old one did, do not update
  185             -> return p
  186         Just (DeclNoRepr forg_ nm_ dom_ _)              -- if the newly found guy doesn't have a repr
  187             | DeclNoRepr _ _ _ region <- refto          -- and the old one didn't have one either
  188                                                         -- preserve the region information
  189             -> return (Reference nm (Just (DeclNoRepr forg_ nm_ dom_ region)))
  190         Just (Alias r) -> do
  191             r' <- resolveX r
  192             return (Reference nm (Just (Alias r')))
  193         Just r ->
  194             return (Reference nm (Just r))
  195 
  196 resolveX (AbstractLiteral lit) = AbstractLiteral <$> resolveAbsLit lit
  197 
  198 resolveX (Domain x) = Domain <$> resolveD x
  199 
  200 resolveX p@Comprehension{} = scope $ do
  201     p' <- shadowing p
  202     case p' of
  203         Comprehension x is -> do
  204             is' <- forM is $ \ i -> case i of
  205                 Generator gen -> do
  206                     (gen', refto) <- case gen of
  207                         GenDomainNoRepr pat dom -> do
  208                             dom' <- resolveD dom
  209                             let gen'' = GenDomainNoRepr pat dom'
  210                             return
  211                                 ( gen''
  212                                 , case pat of
  213                                     Single nm' -> DeclNoRepr Quantified nm' dom' NoRegion
  214                                     _ -> InComprehension gen''
  215                                 )
  216                         GenDomainHasRepr nm dom -> do
  217                             dom' <- resolveD dom
  218                             return
  219                                 ( GenDomainHasRepr nm dom'
  220                                 , DeclHasRepr Quantified nm dom'
  221                                 )
  222                         GenInExpr pat expr -> do
  223                             expr' <- resolveX expr
  224                             let gen'' = GenInExpr pat expr'
  225                             return ( gen'' , InComprehension gen'' )
  226                     forM_ (universeBi (generatorPat gen)) $ \ nm ->
  227                         modify ((nm, refto) :)
  228                     return (Generator gen')
  229                 Condition y -> Condition <$> resolveX y
  230                 ComprehensionLetting pat expr -> do
  231                     expr' <- resolveX expr
  232                     resolveAbsPat p pat expr'
  233                     return (ComprehensionLetting pat expr')
  234             x' <- resolveX x
  235             return (Comprehension x' is')
  236         _ -> bug "NameResolution.resolveX.shadowing"
  237 
  238 resolveX (WithLocals body (AuxiliaryVars locals)) = scope $ do
  239     locals' <- mapM resolveStatement locals
  240     body'   <- resolveX body
  241     return (WithLocals body' (AuxiliaryVars locals'))
  242 
  243 resolveX (WithLocals body (DefinednessConstraints locals)) = scope $ do
  244     locals' <- mapM resolveX locals
  245     body'   <- resolveX body
  246     return (WithLocals body' (DefinednessConstraints locals'))
  247 
  248 resolveX x = descendM resolveX x

  251 resolveD ::
  252     MonadFail m =>
  253     MonadState [(Name, ReferenceTo)] m =>
  254     MonadUserError m =>
  255     NameGen m =>
  256     Data r =>
  257     Default r =>
  258     Pretty r =>
  259     (?typeCheckerMode :: TypeCheckerMode) =>
  260     Domain r Expression -> m (Domain r Expression)
  261 resolveD (DomainReference nm (Just d)) = DomainReference nm . Just <$> resolveD d
  262 resolveD (DomainReference nm Nothing) = do
  263     mval <- gets (lookup nm)
  264     case mval of
  265         Nothing -> userErr1 ("Undefined reference to a domain:" <+> pretty nm)
  266         Just (Alias (Domain r)) -> DomainReference nm . Just <$> resolveD (changeRepr def r)
  267         Just x -> userErr1 ("Expected a domain, but got an expression:" <+> pretty x)
  268 resolveD (DomainRecord ds) = fmap DomainRecord $ forM ds $ \ (n, d) -> do
  269     d' <- resolveD d
  270     t  <- typeOfDomain d'
  271     modify ((n, RecordField n t) :)
  272     return (n, d')
  273 resolveD (DomainVariant ds) = fmap DomainVariant $ forM ds $ \ (n, d) -> do
  274     d' <- resolveD d
  275     t  <- typeOfDomain d'
  276     modify ((n, VariantField n t) :)
  277     return (n, d')
  278 resolveD d = do
  279     d' <- descendM resolveD d
  280     mapM resolveX d'

  283 resolveAbsPat ::
  284     MonadState [(Name, ReferenceTo)] m =>
  285     MonadUserError m =>
  286     Expression -> AbstractPattern -> Expression -> m ()
  287 resolveAbsPat _ AbstractPatternMetaVar{} _ = bug "resolveAbsPat AbstractPatternMetaVar"
  288 resolveAbsPat _ (Single nm) x = modify ((nm, Alias x) :)
  289 resolveAbsPat context (AbsPatTuple ps) x =
  290     sequence_ [ resolveAbsPat context p [essence| &x[&i] |]
  291               | (p, i_) <- zip ps allNats
  292               , let i = fromInt i_
  293               ]
  294 resolveAbsPat context (AbsPatMatrix ps) x =
  295     sequence_ [ resolveAbsPat context p [essence| &x[&i] |]
  296               | (p, i_) <- zip ps allNats
  297               , let i = fromInt i_
  298               ]
  299 resolveAbsPat context (AbsPatSet ps) x = do
  300     ys <- case x of
  301         Constant (ConstantAbstract (AbsLitSet xs)) -> return (map Constant xs)
  302         AbstractLiteral (AbsLitSet xs) -> return xs
  303         _ -> userErr1 $ "Abstract set pattern cannot be used in this context:" <++> pretty context
  304     sequence_ [ resolveAbsPat context p y
  305               | (p,y) <- zip ps ys
  306               ]

  309 resolveAbsLit ::
  310     MonadFail m =>
  311     MonadState [(Name, ReferenceTo)] m =>
  312     MonadUserError m =>
  313     NameGen m =>
  314     (?typeCheckerMode :: TypeCheckerMode) =>
  315     AbstractLiteral Expression -> m (AbstractLiteral Expression)
  316 resolveAbsLit (AbsLitVariant Nothing n x) = do
  317     x'   <- resolveX x
  318     mval <- gets id
  319     let
  320         isTheVariant (Alias (Domain d@(DomainVariant nms))) | Just{} <- lookup n nms = Just d
  321         isTheVariant _ = Nothing
  322     case mapMaybe (isTheVariant . snd) mval of
  323         (DomainVariant dom:_) -> return (AbsLitVariant (Just dom) n x')
  324         _ -> return (AbsLitVariant Nothing n x')
  325 resolveAbsLit lit = (descendBiM resolveX >=> descendBiM resolveD') lit
  326     where
  327         resolveD' d = resolveD (d :: Domain () Expression)
