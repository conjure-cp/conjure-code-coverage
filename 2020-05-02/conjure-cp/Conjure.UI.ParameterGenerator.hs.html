   19 -- | This doesn't do anything to do with correcting categories at the moment, it should.
   20 --   An example:
   21 --      given n : int(1..10)
   22 --      given s : set (size n) of int(1..10)
   23 --   Should output:
   24 --      find n : int(1..10)
   25 --      find s : set (minSize 1, maxSize 10) of int(1..10)
   26 --      such that n = |s|
   27 --   (Just dropping wrong category stuff from attribute list isn't acceptable, because mset.)
   28 parameterGenerator ::
   29     MonadLog m =>
   30     MonadFail m =>
   31     MonadUserError m =>
   32     EnumerateDomain m =>
   33     NameGen m =>
   34     (?typeCheckerMode :: TypeCheckerMode) =>
   35     Integer ->      -- MININT
   36     Integer ->      -- MAXINT
   37     Model -> m ( Model
   38                , [(Name, String)] -- classification for each given
   39                )
   40 parameterGenerator minIntValue maxIntValue model =
   41     runStateAsWriterT $ runNameGen () (resolveNames model) >>= core >>= evaluateBounds >>= return . inlineLettings
   42     where
   43         core m = do
   44             outStatements <- forM (mStatements m) $ \ st -> case st of
   45                 Declaration (FindOrGiven Given nm dom) -> do
   46                     (dom', decls, cons) <- pgOnDomain (Reference nm Nothing) nm dom
   47                     return $ decls ++ [Declaration (FindOrGiven Find nm dom'), SuchThat cons]
   48                 Declaration (FindOrGiven Find  _  _  ) -> return []
   49                 Declaration (Letting _ _)              -> return [st]
   50                 Declaration       {}                   -> return [st]
   51                 SearchOrder       {}                   -> return []
   52                 SearchHeuristic   {}                   -> return []
   53                 Where             xs                   -> do
   54                     xs' <- mapM (transformM fixQuantified) xs
   55                     return [SuchThat xs']
   56                 Objective         {}                   -> return []
   57                 SuchThat          {}                   -> return []
   58             return m { mStatements = concat outStatements }
   59 
   60         evaluateBounds m = do
   61             let symbols = [("MININT", fromInt minIntValue), ("MAXINT", fromInt maxIntValue)]
   62             let eval = transformBiM (trySimplify symbols)
   63             stmtsEvaluated <- mapM eval (mStatements m)
   64             return m { mStatements = stmtsEvaluated }

   67 inlineLettings :: Model -> Model
   68 inlineLettings model =
   69     let
   70         inline p@(Reference nm _) = do
   71             x <- gets (lookup nm)
   72             return (fromMaybe p x)
   73         inline p = return p
   74 
   75         statements = catMaybes
   76                         $ flip evalState []
   77                         $ forM (mStatements model)
   78                         $ \ st ->
   79             case st of
   80                 Declaration (Letting nm x) -> modify ((nm,x) :) >> return Nothing
   81                 _ -> Just <$> transformBiM inline st
   82     in
   83         model { mStatements = statements }

   86 fixQuantified ::
   87     MonadUserError m =>
   88     NameGen m => 
   89     Expression ->
   90     m Expression
   91 fixQuantified (Comprehension body gocs) = do
   92     gocs' <- forM gocs $ \ goc -> case goc of
   93         Generator (GenDomainNoRepr (Single pat) domain) -> do
   94             let go x d =
   95                     case d of
   96                         DomainInt t [RangeBounded fr to] -> do
   97                             (fr', frCons) <-
   98                                 if categoryOf fr < CatParameter
   99                                     then return (fr, [])
  100                                     else do
  101                                         bound <- lowerBoundOfIntExpr fr
  102                                         return (bound, return [essence| &x >= &fr |])
  103                             (to', toCons) <-
  104                                 if categoryOf to < CatParameter
  105                                     then return (to, [])
  106                                     else do
  107                                         bound <- upperBoundOfIntExpr to
  108                                         return (bound, return [essence| &x <= &to |])
  109                             return (DomainInt t [RangeBounded fr' to'], frCons ++ toCons)
  110                         DomainFunction r attr innerFr innerTo -> do
  111                             (jPat, j) <- quantifiedVar
  112                             (innerFr', consFr) <- go [essence| &j[1] |] innerFr
  113                             (innerTo', consTo) <- go [essence| &j[2] |] innerTo
  114                             let innerCons = make opAnd $ fromList $ consFr ++ consTo
  115                             return ( DomainFunction r attr innerFr' innerTo'
  116                                    , return [essence| forAll &jPat in &x . &innerCons |]
  117                                    )
  118                         _ -> return (d, [])
  119 
  120             let patX = Reference pat Nothing
  121             (dom', cons) <- go patX (expandDomainReference domain)
  122             return $ [Generator (GenDomainNoRepr (Single pat) dom')]
  123                    ++ map Condition cons
  124         _ -> return [goc]
  125     return (Comprehension body (concat gocs'))
  126 fixQuantified x = return x

  601             let consToBool = [essence| sum([ toInt(&i[2]) | &iPat <- &x ]) = &refPercentage * |defined(&x)| / 100 |]
  602 
  603             newCons <- concat <$> sequence [cardinalityCons, totalityCons, sizeLbCons, sizeUbCons]
  604             let innerCons = concat $ concat
  605                     [ [consFr | isPartial ] -- only if the function is not total
  606                     , [consTo]
  607                     ]
  608 
  609             return3
  610                 (DomainFunction r attrOut domFr domTo)
  611                 (newDecl ++ concat [ declFr | isPartial ] ++ declTo ++ concat [ [declToBool] | isToBool ])
  612                 (newCons ++ map liftCons innerCons ++ concat [[consToBool] | isToBool ])
  613 
  614         DomainRelation r attr innerDomains -> do
  615 
  616             let nmCardMin = nm `mappend` "_cardMin"
  617             sawTell [(nmCardMin, "i")]
  618             let cardMin = Reference nmCardMin Nothing
  619 
  620             let nmCardMax  = nm `mappend` "_cardMax"
  621             sawTell [(nmCardMax, "i")]
  622             let cardMax = Reference nmCardMax Nothing
  623 
  624             (iPat, i) <- quantifiedVar
  625             let liftCons c = [essence| forAll &iPat in &x . &c |]
  626 
  627             inners <- forM (zip [1..] innerDomains) $ \ (n, d) -> do
  628                 let nE = fromInt n
  629                 let ref = [essence| &i[&nE] |]
  630                 (d', decl, cons) <- pgOnDomain ref (nm `mappend` (Name $ pack $ "_relation" ++ show n)) d
  631                 return (d', decl, map liftCons cons)
  632 
  633             let maxIntN = maxIntTimes (genericLength innerDomains)
  634 
  635             (attrOut, sizeLb, sizeUb, cardDomain) <-
  636                     case attr of
  637                         RelationAttr size binRelAttr -> do
  638                             (sizeOut, lb, ub, cardDomain) <-
  639                                 case size of
  640                                     SizeAttr_None ->
  641                                         return ( SizeAttr_MaxSize maxIntN, Nothing, Nothing
  642                                                , DomainInt TagInt [RangeBounded minInt maxIntN]
  643                                                )
  644                                     SizeAttr_Size a -> do
  645                                         lb <- lowerBoundOfIntExpr a
  646                                         ub <- upperBoundOfIntExpr a
  647                                         return ( SizeAttr_MinMaxSize lb ub, Just a, Just a
  648                                                , DomainInt TagInt [RangeBounded lb ub]
  649                                                )
  650                                     SizeAttr_MinSize a -> do
  651                                         lb <- lowerBoundOfIntExpr a
  652                                         return ( SizeAttr_MinMaxSize lb maxIntN, Just a, Nothing
  653                                                , DomainInt TagInt [RangeBounded lb maxIntN]
  654                                                )
  655                                     SizeAttr_MaxSize a -> do
  656                                         ub <- upperBoundOfIntExpr a
  657                                         return ( SizeAttr_MaxSize ub, Nothing, Just a
  658                                                , DomainInt TagInt [RangeBounded 0 ub]
  659                                                )
  660                                     SizeAttr_MinMaxSize a b -> do
  661                                         lb <- lowerBoundOfIntExpr a
  662                                         ub <- upperBoundOfIntExpr b
  663                                         return ( SizeAttr_MinMaxSize lb ub, Just a, Just b
  664                                                , DomainInt TagInt [RangeBounded lb ub]
  665                                                )
  666                             return (RelationAttr sizeOut binRelAttr, lb, ub, cardDomain)
  667 
  668             let
  669                 newDecl =
  670                     [ Declaration (FindOrGiven Given nmCardMin cardDomain)
  671                     , Declaration (FindOrGiven Given nmCardMax cardDomain)
  672                     ]
  673 
  674             let
  675                 cardinalityCons = return $ return
  676                     [essence|
  677                         |&x| >= &cardMin /\
  678                         |&x| <= &cardMax
  679                     |]
  680 
  681                 sizeLbCons =
  682                     case sizeLb of
  683                         Nothing -> return []
  684                         Just bound -> return $ return [essence| |&x| >= &bound |]
  685 
  686                 sizeUbCons =
  687                     case sizeUb of
  688                         Nothing -> return []
  689                         Just bound -> return $ return [essence| |&x| <= &bound |]
  690 
  691             newCons <- concat <$> sequence [cardinalityCons, sizeLbCons, sizeUbCons]
  692 
  693             return3
  694                 (DomainRelation r attrOut (map fst3 inners))
  695                 (newDecl ++ concatMap snd3 inners)
  696                 (newCons ++ concatMap thd3 inners)
  697 
  698         _ -> userErr1 $ "Unhandled domain:" <++> vcat [ pretty dom
  699                                                       , pretty (show dom)
  700                                                       ]

  705 return3 :: Monad m => a -> b -> c -> m (a,b,c)
  706 return3 x y z = return (x,y,z)

  708 minInt :: Expression
  709 minInt = Reference "MININT" Nothing
  710 
  711 maxInt :: Expression
  712 maxInt = Reference "MAXINT" Nothing

  714 maxIntTimes :: Integer -> Expression
  715 maxIntTimes n = let m = fromInt n in [essence| &maxInt ** &m |]

  718 minOfIntDomain :: MonadUserError m => Domain () Expression -> m Expression
  719 minOfIntDomain (DomainInt _ rs) = do
  720     xs <- sortNub <$> mapM minOfIntRange rs
  721     case xs of
  722         []  -> return minInt
  723         [x] -> return x
  724         _   -> return $ make opMax $ fromList xs
  725 minOfIntDomain d = userErr1 $ "Expected integer domain, but got:" <++> vcat [pretty d, pretty (show d)]
  726 
  727 minOfIntRange :: Monad m => Range Expression -> m Expression
  728 minOfIntRange (RangeSingle lb) = return lb
  729 minOfIntRange (RangeLowerBounded lb) = return lb
  730 minOfIntRange (RangeBounded lb _) = return lb
  731 minOfIntRange _ = return minInt
  732 
  733 lowerBoundOfIntExpr :: MonadUserError m => Expression -> m Expression
  734 lowerBoundOfIntExpr x@Constant{} = return x
  735 lowerBoundOfIntExpr x | x == minInt = return minInt
  736 lowerBoundOfIntExpr x | x == maxInt = return maxInt
  737 lowerBoundOfIntExpr (Reference _ (Just (DeclNoRepr Given _ dom _))) = minOfIntDomain dom
  738 lowerBoundOfIntExpr (Reference _ (Just (Alias x))) = lowerBoundOfIntExpr x
  739 lowerBoundOfIntExpr (Op (MkOpMinus (OpMinus a b))) = do
  740     aLower <- lowerBoundOfIntExpr a
  741     bUpper <- upperBoundOfIntExpr b
  742     return $ make opMinus aLower bUpper
  743 lowerBoundOfIntExpr (Op (MkOpSum (OpSum x))) | Just xs <- listOut x = do
  744     bounds <- mapM lowerBoundOfIntExpr xs
  745     return $ make opSum $ fromList bounds
  746 -- TODO: check for negatives
  747 lowerBoundOfIntExpr (Op (MkOpProduct (OpProduct x))) | Just xs <- listOut x = do
  748     bounds <- mapM lowerBoundOfIntExpr xs
  749     return $ make opProduct $ fromList bounds
  750 lowerBoundOfIntExpr (Op (MkOpMin (OpMin x))) | Just xs <- listOut x = do
  751     bounds <- mapM lowerBoundOfIntExpr xs
  752     return $ make opMin $ fromList bounds
  753 lowerBoundOfIntExpr (Op (MkOpMax (OpMax x))) | Just xs <- listOut x = do
  754     bounds <- mapM lowerBoundOfIntExpr xs
  755     return $ make opMin $ fromList bounds
  756 lowerBoundOfIntExpr (Op (MkOpNegate (OpNegate x))) = do
  757     bound <- upperBoundOfIntExpr x
  758     return (make opNegate bound)
  759 lowerBoundOfIntExpr x = userErr1 $ "Cannot compute lower bound of integer expression:" <++> vcat [pretty x, pretty (show x)]
  760

  762 maxOfIntDomain :: MonadUserError m => Domain () Expression -> m Expression
  763 maxOfIntDomain (DomainInt _ rs) = do
  764     xs <- sortNub <$> mapM maxOfIntRange rs
  765     case xs of
  766         []  -> return maxInt
  767         [x] -> return x
  768         _   -> return $ make opMin $ fromList xs
  769 maxOfIntDomain d = userErr1 $ "Expected integer domain, but got:" <++> pretty d

  771 maxOfIntRange :: Monad m => Range Expression -> m Expression
  772 maxOfIntRange (RangeSingle ub) = return ub
  773 maxOfIntRange (RangeUpperBounded ub) = return ub
  774 maxOfIntRange (RangeBounded _ ub) = return ub
  775 maxOfIntRange _ = return maxInt

  777 upperBoundOfIntExpr :: MonadUserError m => Expression -> m Expression
  778 upperBoundOfIntExpr x@Constant{} = return x
  779 upperBoundOfIntExpr x | x == minInt = return minInt
  780 upperBoundOfIntExpr x | x == maxInt = return maxInt
  781 upperBoundOfIntExpr (Reference _ (Just (DeclNoRepr Given _ dom _))) = maxOfIntDomain dom
  782 upperBoundOfIntExpr (Reference _ (Just (Alias x))) = upperBoundOfIntExpr x
  783 upperBoundOfIntExpr (Op (MkOpMinus (OpMinus a b))) = do
  784     aUpper <- upperBoundOfIntExpr a
  785     bLower <- lowerBoundOfIntExpr b
  786     return $ make opMinus aUpper bLower
  787 upperBoundOfIntExpr (Op (MkOpSum (OpSum x))) | Just xs <- listOut x = do
  788     bounds <- mapM upperBoundOfIntExpr xs
  789     return $ make opSum $ fromList bounds
  790 -- TODO: check for negatives
  791 upperBoundOfIntExpr (Op (MkOpProduct (OpProduct x))) | Just xs <- listOut x = do
  792     bounds <- mapM upperBoundOfIntExpr xs
  793     return $ make opProduct $ fromList bounds
  794 upperBoundOfIntExpr (Op (MkOpMin (OpMin x))) | Just xs <- listOut x = do
  795     bounds <- mapM upperBoundOfIntExpr xs
  796     return $ make opMax $ fromList bounds
  797 upperBoundOfIntExpr (Op (MkOpMax (OpMax x))) | Just xs <- listOut x = do
  798     bounds <- mapM upperBoundOfIntExpr xs
  799     return $ make opMax $ fromList bounds
  800 upperBoundOfIntExpr (Op (MkOpNegate (OpNegate x))) = do
  801     bound <- lowerBoundOfIntExpr x
  802     return (make opNegate bound)
  803 upperBoundOfIntExpr x = userErr1 $ "Cannot compute lower bound of integer expression:" <++> vcat [pretty x, pretty (show x)]
