  283 lexemeFace :: Lexeme -> Pr.Doc
  284 lexemeFace L_Newline = "new line"
  285 lexemeFace L_Carriage = "\\r"
  286 lexemeFace L_Space   = "space character"
  287 lexemeFace L_Tab     = "tab character"
  288 lexemeFace (LIntLiteral i) = Pr.integer i
  289 lexemeFace (LIdentifier i) = Pr.text (T.unpack i)
  290 lexemeFace (LComment    i) = Pr.text (T.unpack i)
  291 lexemeFace l =
  292     case M.lookup l mapLexemeToText of
  293         Nothing -> Pr.text (show l)
  294         Just t  -> Pr.text (T.unpack t)

  296 lexemeWidth :: Lexeme -> Int
  297 lexemeWidth L_Carriage = 0
  298 lexemeWidth L_Tab = 4
  299 lexemeWidth (LIntLiteral i) = length (show i)
  300 lexemeWidth (LIdentifier i) = T.length i
  301 lexemeWidth (LComment    i) = T.length i
  302 lexemeWidth l =
  303     case lookup l (map swap lexemes) of
  304         Nothing -> 0
  305         Just t  -> T.length t

  491 runLexer :: MonadFail m => T.Text -> m [LexemePos]
  492 runLexer text = do
  493     ls <- go text
  494     let lsPaired = calcPos (initialPos "") ls
  495     return lsPaired
  496     where
  497         go t = do
  498             let results = catMaybes $  tryLexMetaVar t
  499                                     :  map (tryLex t) lexemes
  500                                     ++ [ tryLexIntLiteral t
  501                                        , tryLexIden t
  502                                        , tryLexQuotedIden t
  503                                        , tryLexComment t
  504                                        ]
  505             if T.null t
  506                 then return []
  507                 else case results of
  508                         [] -> fail ("Lexing error:" Pr.<+> Pr.text (T.unpack t))
  509                         ((rest,lexeme):_) -> (lexeme:) <$> go rest
  510 
  511         -- attach source positions to lexemes
  512         -- discard whitespace, but calculate their contribution to source positions
  513         calcPos :: SourcePos -> [Lexeme] -> [LexemePos]
  514         calcPos _pos [] = []
  515         calcPos  pos (this:rest) | isLexemeSpace this                   -- skip if this one is whitespace
  516                                  = calcPos (nextPos pos this) rest      -- can only happen at the beginning
  517         calcPos  pos (this:rest) =
  518             let (restSpaces, restNonSpace) = span isLexemeSpace rest    -- eat up all the whitespace after "this"
  519                 pos' = foldl nextPos pos (this:restSpaces)
  520             in
  521                 if null restNonSpace
  522                     then [LexemePos this pos (nextPos pos this)]        -- if this is the last non-whitespace lexeme
  523                                                                         -- do not include the whitespace after it
  524                     else LexemePos this pos pos' : calcPos pos' restNonSpace
  525 
  526         nextPos :: SourcePos -> Lexeme -> SourcePos
  527         nextPos pos L_Newline  = incSourceLine (setSourceColumn pos 1) 1
  528         nextPos pos L_Carriage = pos -- just ignore '\r's
  529         nextPos pos l          = incSourceColumn pos (lexemeWidth l)
  530 
  531 isLexemeSpace :: Lexeme -> Bool
  532 isLexemeSpace L_Newline {} = True
  533 isLexemeSpace L_Carriage{} = True
  534 isLexemeSpace L_Tab     {} = True
  535 isLexemeSpace L_Space   {} = True
  536 isLexemeSpace LComment  {} = True
  537 isLexemeSpace _            = False

  539 tryLex :: T.Text -> (T.Text, Lexeme) -> Maybe (T.Text, Lexeme)
  540 tryLex running (face,lexeme) = do
  541     rest <- T.stripPrefix face running
  542     if T.all isIdentifierLetter face
  543         then
  544             case T.uncons rest of
  545                 Just (ch, _) | isIdentifierLetter ch -> Nothing
  546                 _                                    -> Just (rest, lexeme)
  547         else Just (rest, lexeme)

  549 tryLexIntLiteral :: T.Text -> Maybe (T.Text, Lexeme)
  550 tryLexIntLiteral t =
  551     case T.decimal t of
  552         Left _ -> Nothing
  553         Right (x, rest) -> Just (rest, LIntLiteral x)

  563 isIdentifierFirstLetter :: Char -> Bool
  564 isIdentifierFirstLetter ch = isAlpha ch || ch `elem` ("_" :: String) || ch `elem` emojis
  565 
  566 isIdentifierLetter :: Char -> Bool
  567 isIdentifierLetter ch = isAlphaNum ch || ch `elem` ("_'" :: String) || ch `elem` emojis

  569 tryLexMetaVar :: T.Text -> Maybe (T.Text, Lexeme)
  570 tryLexMetaVar running = do
  571     ('&', rest) <- T.uncons running
  572     (rest2, LIdentifier iden) <- tryLexIden rest
  573     return (rest2, LMetaVar iden)
  574 
  575 tryLexIden :: T.Text -> Maybe (T.Text, Lexeme)
  576 tryLexIden running = do
  577     let (iden,rest) = T.span isIdentifierLetter running
  578     (ch, _) <- T.uncons running
  579     if isIdentifierFirstLetter ch
  580         then
  581             if T.null iden
  582                 then Nothing
  583                 else Just (rest, LIdentifier iden)
  584         else Nothing
  585 
  586 tryLexQuotedIden :: T.Text -> Maybe (T.Text, Lexeme)
  587 tryLexQuotedIden running = do
  588     let
  589         go inp = do
  590             ('\"', rest) <- T.uncons inp
  591             go2 "\"" rest
  592 
  593         -- after the first "
  594         go2 sofar inp = do
  595             (ch, rest) <- T.uncons inp
  596             case ch of
  597                 -- end
  598                 '\"'
  599                     | sofar /= "\""         -- so we don't allow empty strings
  600                     -> Just (rest, LIdentifier (T.pack (reverse ('\"' : sofar))))
  601                 -- escaped
  602                 '\\' -> do
  603                     (ch2, rest2) <- T.uncons rest
  604                     case ch2 of
  605                         '\"' -> go2 ('\"':sofar) rest2
  606                         '\\' -> go2 ('\\':sofar) rest2
  607                         _ -> Nothing
  608                 _ -> go2 (ch:sofar) rest
  609     go running
  610 
  611 tryLexComment :: T.Text -> Maybe (T.Text, Lexeme)
  612 tryLexComment running = let (dollar,rest1) = T.span (=='$') running
  613                         in  if T.null dollar
  614                                 then Nothing
  615                                 else let (commentLine,rest2) = T.span (/='\n') rest1
  616                                      in  Just (rest2, LComment commentLine)
