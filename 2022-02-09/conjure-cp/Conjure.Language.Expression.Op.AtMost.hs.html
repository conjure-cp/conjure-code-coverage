   14                            x -- o[i] is an upper bound for the nb of occurrences of v[i] in X, indexed by T
   15                            x -- v[i] values, indexed by T

   23 instance (TypeOf x, Pretty x) => TypeOf (OpAtMost x) where
   24     typeOf p@(OpAtMost vars occs vals) = do
   25         tyVars <- typeOf vars
   26         tyOccs <- typeOf occs
   27         tyVals <- typeOf vals
   28         let
   29             tyError = raiseTypeError $ vcat [ pretty p
   30                                             , pretty vars <+> "has type" <+> pretty tyVars
   31                                             , pretty occs <+> "has type" <+> pretty tyOccs
   32                                             , pretty vals <+> "has type" <+> pretty tyVals
   33                                             ]
   34             listLike (TypeList TypeInt{}) = return ()
   35             listLike (TypeMatrix _ TypeInt{}) = return ()
   36             listLike _ = tyError
   37         listLike tyVars
   38         listLike tyVals
   39         listLike tyOccs
   40         -- TODO: check the indices and the category of vals&occs
   41         return TypeBool
