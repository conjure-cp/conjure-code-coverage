   16 -- | To use a lens for constructing stuf.
   17 make :: (Proxy Identity -> (a, b)) -> a
   18 make  f = fst (f (Proxy :: Proxy Identity))
   19 
   20 -- | To use a lens for deconstructing stuf.
   21 match :: (Proxy (m :: * -> *) -> (a, b -> m c)) -> b -> m c
   22 match f = snd (f Proxy)
   23 
   24 followAliases :: CanBeAnAlias b => (b -> c) -> b -> c
   25 followAliases m (isAlias -> Just x) = followAliases m x
   26 followAliases m x = m x
   27 
   28 tryMatch :: (Proxy Maybe -> (a, b -> Maybe c)) -> b -> Maybe c
   29 tryMatch f = match f
   30 
   31 matchOr :: c -> (Proxy Maybe -> (a, b -> Maybe c)) -> b -> c
   32 matchOr defOut f inp = fromMaybe defOut (match f inp)
   33 
   34 matchDef :: (Proxy Maybe -> (a, b -> Maybe b)) -> b -> b
   35 matchDef f inp = matchOr inp f inp
   36 
   37 matchDefs :: CanBeAnAlias b => [Proxy Maybe -> (a, b -> Maybe b)] -> b -> b
   38 matchDefs fs inp =
   39     case mapMaybe (`match` inp) fs of
   40         []      -> inp
   41         (out:_) -> matchDefs fs out

  513 opMatrixIndexing
  514     :: ( Op x :< x
  515        , Pretty x
  516        , TypeOf x
  517        , MonadFail m
  518        , ?typeCheckerMode :: TypeCheckerMode
  519        )
  520     => Proxy (m :: * -> *)
  521     -> ( x -> [x] -> x
  522        , x -> m (x,[x])
  523        )
  524 opMatrixIndexing _ =
  525     ( foldl (make opIndexing)
  526     , \ p -> do
  527         (m, is) <- go p
  528         if null is
  529             then na ("Lenses.opMatrixIndexing:" <+> pretty p)
  530             else return (m, is)
  531     )
  532     where
  533         go p = case project p of
  534             Just (MkOpIndexing (OpIndexing x i)) -> do
  535                 ty <- typeOf x
  536                 case ty of
  537                     TypeMatrix{} -> return ()
  538                     TypeList{} -> return ()
  539                     _ -> na ("Lenses.opMatrixIndexing:" <+> pretty p)
  540                 (m,is) <- go x
  541                 return (m, is ++ [i])
  542             _ -> return (p, [])

  545 opMatrixIndexingSlicing
  546     :: ( Op x :< x
  547        , Pretty x
  548        , MonadFail m
  549        , TypeOf x
  550        , ?typeCheckerMode :: TypeCheckerMode
  551        )
  552     => Proxy (m :: * -> *)
  553     -> ( x -> [Either x (Maybe x, Maybe x)] -> x
  554        , x -> m (x, [Either x (Maybe x, Maybe x)])           -- either an index or a slice
  555        )
  556 opMatrixIndexingSlicing _ =
  557     ( mk
  558     , \ p -> do
  559         (m, is, wasThereAnySlicing) <- go p
  560         if not (null is) && wasThereAnySlicing
  561             then return (m, is)
  562             else na ("Lenses.opMatrixIndexingSlicing:" <+> pretty p)
  563     )
  564     where
  565         mk m [] = m
  566         mk m (Left i:is) = mk (make opIndexing m i) is
  567         mk m (Right (lb, ub):is) = mk (make opSlicing m lb ub) is
  568 
  569         go p = case project p of
  570             Just (MkOpIndexing (OpIndexing x i)) -> do
  571                 ty <- typeOf x
  572                 case ty of
  573                     TypeMatrix{} -> return ()
  574                     TypeList{} -> return ()
  575                     _ -> na ("Lenses.opMatrixIndexingSlicing:" <+> pretty p)
  576                 (m, is, wasThereAnySlicing) <- go x
  577                 return (m, is ++ [Left i], wasThereAnySlicing)
  578             Just (MkOpSlicing (OpSlicing x lb ub)) -> do
  579                 ty <- typeOf x
  580                 case ty of
  581                     TypeMatrix{} -> return ()
  582                     TypeList{} -> return ()
  583                     _ -> na ("Lenses.opMatrixIndexingSlicing:" <+> pretty p)
  584                 (m, is, _) <- go x
  585                 return (m, is ++ [Right (lb, ub)], True)
  586             _ -> return (p, [], False)

 1136                 , x

 1251        )

 1270 onMatrixLiteral
 1271     :: (Functor m, Applicative m, Monad m, NameGen m)
 1272     => Maybe Int                                    -- how many levels to go down. all the way if Nothing.
 1273     -> (Expression -> m Expression)
 1274     ->  Expression -> m Expression
 1275 onMatrixLiteral mlvl f = case mlvl of
 1276                             Nothing  -> followAliases go
 1277                             Just lvl -> followAliases (goL lvl)
 1278     where
 1279         go (Constant (ConstantAbstract (AbsLitMatrix index xs))) =
 1280             AbstractLiteral . AbsLitMatrix (fmap Constant index) <$> mapM (go . Constant) xs
 1281         go (AbstractLiteral (AbsLitMatrix index xs)) =
 1282             AbstractLiteral . AbsLitMatrix index <$> mapM go xs
 1283         go (Typed x _) = go x
 1284         go (Constant (TypedConstant x _)) = go (Constant x)
 1285         go p = f p
 1286 
 1287         goL 0 p = f p
 1288         goL lvl (Constant (ConstantAbstract (AbsLitMatrix index xs))) =
 1289             AbstractLiteral . AbsLitMatrix (fmap Constant index) <$> mapM (goL (lvl-1) . Constant) xs
 1290         goL lvl (AbstractLiteral (AbsLitMatrix index xs)) =
 1291             AbstractLiteral . AbsLitMatrix index <$> mapM (goL (lvl-1)) xs
 1292         goL lvl (Typed x _) = goL lvl x
 1293         goL lvl (Constant (TypedConstant x _)) = goL lvl (Constant x)
 1294         goL lvl p = do
 1295             (iPat, i) <- quantifiedVar
 1296             body <- goL (lvl-1) i
 1297             return $ Comprehension body [Generator (GenInExpr iPat p)]
