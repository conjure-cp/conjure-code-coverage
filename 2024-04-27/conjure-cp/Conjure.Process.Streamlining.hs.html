   18 streamliningToStdout ::
   19     MonadFailDoc m =>
   20     MonadLog m =>
   21     MonadUserError m =>
   22     NameGen m =>
   23     MonadIO m =>
   24     (?typeCheckerMode :: TypeCheckerMode) =>
   25     Model -> m ()
   26 streamliningToStdout model = do
   27     let
   28         whitespace '\t' = ' '
   29         whitespace '\n' = ' '
   30         whitespace ch   = ch
   31 
   32         showStr :: String -> Doc
   33         showStr = pretty . show
   34 
   35     streamliners <- streamlining model
   36 
   37     liftIO $ print $ prettyList prBraces ","
   38         [ showStr (show i) <> ":" <+> prBraces (vcat
   39             [ showStr "onVariable" <> ":" <+> showStr (show (pretty nm)) <> ","
   40             , showStr "groups"     <> ":" <+> prettyList prBrackets "," (map showStr (nub groups)) <> ","
   41             , showStr "constraint" <> ":" <+> showStr (map whitespace $ show $ pretty cons)
   42             ])
   43         | (i, (nm, (cons, groups))) <- zip allNats streamliners
   44         ]
   45     traceM $ show $ "Number of streamliners: " <+> pretty (length streamliners)
   46

   48 streamlining ::
   49     MonadFailDoc m =>
   50     MonadLog m =>
   51     MonadUserError m =>
   52     NameGen m =>
   53     (?typeCheckerMode :: TypeCheckerMode) =>
   54     Model -> m [(Name, Streamliner)]
   55 streamlining model = do
   56     concatForM (mStatements model) $ \case
   57         Declaration (FindOrGiven Find nm domain) -> do
   58             let ref = Reference nm (Just (DeclNoRepr Find nm domain NoRegion))
   59             streamliners <- streamlinersForSingleVariable ref
   60             -- traceM $ show $ vcat [ "Streamliners for --" <+> pretty statement
   61             --                      , vcat [ nest 4 (vcat (pretty x : map pretty gs)) | (x,gs) <- streamliners ]
   62             --                      ]
   63             return [ (nm, s) | s <- streamliners ]
   64         _ -> noStreamliner
   65 
   66 
   67 type StreamlinerGroup = String
   68 
   69 type Streamliner = (Expression, [StreamlinerGroup])
   70 
   71 type StreamlinerGen m = Expression -> m [Streamliner]
   72 
   73 mkStreamliner
   74     :: Monad m
   75     => StreamlinerGroup         -- the group label
   76     -> Expression               -- the streamlining constraint
   77     -> m [Streamliner]
   78 mkStreamliner grp x = return [(x, [grp])]
   79 
   80 noStreamliner :: Monad m => m [a]
   81 noStreamliner = return []
   82 
   83 attachGroup
   84     :: Monad m
   85     => [StreamlinerGroup]
   86     -> Expression
   87     -> m Streamliner
   88 attachGroup grps x = return (x, grps)

  150 -- given an integer expression (which can be a reference to a decision variable),
  151 -- generate a constraint forcing it to be odd
  152 intOdd ::
  153     MonadFailDoc m =>
  154     (?typeCheckerMode :: TypeCheckerMode) =>
  155     StreamlinerGen m
  156 intOdd x = do
  157     ty <- typeOf x
  158     if typeUnify ty (TypeInt TagInt)
  159         then mkStreamliner "IntOddEven" [essence| &x % 2 = 1 |]
  160         else noStreamliner

  174 intLowerHalf ::
  175     MonadFailDoc m =>
  176     NameGen m =>
  177     (?typeCheckerMode :: TypeCheckerMode) =>
  178     StreamlinerGen m
  179 intLowerHalf x = do
  180     ty <- typeOf x
  181     dom <- expandDomainReference <$> domainOf x
  182     case dom of
  183         DomainInt _ [RangeBounded _lower upper] -> do
  184             -- traceM $ show $ "DomainInt " <+> pretty (lower, upper)
  185             if typeUnify ty (TypeInt TagInt)
  186                 then mkStreamliner "IntLowHigh" [essence| &x < 1 + (&upper -1) /2 |]
  187                 else noStreamliner
  188         _ -> noStreamliner

  213 matrixAll ::
  214     MonadFailDoc m =>
  215     NameGen m =>
  216     (?typeCheckerMode :: TypeCheckerMode) =>
  217     StreamlinerGen m -> StreamlinerGen m
  218 matrixAll innerStreamliner x = do
  219     dom <- expandDomainReference <$> domainOf x
  220     case dom of
  221         DomainMatrix indexDom innerDom -> do
  222             nm <- nextName "q"
  223             -- traceM $ show $ "matrixAll nm" <+> pretty nm
  224             let pat = Single nm
  225                 ref = Reference nm (Just (DeclNoRepr Find nm innerDom NoRegion))
  226 
  227                 liftMatrix (Reference n _) | n == nm = [essence| &x[&ref] |]
  228                 liftMatrix p = p
  229 
  230             innerConstraints <- transformBi liftMatrix <$> innerStreamliner ref
  231             -- traceM $ show $ "maybeInnerConstraint" <+> vcat (map pretty innerConstraints)
  232             forM innerConstraints $ \ (innerConstraint, grps) ->
  233                     attachGroup ("MatrixCardinality": grps) [essence| forAll &pat : &indexDom . &innerConstraint |]
  234         _ -> noStreamliner

  382             -- traceM $ show $ "maybeInnerConstraint" <+> vcat (map pretty innerConstraints)
  383             forM innerConstraints $ \ (innerConstraint, grps) ->
  384                     attachGroup (newTag: grps) [essence| forAll &pat in &x . &innerConstraint |]
  385         _ -> noStreamliner

  519 relationCardinality ::
  520     MonadFailDoc m =>
  521     NameGen m =>
  522     (?typeCheckerMode :: TypeCheckerMode) =>
  523     StreamlinerGen m
  524 relationCardinality x = do
  525     dom <- expandDomainReference <$> domainOf x
  526     case dom of
  527         DomainRelation _ _ inners -> do
  528             let maxCard = make opProduct $ fromList [ [essence| |`&i`| |] | i <- inners ]
  529             sequence
  530                 [ case lowerOrUpper of
  531                     "lowerBound" -> return ([essence| |&x| >= &maxCard / &slice |], [grp])
  532                     "upperBound" -> return ([essence| |&x| <= &maxCard / &slice |], [grp])
  533                     _ -> bug "relationCardinality"
  534                 | slice <- [1,2,4,8,16,32]
  535                 , lowerOrUpper <- ["LowerBound","UpperBound"]
  536                 , let grp = "RelationCardinality-" ++ lowerOrUpper
  537                 ]
  538         _ -> noStreamliner

  541 binRelAttributes ::
  542     MonadFailDoc m =>
  543     NameGen m =>
  544     (?typeCheckerMode :: TypeCheckerMode) =>
  545     StreamlinerGen m
  546 binRelAttributes x = do
  547     dom <- expandDomainReference <$> domainOf x
  548     case dom of
  549         -- we don't insist on inner1 == inner2 any more
  550         DomainRelation _ _ [inner1, inner2] -> sequence
  551             [ do
  552                 out <- case softness of
  553                         Nothing -> mkBinRelCons (BinaryRelationAttrs (S.singleton attr)) inner x
  554                         Just s -> mkBinRelConsSoft maxNum s (BinaryRelationAttrs (S.singleton attr)) inner x
  555                 return (make opAnd (fromList out), [grp])
  556             | inner <- [inner1, inner2]
  557             , (attr, maxNum) <- [ ( BinRelAttr_Reflexive     , [essence| |`&inner`| |] )
  558                                 , ( BinRelAttr_Irreflexive   , [essence| |`&inner`| |] )
  559                                 , ( BinRelAttr_Coreflexive   , [essence| |`&inner`| * |`&inner`| |] )
  560                                 , ( BinRelAttr_Symmetric     , [essence| |`&inner`| * |`&inner`| |] )
  561                                 , ( BinRelAttr_AntiSymmetric , [essence| |`&inner`| * |`&inner`| |] )
  562                                 , ( BinRelAttr_ASymmetric    , [essence| |`&inner`| * |`&inner`| |] )
  563                                 , ( BinRelAttr_Transitive    , [essence| |`&inner`| * |`&inner`| * |`&inner`| |] )
  564                                 , ( BinRelAttr_Total         , [essence| |`&inner`| * |`&inner`| |] )
  565                                 , ( BinRelAttr_Connex        , [essence| |`&inner`| * |`&inner`| |] )
  566                                 , ( BinRelAttr_Euclidean     , [essence| |`&inner`| * |`&inner`| * |`&inner`| |] )
  567                                 , ( BinRelAttr_Serial        , [essence| |`&inner`| |] )
  568                                 , ( BinRelAttr_Equivalence   , [essence| |`&inner`| * |`&inner`| * |`&inner`| |] )
  569                                 , ( BinRelAttr_PartialOrder  , [essence| |`&inner`| * |`&inner`| * |`&inner`| |] )
  570                                 ]
  571             , let grp = show attr
  572             , softness <- [Nothing, Just 2, Just 4, Just 8, Just 16, Just 32]
  573             ]
  574         _ -> noStreamliner

  582 monotonicallyIncreasing ::
  583     MonadFailDoc m =>
  584     NameGen m =>
  585     (?typeCheckerMode :: TypeCheckerMode) =>
  586     StreamlinerGen m
  587 monotonicallyIncreasing x = do
  588     -- traceM $ show $ "Monotonically Increasing [1]" <+> pretty x
  589     dom <- expandDomainReference <$> domainOf x
  590     -- traceM $ show $ "Monotonically Increasing [2]" <+> pretty dom
  591     let
  592         applicable = case dom of
  593                         DomainFunction _ _ DomainInt{} DomainInt{} -> True
  594                         DomainSequence _ _             DomainInt{} -> True
  595                         _ -> False
  596     if applicable
  597         then do
  598             (iPat, i) <- quantifiedVar
  599             (jPat, j) <- quantifiedVar
  600             mkStreamliner "FuncIncreaseDecrease" [essence|
  601                 forAll &iPat in defined(&x) .
  602                     forAll &jPat in defined(&x) .
  603                         &i < &j -> &x(&i) <= &x(&j)
  604             |]
  605         else noStreamliner

  744 onTuple ::
  745     MonadFailDoc m =>
  746     NameGen m =>
  747     (?typeCheckerMode :: TypeCheckerMode) =>
  748     Integer -> StreamlinerGen m -> StreamlinerGen m
  749 onTuple n innerStreamliner x = do
  750     dom <- expandDomainReference <$> domainOf x
  751     case dom of
  752         DomainTuple ds | n >= 1 && n <= genericLength ds -> do
  753             nm <- nextName "q"
  754             let ref = Reference nm (Just (DeclNoRepr Find nm (ds `genericIndex` (n-1)) NoRegion))
  755             innerConstraints <- innerStreamliner ref
  756 
  757             let
  758                 nE = fromInt n
  759                 replaceRef (Reference nm2 _) | nm2 == nm = [essence| &x[&nE] |]
  760                 replaceRef p = p
  761 
  762             return [ (transform replaceRef cons, grps)
  763                    | (cons, grps) <- innerConstraints
  764                    ]
  765 
  766         _ -> noStreamliner

  921     MonadFailDoc m =>
  922     NameGen m =>

  927     case dom of
  928         DomainPartition _ _ partitionDomain -> do
  929             -- traceM $ show $ "partition"

  931             let setDomain = DomainSet () def (DomainSet () def partitionDomain)
  932                 ref =  Reference nm (Just (DeclNoRepr Find nm setDomain NoRegion))
  933 
  934             innerConstraints <- innerStreamliner ref
  935 
  936             let
  937                 replaceWithRangeOfX (Reference n _) | n == nm = [essence| parts(&x) |]
  938                 replaceWithRangeOfX p = p
  939 
  940             return [ (transform replaceWithRangeOfX cons, grps)
  941                    | (cons, grps) <- innerConstraints
  942                    ]
  943
