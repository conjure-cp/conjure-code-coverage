   23 calculateFeatures ::
   24     MonadIO m =>
   25     MonadUserError m =>
   26     Model -> Model -> m ()
   27 calculateFeatures model param = do
   28     let
   29         domains :: M.HashMap Name (Domain () Expression)
   30         domains = M.fromList [ (name, domain)
   31                              | Declaration (FindOrGiven Given name domain) <- mStatements model
   32                              ]
   33 
   34         values :: M.HashMap Name Constant
   35         values = M.fromList [ (name, value)
   36                             | Declaration (Letting name (Constant value)) <- mStatements param
   37                             ]
   38 
   39         parameters :: [Param]
   40         parameters = [ (nm, M.lookup nm domains, value)
   41                      | Declaration (Letting nm (Constant value)) <- mStatements param
   42                      ]
   43 
   44         allIntValues :: [Integer]
   45         allIntValues = [ value | Declaration (Letting _ (Constant (ConstantInt _ value))) <- mStatements param ]
   46 
   47         allIndicators :: [Indicator]
   48         allIndicators = concat [ i p
   49                                | p <- parameters
   50                                , i <- allIndicatorsGens
   51                                ]
   52 
   53         allFeature1s :: [Feature]
   54         allFeature1s = catMaybes [ f allIntValues i
   55                                  | i <- allIndicators
   56                                  , f <- allFeatureGen1s
   57                                  ]
   58 
   59         allFeature2s :: [Feature]
   60         allFeature2s = catMaybes [ f allIntValues i j
   61                                  | i <- allIndicators
   62                                  , j <- allIndicators
   63                                  , fst i < fst j
   64                                  , f <- allFeatureGen2s
   65                                  ]
   66 
   67         allDirectFeatures :: [Feature]
   68         allDirectFeatures = concat [ f p
   69                                    | p <- parameters
   70                                    , f <- allDirectFeatureGens
   71                                    ]
   72 
   73         allFeatures :: [Feature]
   74         allFeatures = allFeature1s ++ allFeature2s ++ allDirectFeatures
   75 
   76     forM_ allFeatures $ \ (names, value) ->
   77         liftIO $ putStrLn $ renderWide $ pretty (mconcat (intersperse "_" names)) <> ":" <+> pretty value

  103 -- types
  104 
  105 type Dom = Maybe (Domain () Expression)
  106 
  107 type Param = (Name, Dom, Constant)
  108 
  109 -- an indicator is some value (like an int itself, the card of a set of the max of the range of a function)
  110 type Indicator = ([Name], Constant)
  111 
  112 type IndicatorsGen = Param -> [Indicator]
  113 
  114 data FeatureValue = B Bool | I Integer | D Double
  115     deriving (Eq, Ord, Show)
  116 
  117 type Feature = ([Name], FeatureValue)
  118 
  119 type FeatureGen1 = [Integer] -> Indicator -> Maybe Feature
  120 
  121 -- something like a ratio between two ints
  122 type FeatureGen2 = [Integer] -> Indicator -> Indicator -> Maybe Feature
  123 
  124 type DirectFeatureGen = Param -> [Feature]
  125 
  126 instance Pretty FeatureValue where
  127     pretty (B x) = pretty x
  128     pretty (I x) = pretty x
  129     pretty (D x) = pretty x

  137 --------------------------------------------------------------------------------
  138 -- int indicators
  139 
  140 allIntIndicators :: [IndicatorsGen]
  141 allIntIndicators = [intValue, cardinality, minMaxOfCollection]
  142 
  143 intValue :: IndicatorsGen
  144 intValue (name, _, value@ConstantInt{}) = return ([name, "intValue"], value)
  145 intValue _ = []
  146 
  147 cardinality :: IndicatorsGen
  148 cardinality (name, _, ConstantAbstract lit) =
  149     let len = case lit of
  150                 AbsLitMatrix _  xs -> Just (genericLength xs)
  151                 AbsLitSet       xs -> Just (genericLength xs)
  152                 AbsLitMSet      xs -> Just (genericLength xs)
  153                 AbsLitFunction  xs -> Just (genericLength xs)
  154                 AbsLitSequence  xs -> Just (genericLength xs)
  155                 AbsLitRelation  xs -> Just (genericLength xs)
  156                 AbsLitPartition xs -> Just (genericLength xs)
  157                 _ -> Nothing
  158     in  case len of
  159             Just l -> return ([name, "cardinality"], ConstantInt TagInt l)
  160             Nothing -> []
  161 cardinality _ = []
  162 
  163 minMaxOfCollection :: IndicatorsGen
  164 minMaxOfCollection (name, _, ConstantAbstract lit) =
  165     let intss = case lit of
  166                     AbsLitMatrix _  xs -> [extractAllInts xs]
  167                     AbsLitSet       xs -> [extractAllInts xs]
  168                     AbsLitMSet      xs -> [extractAllInts xs]
  169                     AbsLitFunction  xs -> [extractAllInts (map fst xs), extractAllInts (map snd xs)]
  170                     AbsLitSequence  xs -> [extractAllInts xs]
  171                     AbsLitRelation  xss -> map extractAllInts (transpose xss)
  172                     AbsLitPartition xss -> [extractAllInts (concat xss)]
  173                     _ -> []
  174     in  [ ([name, indexS, txt], ConstantInt TagInt (reducer ints))
  175         | (index, ints) <- zip allNats intss
  176         , let indexS = Name $ stringToText $ show index
  177         , not (null ints)
  178         , (txt, reducer) <- [("min", minimum), ("max", maximum)]
  179         ] 
  180 minMaxOfCollection _ = []
  181 
  182 extractAllInts :: [Constant] -> [Integer]
  183 extractAllInts xs = [ x | ConstantInt _ x <- xs ]
  184

  207 -- linear, via indicators

  209 allFeatureGen1s :: [FeatureGen1]
  210 allFeatureGen1s = [ valueis, intRatioToMaxInt, intIsOffByOne, intIsRepeated ]
  211 
  212 valueis :: FeatureGen1
  213 valueis _ (name, ConstantBool value) = Just (name, B value)
  214 valueis _ (name, ConstantInt _ value) = Just (name, I value)
  215 valueis _ _ = Nothing
  216 
  217 intRatioToMaxInt :: FeatureGen1
  218 intRatioToMaxInt allIntValues (name, ConstantInt _ value) =
  219     Just (name ++ ["ratioToMax"], D (fromIntegral value / fromIntegral (maximum allIntValues)))
  220 intRatioToMaxInt _ _ = Nothing
  221 
  222 intIsOffByOne :: FeatureGen1
  223 intIsOffByOne allIntValues (name, ConstantInt _ value) =
  224     let flag = any (\ v -> abs (value - v) == 1) allIntValues
  225     in  Just (name ++ ["intIsOffByOne"], B flag)
  226 intIsOffByOne _ _ = Nothing
  227 
  228 intIsRepeated :: FeatureGen1
  229 intIsRepeated allIntValues (name, ConstantInt _ value) =
  230     let flag = sum [1 :: Int | v <- allIntValues, v == value] >= 2
  231     in  Just (name ++ ["intIsRepeated"], B flag)
  232 intIsRepeated _ _ = Nothing

  239 allDirectFeatureGens :: [DirectFeatureGen]
  240 allDirectFeatureGens = [correlation, collectionStats, density]
  241 
  242 correlation :: DirectFeatureGen
  243 correlation (name, _, ConstantAbstract (AbsLitFunction xs)) =
  244     let nums = [ (fromIntegral a, fromIntegral b) | (ConstantInt _ a, ConstantInt _ b) <- xs ]
  245     in  if length nums == length xs && length xs > 0
  246             then return ([name, "correlation"], D $ pearson (Vector.fromList nums))
  247             else []
  248 correlation _ = []
  249 
  250 collectionStats :: DirectFeatureGen
  251 collectionStats (name, _, ConstantAbstract lit) =
  252     let intss = case lit of
  253                     AbsLitMatrix _  xs -> [extractAllInts xs]
  254                     AbsLitSet       xs -> [extractAllInts xs]
  255                     AbsLitMSet      xs -> [extractAllInts xs]
  256                     AbsLitFunction  xs -> [extractAllInts (map fst xs), extractAllInts (map snd xs)]
  257                     AbsLitSequence  xs -> [extractAllInts xs]
  258                     AbsLitRelation  xss -> map extractAllInts (transpose xss)
  259                     AbsLitPartition xss -> [extractAllInts (concat xss)]
  260                     _ -> []
  261         median v = v Vector.! (Vector.length v `div` 2)
  262     in  [ ([name, indexS, txt], D (reducer nums))
  263         | (index, ints) <- zip allNats intss
  264         , let indexS = Name $ stringToText $ show index
  265         , not (null ints)
  266         , let nums = Vector.fromList (map fromIntegral ints)
  267         , (txt, reducer) <- [ ("median", median)
  268                             , ("mean", mean)
  269                             , ("stdDev", stdDev)
  270                             , ("harmonicMean", harmonicMean)
  271                             , ("geometricMean", geometricMean)
  272                             , ("skewness", skewness)
  273                             , ("kurtosis", kurtosis)
  274                             ]
  275         ]
  276 collectionStats _ = []
