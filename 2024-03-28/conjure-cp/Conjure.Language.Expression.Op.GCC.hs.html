   28         tyOccs <- typeOf occs
   29         let
   30             tyError = raiseTypeError $ vcat [ pretty p
   31                                             , pretty vars <+> "has type" <+> pretty tyVars
   32                                             , pretty vals <+> "has type" <+> pretty tyVals

   35             listLike (TypeList TypeInt{}) = return ()
