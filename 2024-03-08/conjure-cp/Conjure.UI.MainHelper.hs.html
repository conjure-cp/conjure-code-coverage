  383                                     let parts = case (head parts', parts', last parts') of

  386                                                             Nothing -> [model, intercalate "-" paramparts]

  400                                                                 </> essenceBasename
  401                                                                 <.> ext
  402                                                 else copySolution file $ essenceDir
  403                                                                 </> intercalate "-" [essenceBasename, solnum]
  404                                                                 <.> ext
  405                                         -- parameterised, but no solution numbers. must be using solutionsInOneFile.
  406                                         (True, False, _singleSolution, [_model, param]) | param `elem` params ->
  407                                             copySolution file $ essenceDir
  408                                                                 </> intercalate "-" [ essenceBasename
  409                                                                                     , param
  410                                                                                     ]
  411                                                                 <.> ext
  412                                         -- parameterised, with solution numbers
  413                                         (False, False, singleSolution, [_model, param, (stripPrefix "solution" -> Just solnum)])

  415                                                  , or [('/' : param) `isSuffixOf` p | p <- params] ] ->

  417                                                     then when (solnum == "000001") $ -- only copy the first solution
  418                                                         copySolution file $ essenceDir
  419                                                                     </> intercalate "-" [ essenceBasename
  420                                                                                         , param
  421                                                                                         ]
  422                                                                     <.> ext
  423                                                     else copySolution file $ essenceDir
  424                                                                     </> intercalate "-" [ essenceBasename
  425                                                                                         , param
  426                                                                                         , solnum
  427                                                                                         ]
  428                                                                     <.> ext

  464                 then autoParallel [ savileRowNoParam config m

  467                 else autoParallel [ savileRowWithParams config m p
  468                                   | m <- eprimes

  478                                | (_, p, Just sol) <- solutions ]

  503                     modelsSoFar' <- l (Just portfolioSize) modelsSoFar
  504                     go modelsSoFar' ls

  783 savileRowNoParam ::
  784     (?typeCheckerMode :: TypeCheckerMode) =>
  785     UI ->
  786     (FilePath, Model) ->        -- model
  787     IO (Either
  788      [Doc]                      -- user error

  892 splitSolverName :: MonadUserError m => String -> m (String, String)
  893 splitSolverName solver = do
  894     let
  895 
  896     (solverName, smtLogicName) <-
  897             case splitOn "-" solver of
  898                 [solverName] | solverName `elem` smtSolvers -> return (solverName, "bv")
  899                 [solverName , logic] | solverName `elem` smtSolvers -> do
  900                     unless (logic `elem` smtSupportedLogics solverName) $
  901                         userErr1 $ vcat [ "SMT logic not supported by Savile Row:" <+> pretty logic
  902                                         , "Supported logics:" <+> prettyList id "," (smtSupportedLogics solverName)
  903                                         ]
  904                     return (solverName, logic)
  905                 _ -> return (solver, "") -- not an smt solver
  906     return (solverName, smtLogicName)

 1014 srStdoutHandler ::
 1015     (FilePath, FilePath, FilePath, UI) ->
 1016     (Model -> NameGenM (IdentityT IO) Model) ->
 1017     Int ->
 1018     Handle ->
 1019     IO [Either String (FilePath, FilePath, Maybe FilePath)]
 1020 srStdoutHandler
 1021         args@(outBase , modelPath, paramPath , Solve{..})
 1022         tr solutionNumber h = do
 1023     eof <- hIsEOF h
 1024     if eof
 1025         then do
 1026             hClose h
 1027             return []
 1028         else do
 1029             line <- hGetLine h
 1030             when (logLevel >= LogDebug) $ do
 1031                 pp logLevel ("SR:" <+> pretty line)
 1032             case stripPrefix "Solution: " line of
 1033                 Nothing -> do
 1034                     if isPrefixOf "Created output file for domain filtering" line
 1035                         then pp logLevel $ hsep ["Running minion for domain filtering."]
 1036                         else if isPrefixOf "Created output" line
 1037                             then pp logLevel $ hsep ["Running solver:", pretty solver]
 1038                             else return ()
 1039                     fmap (Left line :)
 1040                          (srStdoutHandler args tr solutionNumber h)
 1041                 Just solutionText -> do
 1042                     eprimeSol  <- readModel ParserC.parseModel (Just id) ("<memory>", stringToText solutionText)
 1043                     essenceSol <- ignoreLogs $ runNameGen () $ tr eprimeSol
 1044                     case solutionsInOneFile of
 1045                         False -> do
 1046                             let mkFilename ext = outputDirectory </> outBase
 1047                                                         ++ "-solution" ++ padLeft 6 '0' (show solutionNumber)
 1048                                                         ++ ext
 1049                             let filenameEprimeSol  = mkFilename ".eprime-solution"
 1050                             let filenameEssenceSol = mkFilename ".solution"
 1051                             writeModel lineWidth Plain (Just filenameEprimeSol) eprimeSol
 1052                             writeModel lineWidth Plain (Just filenameEssenceSol) essenceSol
 1053                             case outputFormat of
 1054                                 JSON -> writeModel lineWidth outputFormat (Just (filenameEssenceSol ++ ".json")) essenceSol
 1055                                 JSONStream -> writeModel 0 outputFormat (Just (filenameEssenceSol ++ ".json")) essenceSol
 1056                                 _ -> return ()
 1057                             when (outputFormat == MiniZinc) $
 1058                                 writeModel lineWidth outputFormat (Just (filenameEssenceSol ++ ".minizinc")) essenceSol
 1059                             fmap (Right (modelPath, paramPath, Just filenameEssenceSol) :)
 1060                                  (srStdoutHandler args tr (solutionNumber+1) h)
 1061                         True -> do
 1062                             let mkFilename ext = outputDirectory </> outBase
 1063                                                         ++ ext
 1064                             let filenameEprimeSol  = mkFilename ".eprime-solutions"
 1065                             let filenameEssenceSol = mkFilename ".solutions"
 1066                             let filenameEssenceSolJSON = mkFilename ".solutions.json"
 1067                             -- remove the solutions files before writing the first solution
 1068                             if solutionNumber == 1
 1069                                 then do
 1070                                     removeFileIfExists filenameEprimeSol
 1071                                     removeFileIfExists filenameEssenceSol
 1072                                     removeFileIfExists filenameEssenceSolJSON
 1073                                     case outputFormat of
 1074                                         JSON -> writeFile filenameEssenceSolJSON "[\n"
 1075                                         _ -> return ()
 1076                                 else
 1077                                     case outputFormat of
 1078                                         JSON -> appendFile filenameEssenceSolJSON ",\n"
 1079                                         _ -> return ()
 1080                             appendFile filenameEprimeSol  ("$ Solution: " ++ padLeft 6 '0' (show solutionNumber) ++ "\n")
 1081                             appendFile filenameEprimeSol  (render lineWidth eprimeSol  ++ "\n\n")
 1082                             appendFile filenameEssenceSol ("$ Solution: " ++ padLeft 6 '0' (show solutionNumber) ++ "\n")
