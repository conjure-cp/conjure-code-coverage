   49 mkSolveStats :: UI -> String -> Text -> IO SolveStats
   50 mkSolveStats Solve {..} raw stdout = do
   51   let info = M.fromList [(k, v) | [k, v] <- map (splitOn ":") (lines raw)]
   52       status
   53         | M.lookup "SavileRowTimeOut" info == Just "1" = TimeOut
   54         | M.lookup "SavileRowClauseOut" info == Just "1" = TimeOut
   55         | M.lookup "SolverTimeOut" info == Just "1" = TimeOut
   56         | T.isInfixOf "Savile Row timed out." stdout = TimeOut
   57         | T.isInfixOf "java.lang.OutOfMemoryError" stdout = MemOut
   58         | otherwise = OK
   59       totalTime
   60         | Just srTotalTime <- M.lookup "SavileRowTotalTime" info >>= readMay,
   61           Just solverTotalTime <- M.lookup "SolverTotalTime" info >>= readMay =
   62             Just (srTotalTime + solverTotalTime)
   63         | otherwise = Nothing
   64       savilerowInfo = info
   65   computer <- getHostName
   66   timestamp <- getCurrentTime
   67   let conjureVersion = repositoryVersion
   68   savileRowVersion <- head . lines . textToString <$> sh (run "savilerow" ["-help"])
   69   return SolveStats {..}
   70 mkSolveStats _ _ _ = bug "mkSolveStats"
