   21 instance (TypeOf x, Pretty x) => TypeOf (OpTable x) where
   22     typeOf p@(OpTable vars table) = do
   23         tyVars <- typeOf vars
   24         tyTable <- typeOf table
   25         let
   26             tyError = raiseTypeError $ vcat [ pretty p
   27                                             , pretty vars <+> "has type" <+> pretty tyVars
   28                                             , pretty table <+> "has type" <+> pretty tyTable
   29                                             ]
   30             listLike (TypeList TypeInt{}) = return ()
   31             listLike (TypeMatrix _ TypeInt{}) = return ()
   32             listLike _ = tyError
   33         listLike tyVars
   34         case tyTable of
   35             TypeList inner -> listLike inner
   36             TypeMatrix _ inner -> listLike inner
   37             _ -> tyError
   38         return TypeBool
