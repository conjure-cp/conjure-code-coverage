   25 -- | Try to simplify an expression recursively.
   26 trySimplify ::
   27     MonadUserError m =>
   28     EnumerateDomain m =>
   29     NameGen m =>
   30     (?typeCheckerMode :: TypeCheckerMode) =>
   31     [(Name, Expression)] -> Expression -> m Expression
   32 trySimplify ctxt x = do
   33     res <- runMaybeT $ instantiateExpression ctxt x
   34     case res of
   35         Just c                                                  -- if the expression can be evaluated into a Constant
   36             | null [() | ConstantUndefined{} <- universe c]     -- and if it doesn't contain undefined's in it
   37             -> return (Constant c)                              -- evaluate to the constant
   38         _   -> descendM (trySimplify ctxt) x                    -- otherwise, try the same on its children

   41 instantiateExpression ::
   42     MonadFail m =>
   43     EnumerateDomain m =>
   44     NameGen m =>
   45     (?typeCheckerMode :: TypeCheckerMode) =>
   46     [(Name, Expression)] -> Expression -> m Constant
   47 instantiateExpression ctxt x = do
   48     constant <- normaliseConstant <$> evalStateT (instantiateE x) ctxt
   49     case (emptyCollection constant, constant) of
   50         (_, TypedConstant{}) -> return constant
   51         (True, _) -> do
   52             ty <- typeOf x
   53             return (TypedConstant constant ty)
   54         (False, _) -> return constant

   71 instantiateE ::
   72     MonadFail m =>
   73     MonadState [(Name, Expression)] m =>
   74     EnumerateDomain m =>
   75     NameGen m =>
   76     (?typeCheckerMode :: TypeCheckerMode) =>
   77     Expression -> m Constant
   78 instantiateE (Comprehension body gensOrConds) = do
   79     let
   80         loop ::
   81             MonadFail m =>
   82             MonadState [(Name, Expression)] m =>
   83             EnumerateDomain m =>
   84             NameGen m =>
   85             [GeneratorOrCondition] -> WriterT HasUndef m [Constant]
   86         loop [] = return <$> instantiateE body
   87         loop (Generator (GenDomainNoRepr pat domain) : rest) = do
   88             DomainInConstant domainConstant <- instantiateE (Domain domain)
   89             let undefinedsInsideTheDomain =
   90                     [ und
   91                     | und@ConstantUndefined{} <- universeBi domainConstant
   92                     ]
   93             if null undefinedsInsideTheDomain
   94                 then do
   95                     enumeration <- enumerateDomain domainConstant
   96                     concatMapM
   97                         (\ val -> scope $ do
   98                             valid <- bind pat val
   99                             if valid
  100                                 then loop rest
  101                                 else return [] )
  102                         enumeration
  103                 else do
  104                     tell (HasUndef (Any True))
  105                     return []
  106         loop (Generator (GenDomainHasRepr pat domain) : rest) =
  107             loop (Generator (GenDomainNoRepr (Single pat) (forgetRepr domain)) : rest)
  108         loop (Generator (GenInExpr pat expr) : rest) = do
  109             exprConstant <- instantiateE expr
  110             enumeration <- enumerateInConstant exprConstant
  111             concatMapM
  112                 (\ val -> scope $ do
  113                     valid <- bind pat val
  114                     if valid
  115                         then loop rest
  116                         else return [] )
  117                 enumeration
  118         loop (Condition expr : rest) = do
  119             constant <- instantiateE expr
  120             if constant == ConstantBool True
  121                 then loop rest
  122                 else return []
  123         loop (ComprehensionLetting pat expr : rest) = do
  124             constant <- instantiateE expr
  125             valid <- bind pat constant
  126             unless valid (bug "ComprehensionLetting.bind expected to be valid")
  127             loop rest
  128 
  129 
  130     (constants, HasUndef (Any undefinedsInsideGeneratorDomains)) <- runWriterT (loop gensOrConds)
  131     if undefinedsInsideGeneratorDomains
  132         then do
  133             ty <- typeOf (Comprehension body gensOrConds)
  134             return $ ConstantUndefined
  135                 "Comprehension contains undefined values inside generator domains."
  136                 ty
  137         else
  138             return $ ConstantAbstract $ AbsLitMatrix
  139                 (DomainInt TagInt [RangeBounded 1 (fromInt (genericLength constants))])
  140                 constants

  142 instantiateE (Reference name (Just (RecordField _ ty))) = return $ ConstantField name ty
  143 instantiateE (Reference name (Just (VariantField _ ty))) = return $ ConstantField name ty
  144 instantiateE (Reference _    (Just (Alias x))) = instantiateE x
  145 instantiateE (Reference name _) = do
  146     ctxt <- gets id
  147     case name `lookup` ctxt of
  148         Nothing -> fail $ vcat
  149             $ ("No value for:" <+> pretty name)
  150             : "Bindings in context:"
  151             : prettyContext ctxt
  152         Just x -> instantiateE x
  153 
  154 instantiateE (Constant c) = return c
  155 instantiateE (AbstractLiteral lit) = instantiateAbsLit lit
  156 instantiateE (Typed x ty) = TypedConstant <$> instantiateE x <*> pure ty
  157 instantiateE (Op op) = instantiateOp op
  158 
  159 -- "Domain () Expression"s inside expressions are handled specially
  160 instantiateE (Domain (DomainReference _ (Just d))) = instantiateE (Domain d)
  161 instantiateE (Domain (DomainReference name Nothing)) = do
  162     ctxt <- gets id
  163     case name `lookup` ctxt of
  164         Just (Domain d) -> instantiateE (Domain d)
  165         _ -> fail $ vcat
  166             $ ("No value for:" <+> pretty name)
  167             : "Bindings in context:"
  168             : prettyContext ctxt
  169 instantiateE (Domain domain) = DomainInConstant <$> instantiateD domain
  170 
  171 instantiateE (WithLocals b (AuxiliaryVars locals)) = do
  172     forM_ locals $ \ local -> case local of
  173         SuchThat xs -> forM_ xs $ \ x -> do
  174             constant <- instantiateE x
  175             case constant of
  176                 ConstantBool True -> return ()
  177                 _                 -> fail $ "local:" <+> pretty constant
  178         _ -> fail $ "local:" <+> pretty local
  179     instantiateE b
  180 
  181 instantiateE (WithLocals b (DefinednessConstraints locals)) = do
  182     forM_ locals $ \ x -> do
  183             constant <- instantiateE x
  184             case constant of
  185                 ConstantBool True -> return ()
  186                 _                 -> fail $ "local:" <+> pretty constant
  187     instantiateE b
  188 
  189 instantiateE x = fail $ "instantiateE:" <+> pretty (show x)

  202 instantiateAbsLit ::
  203     MonadFail m =>
  204     MonadState [(Name, Expression)] m =>
  205     EnumerateDomain m =>
  206     NameGen m =>
  207     (?typeCheckerMode :: TypeCheckerMode) =>
  208     AbstractLiteral Expression -> m Constant
  209 instantiateAbsLit x = do
  210     c <- mapM instantiateE x
  211     case c of
  212         -- for functions, if the same thing is mapped to multiple values, the result is undefined
  213         AbsLitFunction vals -> do
  214             let nubVals = sortNub vals
  215             if length (sortNub (map fst nubVals)) == length nubVals
  216                 then return $ ConstantAbstract $ AbsLitFunction nubVals
  217                 else do
  218                     ty <- typeOf c
  219                     return $ ConstantUndefined "Multiple mappings for the same value." ty
  220         _ -> return $ ConstantAbstract c

  223 instantiateD ::
  224     MonadFail m =>
  225     MonadState [(Name, Expression)] m =>
  226     EnumerateDomain m =>
  227     NameGen m =>
  228     Pretty r =>
  229     Default r =>
  230     (?typeCheckerMode :: TypeCheckerMode) =>
  231     Domain r Expression -> m (Domain r Constant)
  232 instantiateD (DomainAny t ty) = return (DomainAny t ty)
  233 instantiateD DomainBool = return DomainBool
  234 instantiateD (DomainIntE x) = do
  235     x' <- instantiateE x
  236     let vals = case (x', viewConstantMatrix x', viewConstantSet x') of
  237                 (ConstantInt{}, _, _) -> [x']
  238                 (_, Just (_, xs), _) -> xs
  239                 (_, _, Just xs) -> xs
  240                 _ -> []
  241     return (DomainInt TagInt (map RangeSingle vals))
  242 instantiateD (DomainInt t ranges) = DomainInt t <$> mapM instantiateR ranges
  243 instantiateD (DomainEnum nm Nothing _) = do
  244     st <- gets id
  245     case lookup nm st of
  246         Just (Domain dom) -> instantiateD (defRepr dom)
  247         Just _  -> fail $ ("DomainEnum not found in state, Just:" <+> pretty nm) <++> vcat (map pretty st)
  248         Nothing -> fail $ ("DomainEnum not found in state, Nothing:" <+> pretty nm) <++> vcat (map pretty st)
  249 instantiateD (DomainEnum nm rs0 _) = do
  250     let fmap4 = fmap . fmap . fmap . fmap
  251     let e2c' x = either bug id (e2c x)
  252     rs <- transformBiM (\ x -> Constant <$> instantiateE x ) (rs0 :: Maybe [Range Expression])
  253                 |> fmap4 e2c'
  254     st <- gets id
  255     mp <- forM (universeBi rs :: [Name]) $ \ n -> case lookup n st of
  256             Just (Constant (ConstantInt _ i)) -> return (n, i)
  257             Nothing -> fail $ "No value for member of enum domain:" <+> pretty n
  258             Just c  -> fail $ vcat [ "Incompatible value for member of enum domain:" <+> pretty nm
  259                                    , "    Looking up for member:" <+> pretty n
  260                                    , "    Expected an integer, but got:" <+> pretty c
  261                                    ]
  262     return (DomainEnum nm (rs :: Maybe [Range Constant]) (Just mp))
  263 instantiateD (DomainUnnamed nm s) = DomainUnnamed nm <$> instantiateE s
  264 instantiateD (DomainTuple inners) = DomainTuple <$> mapM instantiateD inners
  265 instantiateD (DomainRecord  inners) = DomainRecord  <$> sequence [ do d' <- instantiateD d ; return (n,d')
  266                                                                  | (n,d) <- inners ]
  267 instantiateD (DomainVariant inners) = DomainVariant <$> sequence [ do d' <- instantiateD d ; return (n,d')
  268                                                                  | (n,d) <- inners ]
  269 instantiateD (DomainMatrix index inner) = DomainMatrix <$> instantiateD index <*> instantiateD inner
  270 instantiateD (DomainSet       r attrs inner) = DomainSet r <$> instantiateSetAttr attrs <*> instantiateD inner
  271 instantiateD (DomainMSet      r attrs inner) = DomainMSet r <$> instantiateMSetAttr attrs <*> instantiateD inner
  272 instantiateD (DomainFunction  r attrs innerFr innerTo) = DomainFunction r <$> instantiateFunctionAttr attrs <*> instantiateD innerFr <*> instantiateD innerTo
  273 instantiateD (DomainSequence  r attrs inner) = DomainSequence r <$> instantiateSequenceAttr attrs <*> instantiateD inner
  274 instantiateD (DomainRelation  r attrs inners) = DomainRelation r <$> instantiateRelationAttr attrs <*> mapM instantiateD inners
  275 instantiateD (DomainPartition r attrs inner) = DomainPartition r <$> instantiatePartitionAttr attrs <*> instantiateD inner
  276 instantiateD (DomainOp nm ds) = DomainOp nm <$> mapM instantiateD ds
  277 instantiateD (DomainReference _ (Just d)) = instantiateD d
  278 instantiateD (DomainReference name Nothing) = do
  279     ctxt <- gets id
  280     case name `lookup` ctxt of
  281         Just (Domain d) -> instantiateD (defRepr d)
  282         _ -> fail $ vcat
  283             $ ("No value for:" <+> pretty name)
  284             : "Bindings in context:"
  285             : prettyContext ctxt
  286 instantiateD DomainMetaVar{} = bug "instantiateD DomainMetaVar"

  401 bind :: (Functor m, MonadState [(Name, Expression)] m)
  402     => AbstractPattern
  403     -> Constant
  404     -> m Bool -- False means skip
  405 bind (Single nm) val = modify ((nm, Constant val) :) >> return True
  406 bind (AbsPatTuple  pats) (ConstantAbstract (AbsLitTuple    vals))
  407     | length pats == length vals = and <$> zipWithM bind pats vals
  408 bind (AbsPatMatrix pats) (ConstantAbstract (AbsLitMatrix _ vals))
  409     | length pats == length vals = and <$> zipWithM bind pats vals
  410 bind (AbsPatSet    pats) (ConstantAbstract (AbsLitSet      vals))
  411     | length pats == length vals = and <$> zipWithM bind pats vals
  412     | otherwise                  = return False
  413 bind pat val = bug $ "Instantiate.bind:" <++> vcat ["pat:" <+> pretty pat, "val:" <+> pretty val]

  416 -- check if the given expression can be evaluated to True
  417 -- False means it is not entailed, as opposed to "it is known to be false"
  418 entailed ::
  419     MonadUserError m =>
  420     EnumerateDomain m =>
  421     NameGen m =>
  422     (?typeCheckerMode :: TypeCheckerMode) =>
  423     Expression -> m Bool
  424 entailed x = do
  425     -- traceM $ show $ "entailed x:" <+> pretty x
  426     c <- trySimplify [] x
  427     -- traceM $ show $ "entailed c:" <+> pretty c
  428     case c of
  429         Constant (ConstantBool True) -> return True
  430         _                            -> return False
