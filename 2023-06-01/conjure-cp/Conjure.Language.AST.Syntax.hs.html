   55 data StatementNode

  102 -- Declaration statements
  103 data DeclarationStatementNode
  104     = FindStatement SToken (Sequence FindStatementNode)

  109 instance Pretty DeclarationStatementNode where
  110     pretty x = case x of
  111         FindStatement lt se -> topLevelPretty [RealToken lt] (pretty se)
  112         GivenStatement lt se -> topLevelPretty [RealToken lt] (pretty se)
  113         LettingStatement lt se -> topLevelPretty [RealToken lt] (pretty se)
  114 data FindStatementNode
  115     = FindStatementNode
  116         (Sequence NameNode) -- names
  117         LToken -- colon
  118         DomainNode -- domain
  119     deriving (Show)
  120 instance Pretty FindStatementNode where
  121     pretty (FindStatementNode names col dom) = pretty names <+> pretty col <+> pretty dom
  122 instance Null FindStatementNode where
  123     isMissing (FindStatementNode n l d) = isMissing n && isMissing l && isMissing d
  124 data GivenStatementNode
  125     = GivenStatementNode
  126         (Sequence NameNode) -- name
  127         LToken -- colon
  128         DomainNode -- domain
  129     | GivenEnumNode
  130         (Sequence NameNode)
  131         LToken -- new
  132         LToken -- type
  133         LToken -- enum
  134     deriving (Show)
  135 instance Pretty GivenStatementNode where
  136     pretty g = case g of
  137         GivenStatementNode se lt dn -> pretty se <+> pretty lt <+> pretty dn
  138         GivenEnumNode se lt lt' lt2 -> pretty se <+> pretty lt <+> pretty lt' <+> pretty lt2
  139 
  140 instance Null GivenStatementNode where
  141     isMissing (GivenStatementNode l t d) = isMissing l && isMissing t && isMissing d
  142     isMissing (GivenEnumNode l a b c) = isMissing l && isMissing a && isMissing b && isMissing c

  144 data LettingStatementNode

  155 data LettingAssignmentNode
  156     = LettingExpr
  157         ExpressionNode
  158     | LettingDomain
  159         SToken -- domain
  160         DomainNode
  161     | LettingEnum
  162         LToken -- lNew
  163         LToken -- lType
  164         LToken -- lEnum
  165         (ListNode NameNode) -- nameList
  166     | LettingAnon
  167         LToken -- lNew
  168         LToken -- lType
  169         LToken -- lOf
  170         LToken -- lSize
  171         ExpressionNode -- expr
  172     deriving (Show)
  173 
  174 instance Pretty LettingAssignmentNode where
  175     pretty a = case a of
  176         LettingExpr en -> pretty en
  177         LettingDomain lt dn -> pretty lt <+> pretty dn
  178         LettingEnum lt lt' lt2 ln -> pretty lt <+> pretty lt' <+> pretty lt2 <+> pretty ln
  179         LettingAnon lt lt' lt2 lt3 en -> pretty lt <+> pretty lt' <+> pretty lt2 <+> pretty lt3 <+> pretty en
  180 instance Null LettingAssignmentNode where
  181     isMissing x = case x of
  182         LettingExpr en -> isMissing en
  183         LettingDomain lt dn -> isMissing lt && isMissing dn
  184         LettingEnum l1 l2 l3 ln -> all isMissing [l1, l2, l3] && isMissing ln
  185         LettingAnon l1 l2 l3 l4 en -> all isMissing [l1, l2, l3, l4] && isMissing en

  224     pretty x = case x of
  225         ParenDomainNode op dom cl -> pretty op <> pretty dom <> pretty cl
  226         BoolDomainNode lt -> pretty lt
  227         RangedIntDomainNode lt m_ln -> pretty lt <> pretty m_ln
  228         RangedEnumNode nn m_ln -> pretty nn <> pretty m_ln

  233         VariantDomainNode lt ln -> pretty lt <> pretty ln
  234         MatrixDomainNode lt m_ibn ln lt' dn ->
  235             pretty lt
  236                 <+> pretty m_ibn
  237                 <+> pretty ln
  238                 <+> pretty lt'
  239                 <+> pretty dn

  243         SequenceDomainNode lt m_ln lt' dn -> pretty lt <+> pretty m_ln <+> pretty lt' <+> pretty dn
  244         RelationDomainNode lt m_ln lt' ln -> pretty lt <+> pretty m_ln <+> pretty lt' <+> pretty ln
  245         PartitionDomainNode lt m_ln lt' dn -> pretty lt <+> pretty m_ln <+> pretty lt' <+> pretty dn

  412 data MatrixLiteralNode
  413     = MatrixLiteralNode
  414         LToken -- openBracket
  415         (Sequence ExpressionNode)

  421 instance Pretty MatrixLiteralNode where
  422     pretty (MatrixLiteralNode bl es d c br) =
  423         group $
  424             align (cat (pretty bl : prettyElems es ++ catMaybes ((pretty <$> d) : comps) ++ [pretty br]))
  425       where
  426         comps = case c of
  427             Nothing -> []
  428             Just (ComprehensionNode l seq) -> pure <$> pretty l : prettyElems seq

  645     deriving (Show)
  646 instance Pretty a => Pretty (SeqElem a) where
  647     pretty (SeqElem i s) = pretty i <> pretty s
  648     pretty _ = emptyDoc
  649 
  650 class Null a where
  651     isMissing :: a -> Bool
  652 
  653 instance (Null a) => Null (Maybe a) where
  654     isMissing Nothing = True
  655     isMissing (Just s) = isMissing s
  656 
  657 prettyTokenAndComments :: LToken -> (Doc ann, Doc ann)
  658 prettyTokenAndComments (RealToken (StrictToken [] t)) = prettySplitComments t
  659 prettyTokenAndComments (o) = (emptyDoc, pretty o)
  660 
  661 topLevelPretty :: [LToken] -> Doc ann -> Doc ann
  662 topLevelPretty (t : (map pretty -> xs)) exprs =
  663     let (cs, ps) = prettyTokenAndComments t
  664         dec = ps <+> hsep xs
  665      in cs <> group (fill 7 dec <+> flatIndent 4 exprs) <> line
  666 topLevelPretty _ exprs = group (fill 7 emptyDoc <+> flatIndent 4 exprs) <> line
  667 
  668 flatIndent :: Int -> Doc ann -> Doc ann
  669 flatIndent amt d = flatAlt (line <> indent amt d) d
  670 
  671 renderAST :: Int -> ProgramTree -> Text
  672 renderAST n = renderStrict . layoutSmart (LayoutOptions $ AvailablePerLine n 0.8) . pretty
