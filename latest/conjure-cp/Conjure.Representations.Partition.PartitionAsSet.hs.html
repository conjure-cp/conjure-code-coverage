   20 partitionAsSet
   21     :: forall m . (MonadFail m, NameGen m, ?typeCheckerMode :: TypeCheckerMode)
   22     => (forall x . DispatchFunction m x)
   23     -> (forall r x . ReprOptionsFunction m r x)
   24     -> Bool
   25     -> Representation m
   26 partitionAsSet dispatch reprOptions useLevels = Representation chck downD structuralCons downC up symmetryOrdering
   27 
   28     where
   29 
   30         chck :: TypeOf_ReprCheck m
   31         chck _ dom1@(DomainPartition _ attrs _) = do
   32             -- this is a "lookahead"
   33             -- do the horizontal representation move: go from "partition of T" to "set of set of T"
   34             -- do representation selection on the set
   35             -- lookup the chosen representations and store them inside Partition_AsSet
   36             dom2 <- outDomain_ dom1
   37             dom3 <- reprOptions dom2
   38             return [ DomainPartition (Partition_AsSet r1 r2) attrs innerDomain
   39                    | DomainSet r1 _ (DomainSet r2 _ innerDomain) <- dom3
   40                    -- special hack: do not use Set_ExplicitVarSizeWithFlags when --representation-levels=yes
   41                    , if useLevels
   42                        then r1 /= Set_ExplicitVarSizeWithFlags && r2 /= Set_ExplicitVarSizeWithFlags
   43                        else True
   44                    ]
   45         chck _ _ = return []
   46 
   47         outName :: Domain HasRepresentation x -> Name -> Name
   48         outName = mkOutName Nothing
   49 
   50         outDomain_ :: Pretty x => Domain () x -> m (Domain () x)
   51         outDomain_ (DomainPartition () PartitionAttr{..} innerDomain) =
   52             return (DomainSet () (SetAttr partsNum) (DomainSet () (SetAttr partsSize) innerDomain))
   53         outDomain_ domain = na $ vcat [ "{outDomain_} PartitionAsSet"
   54                                       , "domain:" <+> pretty domain
   55                                       ]
   56 
   57         outDomain :: Pretty x => Domain HasRepresentation x -> m (Domain HasRepresentation x)
   58         outDomain (DomainPartition (Partition_AsSet repr1 repr2) PartitionAttr{..} innerDomain) =
   59             return (DomainSet repr1 (SetAttr partsNum) (DomainSet repr2 (SetAttr partsSize) innerDomain))
   60         outDomain domain = na $ vcat [ "{outDomain} PartitionAsSet"
   61                                      , "domain:" <+> pretty domain
   62                                      ]
   63 
   64         downD :: TypeOf_DownD m
   65         downD (name, inDom) = do
   66             outDom <- outDomain inDom
   67             return $ Just [ ( outName inDom name , outDom ) ]
   68 
   69         structuralCons :: TypeOf_Structural m
   70         structuralCons f downX1 inDom@(DomainPartition r attrs innerDomain) = return $ \ inpRel -> do
   71             refs <- downX1 inpRel
   72             let
   73 
   74                 fixedPartSize =
   75                     case attrs of
   76                         PartitionAttr _ SizeAttr_Size{} _ -> True
   77                         _                                 -> False
   78 
   79                 exactlyOnce :: Expression -> m [Expression]
   80                 exactlyOnce rel = do
   81                     innerType <- typeOfDomain innerDomain
   82                     let useAllDiff =
   83                             case r of
   84                                 -- we use a sum when the inner set is occurrence
   85                                 Partition_AsSet _ Set_Occurrence -> False
   86                                 _ ->
   87                                     case innerType of
   88                                         TypeInt{} -> True
   89                                         _         -> False      -- or if the inner type is not int
   90 
   91                     (iPat, i) <- quantifiedVar
   92                     (jPat, j) <- quantifiedVar
   93                     if useAllDiff
   94                         then return $ return $ -- for list
   95                                 [essence|
   96                                     allDiff([ &j
   97                                             | &iPat <- &rel
   98                                             , &jPat <- &i
   99                                             ])
  100                                         |]
  101                         else return $ return $ -- for list
  102                                 [essence|
  103                                     forAll &iPat : &innerDomain .
  104                                         1  = sum ([ 1
  105                                                   | &jPat <- &rel
  106                                                   , &i in &j
  107                                                   ])
  108                                         |]
  109 
  110                 regular :: Expression -> m [Expression]
  111                 regular rel | isRegular attrs && not fixedPartSize = do
  112                     (iPat, i) <- quantifiedVar
  113                     (jPat, j) <- quantifiedVar
  114                     return $ return -- for list
  115                         [essence|
  116                             and([ |&i| = |&j|
  117                                 | &iPat <- &rel
  118                                 , &jPat <- &rel
  119                                 ])
  120                         |]
  121                 regular _ = return []
  122 
  123                 partsAren'tEmpty :: Expression -> m [Expression]
  124                 partsAren'tEmpty rel = do
  125                     (iPat, i) <- quantifiedVar
  126                     return $ return [essence| and([ |&i| >= 1 | &iPat <- &rel ]) |]
  127 
  128                 sumOfParts :: Expression -> m [Expression]
  129                 sumOfParts rel = do
  130                     case domainSizeOf innerDomain of
  131                         Left _err -> return []
  132                         Right n   -> do
  133                             (iPat, i) <- quantifiedVar
  134                             return $ return [essence| &n = sum([ |&i| | &iPat <- &rel ]) |]
  135 
  136             case refs of
  137                 [rel] -> do
  138                     outDom                 <- outDomain inDom
  139                     innerStructuralConsGen <- f outDom
  140                     concat <$> sequence
  141                         [ exactlyOnce rel
  142                         , regular rel
  143                         , partsAren'tEmpty rel
  144                         , innerStructuralConsGen rel
  145                         , sumOfParts rel
  146                         ]
  147                 _ -> na $ vcat [ "{structuralCons} PartitionAsSet"
  148                                , pretty inDom
  149                                ]
  150         structuralCons _ _ domain = na $ vcat [ "{structuralCons} PartitionAsSet"
  151                                               , "domain:" <+> pretty domain
  152                                               ]
  153 
  154         downC :: TypeOf_DownC m
  155         downC ( name
  156               , inDom
  157               , ConstantAbstract (AbsLitPartition vals)
  158               ) = do
  159             outDom <- outDomain inDom
  160             rDownC
  161                 (dispatch outDom)
  162                 ( outName inDom name
  163                 , outDom
  164                 , ConstantAbstract $ AbsLitSet $ map (ConstantAbstract . AbsLitSet) vals
  165                 )
  166         downC (name, domain, constant) = na $ vcat [ "{downC} PartitionAsSet"
  167                                                    , "name:" <+> pretty name
  168                                                    , "domain:" <+> pretty domain
  169                                                    , "constant:" <+> pretty constant
  170                                                    ]
  171 
  172         up :: TypeOf_Up m
  173         up ctxt (name, domain@(DomainPartition Partition_AsSet{} _ _)) =
  174             case lookup (outName domain name) ctxt of
  175                 Nothing -> fail $ vcat $
  176                     [ "(in PartitionAsSet up)"
  177                     , "No value for:" <+> pretty (outName domain name)
  178                     , "When working on:" <+> pretty name
  179                     , "With domain:" <+> pretty domain
  180                     ] ++
  181                     ("Bindings in context:" : prettyContext ctxt)
  182                 Just (viewConstantSet -> Just sets) -> do
  183                     let setOut (viewConstantSet -> Just xs) = return xs
  184                         setOut c = fail $ "Expecting a set, but got:" <++> pretty c
  185                     vals <- mapM setOut sets
  186                     return (name, ConstantAbstract (AbsLitPartition vals))
  187                 Just (ConstantUndefined msg ty) ->        -- undefined propagates
  188                     return (name, ConstantUndefined ("PartitionAsSet " `mappend` msg) ty)
  189                 Just constant -> fail $ vcat $
  190                     [ "Incompatible value for:" <+> pretty (outName domain name)
  191                     , "When working on:" <+> pretty name
  192                     , "With domain:" <+> pretty domain
  193                     , "Expected a set value, but got:" <++> pretty constant
  194                     ] ++
  195                     ("Bindings in context:" : prettyContext ctxt)
  196         up _ (name, domain) = na $ vcat [ "{up} PartitionAsSet"
  197                                         , "name:" <+> pretty name
  198                                         , "domain:" <+> pretty domain
  199                                         ]
  200 
  201         symmetryOrdering :: TypeOf_SymmetryOrdering m
  202         symmetryOrdering innerSO downX1 inp domain = do
  203             [inner] <- downX1 inp
  204             Just [(_, innerDomain)] <- downD ("SO", domain)
  205             innerSO downX1 inner innerDomain
