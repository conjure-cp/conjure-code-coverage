   31 -- | This class is only to track where `enumerateDomain` might get called.
   32 --   It is essentially MonadIO, but doesn't allow arbitrary IO.
   33 class (Functor m, Applicative m, Monad m, MonadUserError m) => EnumerateDomain m where liftIO' :: IO a -> m a
   34 instance EnumerateDomain IO where liftIO' = id
   35 instance EnumerateDomain m => EnumerateDomain (IdentityT m) where liftIO' = lift . liftIO'
   36 instance EnumerateDomain m => EnumerateDomain (MaybeT m) where liftIO' = lift . liftIO'
   37 instance EnumerateDomain m => EnumerateDomain (ExceptT m) where liftIO' = lift . liftIO'
   38 instance EnumerateDomain m => EnumerateDomain (ReaderT r m) where liftIO' = lift . liftIO'
   39 instance (EnumerateDomain m, Monoid w) => EnumerateDomain (WriterT w m) where liftIO' = lift . liftIO'
   40 instance EnumerateDomain m => EnumerateDomain (StateT st m) where liftIO' = lift . liftIO'
   41 instance EnumerateDomain m => EnumerateDomain (Pipes.Proxy a b c d m) where liftIO' = lift . liftIO'
   42 instance EnumerateDomain m => EnumerateDomain (NameGenM m) where liftIO' = lift . liftIO'
   43 instance (EnumerateDomain m, MonadFail m) => EnumerateDomain (UserErrorT m) where liftIO' = lift . liftIO'

   45 -- | Use this if you don't want to allow a (EnumerateDomain m => m a) computation actually do IO.
   46 data EnumerateDomainNoIO a = Done a | TriedIO | Failed Doc
   47     deriving (Eq, Show)
   48 
   49 instance Functor EnumerateDomainNoIO where
   50     fmap _ (Failed msg) = Failed msg
   51     fmap _ TriedIO      = TriedIO
   52     fmap f (Done x)     = Done (f x)
   53 
   54 instance Applicative EnumerateDomainNoIO where
   55     pure = return
   56     (<*>) = ap
   57 
   58 instance Monad EnumerateDomainNoIO where
   59     return = Done
   60     Failed msg >>= _ = Failed msg
   61     TriedIO    >>= _ = TriedIO
   62     Done x     >>= f = f x
   63 
   64 instance MonadFailDoc EnumerateDomainNoIO where
   65     failDoc = Failed
   66 instance MonadFail EnumerateDomainNoIO where
   67     fail = Failed . stringToDoc 
   68 
   69 instance MonadUserError EnumerateDomainNoIO where
   70     userErr docs = Failed (vcat $ "User error:" : docs)
   71 
   72 instance NameGen EnumerateDomainNoIO where
   73     nextName _ = failDoc "nextName{EnumerateDomainNoIO}"
   74     exportNameGenState = failDoc "exportNameGenState{EnumerateDomainNoIO}"
   75     importNameGenState _ = failDoc "importNameGenState{EnumerateDomainNoIO}"
   76 
   77 instance EnumerateDomain EnumerateDomainNoIO where liftIO' _ = TriedIO

   88 enumerateDomain :: (MonadFailDoc m, EnumerateDomain m) => Domain () Constant -> m [Constant]
   89 
   90 enumerateDomain d | not (null [ () | ConstantUndefined{} <- universeBi d ]) =
   91     bug $ vcat [ "called enumerateDomain with a domain that has undefinedness values in it."
   92                , pretty d
   93                ]
   94 
   95 enumerateDomain DomainBool = return [ConstantBool False, ConstantBool True]
   96 enumerateDomain (DomainInt _ []) = failDoc "enumerateDomain: infinite domain"
   97 enumerateDomain (DomainInt _ rs) = concatMapM enumerateRange rs
   98 enumerateDomain (DomainUnnamed _ (ConstantInt t n)) = return (map (ConstantInt t) [1..n])
   99 enumerateDomain (DomainEnum _dName (Just rs) _mp) = concatMapM enumerateRange rs
  100 enumerateDomain (DomainTuple ds) = do
  101     inners <- mapM enumerateDomain ds
  102     return $ map (ConstantAbstract . AbsLitTuple) (sequence inners)
  103 enumerateDomain (DomainMatrix (DomainInt t indexDom) innerDom) = do
  104     inners <- enumerateDomain innerDom
  105     indexInts <- rangesInts indexDom
  106     return
  107         [ ConstantAbstract (AbsLitMatrix (DomainInt t indexDom) vals)
  108         | vals <- replicateM (length indexInts) inners
  109         ]

  111 -- the sledgehammer approach
  112 enumerateDomain d = liftIO' $ withSystemTempDirectory ("conjure-enumerateDomain-" ++ show (hash d)) $ \ tmpDir -> do
  113     let model = Model { mLanguage = LanguageVersion "Essence" [1,0]
  114                       , mStatements = [Declaration (FindOrGiven Find "x" (fmap Constant d))]
  115                       , mInfo = def
  116                       }
  117     let essenceFile = tmpDir </> "out.essence"
  118     let outDir = tmpDir </> "outDir"
  119     writeModel 120 Plain (Just essenceFile) model
  120     let
  121         solve :: IO ()
  122         solve = let ?typeCheckerMode = StronglyTyped in ignoreLogs $ mainWithArgs Solve
  123             { UI.essence                    = essenceFile
  124             , validateSolutionsOpt          = False
  125             , outputDirectory               = outDir
  126             , savilerowOptions              =
  127                 [ "-O0"
  128                 , "-preprocess"    , "None"
  129                 , "-timelimit"     , show savilerowTimelimit
  130                 ]
  131             , solverOptions                 =
  132                 [ "-cpulimit"      , show minionTimelimit
  133                 ]
  134             , solver                        = "minion"
  135             , graphSolver                   = False
  136             , cgroups                       = False
  137             , nbSolutions                   = show enumerateDomainMax
  138             , copySolutions                 = False
  139             , solutionsInOneFile            = False
  140             , logLevel                      = LogNone
  141             -- default values for the rest
  142             , essenceParams                 = []
  143             , numberingStart                = 1
  144             , smartFilenames                = False
  145             , verboseTrail                  = False
  146             , rewritesTrail                 = False
  147             , logRuleFails                  = False
  148             , logRuleSuccesses              = False
  149             , logRuleAttempts               = False
  150             , logChoices                    = False
  151             , portfolio                     = Nothing
  152             , strategyQ                     = "f"
  153             , strategyA                     = "c"
  154             , representations               = Nothing
  155             , representationsFinds          = Nothing
  156             , representationsGivens         = Nothing
  157             , representationsAuxiliaries    = Nothing
  158             , representationsQuantifieds    = Nothing
  159             , representationsCuts           = Nothing
  160             , channelling                   = False
  161             , representationLevels          = True
  162             , followModel                   = ""
  163             , useExistingModels             = []
  164             , seed                          = Nothing
  165             , limitModels                   = Nothing
  166             , limitTime                     = Nothing
  167             , outputFormat                  = UI.Plain
  168             , lineWidth                     = 120
  169             , responses                     = ""
  170             , responsesRepresentation       = ""
  171             }
  172     -- catching the (SR timeout) error, and raising a user error
  173     catch solve $ \ (e :: SomeException) -> userErr1 $ vcat
  174         [ "Enumerate domain: too many."
  175         , "When working on domain:" <++> pretty d
  176         , "Exception:" <++> pretty (show e)
  177         ]
  178     solutions   <- filter (".solution" `isSuffixOf`) <$> getDirectoryContents outDir
  179     when (length solutions >= enumerateDomainMax) $ userErr1 $ vcat
  180         [ "Enumerate domain: too many."
  181         , "Gave up after" <+> pretty (length solutions) <+> "solutions."
  182         , "When working on domain:" <++> pretty d
  183         ]
  184     enumeration <- fmap concat $ forM solutions $ \ solutionFile -> do
  185         Model _ decls _ <- readModelFromFile (outDir </> solutionFile)
  186         let (enumeration, errs) = mconcat
  187                 [ case decl of
  188                     Declaration (Letting "x" x) | Just c <- e2c x -> ([c], [])
  189                     _ -> ([], [decl])
  190                 | decl <- decls ]
  191         if null errs
  192             then return enumeration
  193             else failDoc $ vcat $ "enumerateDomain, not Constants!"
  194                              : ("When working on domain:" <++> pretty d)
  195                              :  map pretty errs
  196                              ++ map (pretty . show) errs
  197     removeDirectoryIfExists outDir
  198     removeDirectoryIfExists tmpDir
  199     return enumeration

  202 enumerateRange :: MonadFailDoc m => Range Constant -> m [Constant]
  203 enumerateRange (RangeSingle x) = return [x]
  204 enumerateRange (RangeBounded (ConstantInt t x) (ConstantInt _ y)) = return $ ConstantInt t <$> [x..y]
  205 enumerateRange RangeBounded{} = failDoc "enumerateRange RangeBounded"
  206 enumerateRange RangeOpen{} = failDoc "enumerateRange RangeOpen"
  207 enumerateRange RangeLowerBounded{} = failDoc "enumerateRange RangeLowerBounded"
  208 enumerateRange RangeUpperBounded{} = failDoc "enumerateRange RangeUpperBounded"

  210 enumerateInConstant :: MonadFailDoc m => Constant -> m [Constant]
  211 enumerateInConstant constant = case constant of
  212     ConstantAbstract (AbsLitMatrix _  xs) -> return xs
  213     ConstantAbstract (AbsLitSet       xs) -> return xs
  214     ConstantAbstract (AbsLitMSet      xs) -> return xs
  215     ConstantAbstract (AbsLitFunction  xs) -> return [ ConstantAbstract (AbsLitTuple [i,j]) | (i,j) <- xs ]
  216     ConstantAbstract (AbsLitSequence  xs) -> return [ ConstantAbstract (AbsLitTuple [i,j])
  217                                                     | (i',j) <- zip allNats xs
  218                                                     , let i = fromInt i'
  219                                                     ]
  220     ConstantAbstract (AbsLitRelation  xs) -> return $ map (ConstantAbstract . AbsLitTuple) xs
  221     ConstantAbstract (AbsLitPartition xs) -> return $ map (ConstantAbstract . AbsLitSet) xs
  222     TypedConstant c _                     -> enumerateInConstant c
  223     _ -> failDoc $ vcat [ "enumerateInConstant"
  224                      , "constant:" <+> pretty constant
  225                      ]
