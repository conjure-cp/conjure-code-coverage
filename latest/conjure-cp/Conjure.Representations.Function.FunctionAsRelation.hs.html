   15 functionAsRelation
   16     :: forall m . (MonadFail m, NameGen m)
   17     => (forall x . DispatchFunction m x)
   18     -> (forall r x . ReprOptionsFunction m r x)
   19     -> Representation m
   20 functionAsRelation dispatch reprOptions = Representation chck downD structuralCons downC up symmetryOrdering
   21 
   22     where
   23 
   24         chck :: TypeOf_ReprCheck m
   25         chck _ dom1@(DomainFunction _ attrs _ _) = do
   26             dom2 <- outDomain_ dom1
   27             dom3 <- reprOptions dom2
   28             return [ DomainFunction (Function_AsRelation r) attrs innerDomainFr innerDomainTo
   29                    | DomainRelation r _ [innerDomainFr, innerDomainTo] <- dom3
   30                    ]
   31         chck _ _ = return []
   32 
   33         outName :: Domain HasRepresentation x -> Name -> Name
   34         outName = mkOutName Nothing
   35 
   36         outDomain_ :: Pretty x => Domain () x -> m (Domain () x)
   37         outDomain_ (DomainFunction ()
   38                     (FunctionAttr sizeAttr _partilityAttr _jectivityAttr)
   39                     innerDomainFr innerDomainTo) =
   40             return (DomainRelation () (RelationAttr sizeAttr def) [innerDomainFr, innerDomainTo])
   41         outDomain_ domain = na $ vcat [ "{outDomain_} FunctionAsRelation"
   42                                       , "domain:" <+> pretty domain
   43                                       ]
   44 
   45         outDomain :: Pretty x => Domain HasRepresentation x -> m (Domain HasRepresentation x)
   46         outDomain (DomainFunction (Function_AsRelation repr)
   47                     (FunctionAttr sizeAttr _partilityAttr _jectivityAttr)
   48                     innerDomainFr innerDomainTo) =
   49             return (DomainRelation repr (RelationAttr sizeAttr def) [innerDomainFr, innerDomainTo])
   50         outDomain domain = na $ vcat [ "{outDomain} FunctionAsRelation"
   51                                      , "domain:" <+> pretty domain
   52                                      ]
   53 
   54         downD :: TypeOf_DownD m
   55         downD (name, inDom) = do
   56             outDom <- outDomain inDom
   57             return $ Just [ ( outName inDom name , outDom ) ]
   58 
   59         structuralCons :: TypeOf_Structural m
   60         structuralCons f downX1
   61                 inDom@(DomainFunction _
   62                         (FunctionAttr _ partilityAttr jectivityAttr)
   63                         innerDomainFr innerDomainTo) = return $ \ func -> do
   64             refs <- downX1 func
   65             let
   66                 partialityCons rel =
   67                     case partilityAttr of
   68                         PartialityAttr_Partial -> do
   69                             (iPat, i) <- quantifiedVar
   70                             (jPat, j) <- quantifiedVar
   71                             return $ return $ -- list
   72                                 [essence|
   73                                     forAll &iPat : &innerDomainFr .
   74                                         1 >= sum([ 1
   75                                                 | &jPat <- &rel
   76                                                 , &j[1] = &i
   77                                                 ])
   78                                 |]
   79                         PartialityAttr_Total -> do
   80                             (iPat, i) <- quantifiedVar
   81                             (jPat, j) <- quantifiedVar
   82                             return $ return $ -- list
   83                                 [essence|
   84                                     forAll &iPat : &innerDomainFr .
   85                                         1 =  sum([ 1
   86                                                 | &jPat <- &rel
   87                                                 , &j[1] = &i
   88                                                 ])
   89                                 |]
   90 
   91                 jectivityCons rel = case jectivityAttr of
   92                     JectivityAttr_None       -> return []
   93                     JectivityAttr_Injective  -> injectiveCons rel
   94                     JectivityAttr_Surjective -> surjectiveCons rel
   95                     JectivityAttr_Bijective  -> (++) <$> injectiveCons rel <*> surjectiveCons rel
   96 
   97                 injectiveCons rel = do
   98                     (iPat, i) <- quantifiedVar
   99                     (jPat, j) <- quantifiedVar
  100                     return $ return $ -- list
  101                         [essence|
  102                             and([ &i[1] .< &j[1] -> &i[2] != &j[2]
  103                                 | &iPat <- &rel
  104                                 , &jPat <- &rel
  105                                 ])
  106                         |]
  107 
  108                 surjectiveCons rel = do
  109                     (iPat, i) <- quantifiedVar
  110                     (jPat, j) <- quantifiedVar
  111                     return $ return $ -- list
  112                         [essence|
  113                             forAll &iPat : &innerDomainTo .
  114                                 exists &jPat in &rel .
  115                                     &j[2] = &i
  116                         |]
  117 
  118             case refs of
  119                 [rel] -> do
  120                     outDom                 <- outDomain inDom
  121                     innerStructuralConsGen <- f outDom
  122                     concat <$> sequence
  123                         [ innerStructuralConsGen rel
  124                         , partialityCons rel
  125                         , jectivityCons rel
  126                         ]
  127                 _ -> na $ vcat [ "{structuralCons} FunctionAsRelation"
  128                                , pretty inDom
  129                                ]
  130         structuralCons _ _ inDom =
  131             na $ vcat [ "{structuralCons} FunctionAsRelation"
  132                       , pretty inDom
  133                       ]
  134 
  135         downC :: TypeOf_DownC m
  136         downC ( name
  137               , inDom
  138               , ConstantAbstract (AbsLitFunction vals)
  139               ) = do
  140             outDom <- outDomain inDom
  141             rDownC
  142                 (dispatch outDom)
  143                 ( outName inDom name
  144                 , outDom
  145                 , ConstantAbstract $ AbsLitRelation $ map (\ (a,b) -> [a,b] ) vals
  146                 )
  147         downC (name, domain, constant) = na $ vcat [ "{downC} FunctionAsRelation"
  148                                                    , "name:" <+> pretty name
  149                                                    , "domain:" <+> pretty domain
  150                                                    , "constant:" <+> pretty constant
  151                                                    ]
  152 
  153         up :: TypeOf_Up m
  154         up ctxt (name, domain@(DomainFunction Function_AsRelation{} _ _ _)) =
  155             case lookup (outName domain name) ctxt of
  156                 Just (ConstantAbstract (AbsLitRelation pairs)) -> do
  157                     let pairOut [a,b] = return (a,b)
  158                         pairOut c = fail $ "Expecting a 2-tuple, but got:" <++> prettyList prParens "," c
  159                     vals <- mapM pairOut pairs
  160                     return (name, ConstantAbstract (AbsLitFunction vals))
  161                 Nothing -> fail $ vcat $
  162                     [ "(in FunctionAsRelation up)"
  163                     , "No value for:" <+> pretty (outName domain name)
  164                     , "When working on:" <+> pretty name
  165                     , "With domain:" <+> pretty domain
  166                     ] ++
  167                     ("Bindings in context:" : prettyContext ctxt)
  168                 Just constant -> fail $ vcat $
  169                     [ "Incompatible value for:" <+> pretty (outName domain name)
  170                     , "When working on:" <+> pretty name
  171                     , "With domain:" <+> pretty domain
  172                     , "Expected a set value, but got:" <++> pretty constant
  173                     ] ++
  174                     ("Bindings in context:" : prettyContext ctxt)
  175         up _ (name, domain) = na $ vcat [ "{up} FunctionAsRelation"
  176                                         , "name:" <+> pretty name
  177                                         , "domain:" <+> pretty domain
  178                                         ]
  179 
  180         symmetryOrdering :: TypeOf_SymmetryOrdering m
  181         symmetryOrdering innerSO downX1 inp domain = do
  182             [rel] <- downX1 inp
  183             Just [(_, relDomain)] <- downD ("SO", domain)
  184             innerSO downX1 rel relDomain
