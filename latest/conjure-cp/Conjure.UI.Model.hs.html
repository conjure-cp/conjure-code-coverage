  123 outputModels ::
  124     forall m .
  125     MonadIO m =>
  126     MonadFail m =>
  127     MonadLog m =>
  128     NameGen m =>
  129     EnumerateDomain m =>
  130     MonadUserError m =>
  131     (?typeCheckerMode :: TypeCheckerMode) =>
  132     Maybe Int ->                -- portfolioSize
  133     S.Set Int ->                -- modelHashesBefore
  134     String ->                   -- modelNamePrefix
  135     Config ->
  136     Model ->
  137     m (S.Set Int)               -- hash values, identifying the models
  138 outputModels portfolioSize modelHashesBefore modelNamePrefix config model = do
  139 
  140     liftIO $ writeIORef recordedResponses (responses config)
  141     liftIO $ writeIORef recordedResponsesRepresentation (responsesRepresentation config)
  142 
  143     -- Savile Row does not support ' characters in identifiers
  144     -- We could implement a workaround where we insert a marker (like __PRIME__) for each ' character
  145     -- and recover these after a solution is found.
  146     -- But this will be too hairy, instead we will reject such identifiers for now.
  147     -- If somebody really needs to use a ' character as part of an identifier, we can revisit this decision.
  148     let
  149         primeyIdentifiers = catMaybes
  150             [ if '\'' `elem` textToString identifier
  151                 then Just identifier
  152                 else Nothing
  153             | Declaration decl <- mStatements model
  154             , Name identifier <- universeBi decl
  155             ]
  156     unless (null primeyIdentifiers) $ userErr1 $ vcat
  157         ["Identifiers cannot contain a quotation mark character in them:" <+> prettyList id "," primeyIdentifiers]
  158 
  159     let dir = outputDirectory config
  160 
  161     unless (estimateNumberOfModels config) $
  162         liftIO $ createDirectoryIfMissing True dir
  163 
  164     let
  165         limitModelsIfEstimating :: Pipe LogOrModel LogOrModel m ()
  166         limitModelsIfEstimating =
  167             if estimateNumberOfModels config
  168                 then limitModelsNeeded 1
  169                 else Pipes.cat
  170 
  171         limitModelsIfNeeded :: Pipe LogOrModel LogOrModel m ()
  172         limitModelsIfNeeded = maybe Pipes.cat limitModelsNeeded (limitModels config)
  173 
  174         limitModelsNeeded :: Int -> Pipe LogOrModel LogOrModel m ()
  175         limitModelsNeeded 0 = return ()
  176         limitModelsNeeded n = do
  177             x <- Pipes.await
  178             Pipes.yield x
  179             case x of
  180                 Left {} -> limitModelsNeeded n              -- yielded a log, still n models to produce
  181                 Right{} -> limitModelsNeeded (n-1)          -- yielded a model, produce n-1 more models
  182 
  183         limitModelsPortfolioSize :: Pipe LogOrModel LogOrModel m ()
  184         limitModelsPortfolioSize =
  185             case portfolioSize of
  186                 Nothing -> Pipes.cat
  187                 Just s -> do
  188                     nb <- liftIO (readIORef nbGeneratedModels)
  189                     if nb < s
  190                         then do
  191                             x <- Pipes.await
  192                             Pipes.yield x
  193                             limitModelsPortfolioSize
  194                         else do
  195                             log LogInfo $ "Stopping, generated" <+> pretty nb <+> "models."
  196                             return ()
  197 
  198         each (modelHashes, i) logOrModel =
  199             case logOrModel of
  200                 Left (l,msg) -> do
  201                     log l msg
  202                     return (modelHashes, i)
  203                 Right eprime -> do
  204                     let newHash = hash eprime { mInfo = def, mStatements = sort (mStatements eprime) }
  205                     let gen =
  206                             if modelNamePrefix `elem` ["01_compact", "02_sparse"]
  207                                 then modelNamePrefix
  208                                 else modelNamePrefix ++
  209                                         if smartFilenames config
  210                                             then [ choice
  211                                                  | (_question, choice, numOptions) <-
  212                                                          eprime |> mInfo |> miTrailCompact
  213                                                  , numOptions > 1
  214                                                  ] |> map (('_':) . show)
  215                                                    |> concat
  216                                             else padLeft 6 '0' (show i)
  217                     let filename = dir </> gen ++ ".eprime"
  218                     if S.member newHash modelHashes
  219                         then do
  220                             log LogInfo $ "Skipping duplicate model (" <> pretty filename <> ")"
  221                             return (modelHashes, i)
  222                         else do
  223                             if estimateNumberOfModels config
  224                                 then do
  225                                     let
  226                                         estimate :: Integer
  227                                         estimate = product $ 1 : [ toInteger numOptions
  228                                                                  | (_question, _choice, numOptions) <-
  229                                                                      eprime |> mInfo |> miTrailCompact
  230                                                                  ]
  231                                     log LogInfo $ "These options would generate at least"
  232                                                 <+> pretty estimate
  233                                                 <+> (if estimate == 1 then "model" else "models") <> "."
  234                                 else do
  235                                     case portfolioSize of
  236                                         Nothing -> return ()
  237                                         Just _ -> log LogInfo $ "Saved model in:" <+> pretty filename
  238                                     writeModel (lineWidth config) Plain (Just filename) eprime
  239                                     liftIO $ modifyIORef nbGeneratedModels (+1)
  240                             let modelHashes' = S.insert newHash modelHashes
  241                             return (modelHashes', i+1)
  242 
  243     let ?typeCheckerMode = RelaxedIntegerTags
  244 
  245     Pipes.foldM each
  246                 (return (modelHashesBefore, numberingStart config))
  247                 (\ (modelHashes, _nbModels) -> return modelHashes )
  248                 (toCompletion config model
  249                     >-> limitModelsIfNeeded
  250                     >-> limitModelsIfEstimating
  251                     >-> limitModelsPortfolioSize)

  254 toCompletion :: forall m .
  255     MonadIO m =>
  256     MonadFail m =>
  257     NameGen m =>
  258     EnumerateDomain m =>
  259     (?typeCheckerMode :: TypeCheckerMode) =>
  260     Config ->
  261     Model ->
  262     Producer LogOrModel m ()
  263 toCompletion config m = do
  264     m2 <- let ?typeCheckerMode = StronglyTyped in prologue m
  265     namegenst <- exportNameGenState
  266     let m2Info = mInfo m2
  267     let m3 = m2 { mInfo = m2Info { miStrategyQ = strategyQ config
  268                                  , miStrategyA = strategyA config
  269                                  , miNameGenState = namegenst
  270                                  } }
  271     logDebug $ modelInfo m3
  272     loopy (StartOver m3)
  273     where
  274         driver :: Driver
  275         driver = strategyToDriver config
  276 
  277         loopy :: ModelWIP -> Producer LogOrModel m ()
  278         loopy modelWIP = do
  279             logDebug $ "[loop]" <+> pretty ((modelWIPOut modelWIP) {mInfo = def})
  280             qs <- remainingWIP config modelWIP
  281             if null qs
  282                 then do
  283                     let model = modelWIPOut modelWIP
  284                     model' <- epilogue model
  285                     yield (Right model')
  286                 else do
  287                     nextModels <- driver qs
  288                     mapM_ loopy nextModels

  292     MonadFail m =>
  293     NameGen m =>
  294     EnumerateDomain m =>
  295     MonadLog m =>
  296     (?typeCheckerMode :: TypeCheckerMode) =>
  297     Model -> m JSONValue
  298 modelRepresentationsJSON model = do
  299     reprs <- modelRepresentations model
  300     return $ JSON.Array $ V.fromList
  301         [ JSON.Object $ M.fromList
  302             [ "name" ~~ r name
  303             , "representations" ~~ representationsJSON
  304             ]
  305         | (name, domains) <- reprs
  306         , let representationsJSON = JSON.Array $ V.fromList
  307                 [ JSON.Object $ M.fromList
  308                     [ "description" ~~ r d
  309                     , "answer" ~~ toJSON i
  310                     ]
  311                 | (i, d) <- zip allNats domains
  312                 ]
  313         ]
  314     where

  316         x ~~ y = (x, y)
  317         r s = JSON.String $ stringToText $ render 100000 $ pretty s

  336 -- | If a rule is applied at a position P, the MonadZipper will be retained focused at that location
  337 --   and new rules will be tried using P as the top of the zipper-tree.
  338 --   The whole model (containing P too) will be tried later for completeness.
  339 remainingWIP ::
  340     MonadFail m =>
  341     MonadLog m =>
  342     NameGen m =>
  343     EnumerateDomain m =>
  344     (?typeCheckerMode :: TypeCheckerMode) =>
  345     Config ->
  346     ModelWIP ->
  347     m [Question]
  348 remainingWIP config (StartOver model)
  349     | Just modelZipper <- mkModelZipper model = do
  350         qs <- remaining config modelZipper (mInfo model)
  351         return qs
  352     | otherwise = return []
  353 remainingWIP config wip@(TryThisFirst modelZipper info) = do
  354     qs <- remaining config modelZipper info
  355     case (null qs, Zipper.right modelZipper, Zipper.up modelZipper) of
  356         (False, _, _)  -> return qs                                         -- not null, return
  357         (_, Just r, _) -> remainingWIP config (TryThisFirst r info)         -- there is a sibling to the right
  358         (_, _, Just u) -> remainingWIP config (TryThisFirst u info)         -- there is a parent
  359         _              -> remainingWIP config (StartOver (modelWIPOut wip)) -- we are done here,
  360                                                                             -- start-over the whole model in case
  361                                                                             -- something on the left needs attention.

  364 remaining ::
  365     MonadFail m =>
  366     MonadLog m =>
  367     NameGen m =>
  368     EnumerateDomain m =>
  369     (?typeCheckerMode :: TypeCheckerMode) =>
  370     Config ->
  371     ModelZipper ->
  372     ModelInfo ->
  373     m [Question]
  374 remaining config modelZipper minfo = do
  375     -- note: the call to getQuestions can update the NameGen state
  376     importNameGenState (minfo |> miNameGenState)
  377     questions <- getQuestions config modelZipper
  378     namegenst0 <- exportNameGenState
  379     forM questions $ \ (focus, answers0) -> do
  380         answers1 <- forM answers0 $ \ (ruleName, RuleResult{..}) -> do
  381             importNameGenState namegenst0
  382             ruleResultExpr <- ruleResult
  383             -- ruleResultExpr <- fmap fixRelationProj ruleResult   -- TODO: do we need the fixRelationProj?
  384             let fullModelBeforeHook = replaceHole ruleResultExpr focus
  385             let mtyBefore = typeOf (hole focus)
  386             let mtyAfter  = typeOf ruleResultExpr
  387             case (mtyBefore, mtyAfter) of
  388                 (Right tyBefore, Right tyAfter) ->
  389                     unless (typesUnify [tyBefore, tyAfter]) $
  390                         bug $ vcat
  391                                 [ "Rule application changes type:" <+> pretty ruleName
  392                                 , "Before:" <+> pretty (hole focus)
  393                                 , "After :" <+> pretty ruleResultExpr
  394                                 , "Type before:" <+> pretty (show tyBefore)
  395                                 , "Type after :" <+> pretty (show tyAfter)
  396                                 ]
  397                 (Left msg, _) -> bug $ vcat
  398                                 [ "Type error before rule application:" <+> pretty ruleName
  399                                 , "Before:" <+> pretty (hole focus)
  400                                 , "After :" <+> pretty ruleResultExpr
  401                                 , "Error :" <+> pretty msg
  402                                 ]
  403                 (_, Left msg) -> bug $ vcat
  404                                 [ "Type error after rule application:" <+> pretty ruleName
  405                                 , "Before:" <+> pretty (hole focus)
  406                                 , "After :" <+> pretty ruleResultExpr
  407                                 , "Error :" <+> pretty msg
  408                                 ]
  409 
  410             fullModelAfterHook <- case ruleResultHook of
  411                 Nothing   -> do
  412                     namegenst <- exportNameGenState
  413                     return (TryThisFirst fullModelBeforeHook minfo { miNameGenState = namegenst })
  414                 Just hook -> do
  415                     namegenst1 <- exportNameGenState
  416                     let m1 = fromModelZipper fullModelBeforeHook minfo { miNameGenState = namegenst1 }
  417                     m2 <- hook m1
  418                     namegenst2 <- exportNameGenState
  419                     let m3 = m2 { mInfo = (mInfo m2) { miNameGenState = namegenst2 } }
  420                     return (StartOver m3)
  421 
  422             return
  423                 ( Answer
  424                     { aText = ruleName <> ":" <+> ruleResultDescr
  425                     , aRuleName = ruleName
  426                     , aBefore = hole focus
  427                     , aAnswer = ruleResultExpr
  428                     , aFullModel = fullModelAfterHook
  429                     }
  430                 , ruleResultType
  431                 )
  432         let qTypes = map snd answers1
  433         qType' <- if all (head qTypes ==) (tail qTypes)
  434                     then return (head qTypes)
  435                     else bug "Rules of different rule kinds applicable, this is a bug."
  436         return Question
  437             { qType = qType'
  438             , qHole = hole focus
  439             , qAscendants = tail (ascendants focus)
  440             , qAnswers = map fst answers1
  441             }

  444 -- | Computes all applicable questions.
  445 --   strategyQ == PickFirst is special-cased for performance.
  446 getQuestions ::
  447     MonadLog m =>
  448     MonadFail m =>
  449     NameGen m =>
  450     EnumerateDomain m =>
  451     (?typeCheckerMode :: TypeCheckerMode) =>
  452     Config ->
  453     ModelZipper ->
  454     m [(ModelZipper, [(Doc, RuleResult m)])]
  455 getQuestions config modelZipper | strategyQ config == PickFirst = maybeToList <$>
  456     let
  457         loopLevels :: Monad m => [m (Maybe a)] -> m (Maybe a)
  458         loopLevels [] = return Nothing
  459         loopLevels (a:as) = do bs <- a
  460                                case bs of
  461                                    Nothing -> loopLevels as
  462                                    Just {} -> return bs
  463 
  464         processLevel :: (MonadFail m, MonadLog m, NameGen m, EnumerateDomain m)
  465                      => [Rule]
  466                      -> m (Maybe (ModelZipper, [(Doc, RuleResult m)]))
  467         processLevel rulesAtLevel =
  468             let
  469                 go [] = return Nothing
  470                 go (x:xs) = do
  471                      ys <- applicableRules config rulesAtLevel x
  472                      if null ys
  473                          then go xs
  474                          else return (Just (x, ys))
  475             in
  476                 go (allContextsExceptReferences modelZipper)
  477     in
  478         loopLevels (map processLevel (allRules config))
  479 getQuestions config modelZipper =
  480     let
  481         loopLevels :: Monad m => [m [a]] -> m [a]
  482         loopLevels [] = return []
  483         loopLevels (a:as) = do bs <- a
  484                                if null bs
  485                                    then loopLevels as
  486                                    else return bs
  487 
  488         processLevel :: (MonadFail m, MonadLog m, NameGen m, EnumerateDomain m)
  489                      => [Rule]
  490                      -> m [(ModelZipper, [(Doc, RuleResult m)])]
  491         processLevel rulesAtLevel =
  492             fmap catMaybes $ forM (allContextsExceptReferences modelZipper) $ \ x -> do
  493                 ys <- applicableRules config rulesAtLevel x
  494                 return $ if null ys
  495                             then Nothing
  496                             else Just (x, ys)
  497     in
  498         loopLevels (map processLevel (allRules config))

  501 strategyToDriver :: Config -> Driver
  502 strategyToDriver config questions = do
  503     let optionsQ =
  504             [ (doc, q)
  505             | (n, q) <- zip allNats questions
  506             , let doc =
  507                     vcat $ ("Question" <+> pretty n <> ":" <+> pretty (qHole q))
  508                          : [ nest 4 ("Context #" <> pretty i <> ":" <+> pretty c)
  509                            | (i,c) <- zip allNats (qAscendants q)
  510                            -- if logLevel < LogDebugVerbose, only show a select few levels
  511                            , logLevel config == LogDebugVerbose || i `elem` [1,3,5,10,25]
  512                            ]
  513             ]
  514     pickedQs <- executeStrategy (bug "strategyToDriver no Question") optionsQ (strategyQ config)
  515     fmap concat $ forM pickedQs $ \ (pickedQNumber, pickedQDescr, pickedQ) -> do
  516         let optionsA =
  517                 [ (doc, a)
  518                 | (n, a) <- zip allNats (qAnswers pickedQ)
  519                 , let doc = nest 4 $ "Answer" <+> pretty n <> ":" <+>
  520                         if "choose-repr" `isPrefixOf` show (aRuleName a)
  521                             then pretty (aText a)
  522                             else vcat [ pretty (aText a)
  523                                       , sep [pretty (qHole pickedQ),  "~~>", pretty (aAnswer a)]
  524                                       ]
  525                 ]
  526         let strategyA' = case qType pickedQ of
  527                 ChooseRepr            -> representations
  528                 ChooseRepr_Find{}     -> representationsFinds
  529                 ChooseRepr_Given{}    -> representationsGivens
  530                 ChooseRepr_Auxiliary  -> representationsAuxiliaries
  531                 ChooseRepr_Quantified -> representationsQuantifieds
  532                 ChooseRepr_Cut{}      -> representationsCuts
  533                 ExpressionRefinement  -> strategyA
  534         pickedAs <- executeAnswerStrategy config pickedQ optionsA (strategyA' config)
  535         return
  536             [ theModel
  537             | (pickedANumber, pickedADescr, pickedA) <- pickedAs
  538             , let upd = addToTrail
  539                             config
  540                             (strategyQ  config) pickedQNumber                   pickedQDescr pickedQ
  541                             (strategyA' config) pickedANumber (length optionsA) pickedADescr pickedA
  542             , let theModel = updateModelWIPInfo upd (aFullModel pickedA)
  543             ]

  559 executeStrategy :: (MonadIO m, MonadLog m) => Question -> [(Doc, a)] -> Strategy -> m [(Int, Doc, a)]
  560 executeStrategy _ [] _ = bug "executeStrategy: nothing to choose from"
  561 executeStrategy _ [(doc, option)] (viewAuto -> (_, True)) = do
  562     logDebug ("Picking the only option:" <+> doc)
  563     return [(1, doc, option)]
  564 executeStrategy question options@((doc, option):_) (viewAuto -> (strategy, _)) =
  565     case strategy of
  566         Auto _      -> bug "executeStrategy: Auto"
  567         PickFirst   -> do
  568             logDebug ("Picking the first option:" <+> doc)
  569             return [(1, doc, option)]
  570         Sparse     -> do
  571             logDebug ("Picking the first option (in sparse order):" <+> doc)
  572             return [(1, doc, option)]
  573         PickAll     -> return [ (i,d,o) | (i,(d,o)) <- zip [1..] options ]
  574         Interactive -> liftIO $ do

  620                                         userErr1 $ vcat [ "Recorded response out of range."
  621                                                         , nest 4 $ "Expected a value between 1 and" <+> pretty (length options)
  622                                                         , nest 4 $ "But got: " <+> pretty recorded
  623                                                         ]

  626                                     putStr "Pick option: "
  627                                     hFlush stdout

  767 oneSuchThat :: Model -> Model
  768 oneSuchThat m = m { mStatements = onStatements (mStatements m)
  769                                     |> nubBy ((==) `on` normaliseQuantifiedVariablesS) }
  770 
  771     where
  772 
  773         onStatements :: [Statement] -> [Statement]
  774         onStatements xs =
  775             let
  776                 (suchThats0, objectives, others) = xs |> map collect |> mconcat
  777                 suchThats = suchThats0
  778                       |> map breakConjunctions                              -- break top level /\'s
  779                       |> mconcat
  780                       |> filter (/= Constant (ConstantBool True))           -- remove top level true's
  781                       |> nubBy ((==) `on` normaliseQuantifiedVariablesE)    -- uniq
  782             in
  783                 others ++ objectives ++ [SuchThat (combine suchThats)]
  784 
  785         collect :: Statement -> ( [Expression]                  -- SuchThats
  786                                 , [Statement]                   -- Objectives
  787                                 , [Statement]                   -- other statements
  788                                 )
  789         collect (SuchThat s) = (s, [], [])
  790         collect s@Objective{} = ([], [s], [])
  791         collect s = ([], [], [s])
  792 
  793         combine :: [Expression] -> [Expression]
  794         combine xs = if null xs
  795                         then [Constant (ConstantBool True)]
  796                         else xs
  797 
  798         breakConjunctions :: Expression -> [Expression]
  799         breakConjunctions p@(Op (MkOpAnd (OpAnd x))) =
  800             case listOut x of
  801                 Nothing -> [p] -- doesn't contain a list
  802                 Just xs -> concatMap breakConjunctions xs
  803         breakConjunctions x = [x]

  855 dropTagForSR ::
  856     MonadFail m =>
  857     (?typeCheckerMode :: TypeCheckerMode) =>
  858     Model -> m Model
  859 dropTagForSR m = do
  860     let
  861         replacePredSucc [essence| pred(&x) |] = do
  862             ty <- typeOf x
  863             case ty of
  864                 TypeBool{} -> return [essence| false |]
  865                                 -- since True becomes False
  866                                 --       False becomes out-of-bounds, hence False
  867                 TypeInt{}  -> do
  868                     let xTagInt = reTag TagInt x
  869                     return [essence| &xTagInt - 1 |]
  870                 _          -> bug "predSucc"
  871         replacePredSucc [essence| succ(&x) |] = do
  872             ty <- typeOf x
  873             case ty of
  874                 TypeBool{} -> return [essence| !&x |]
  875                                 -- since False becomes True
  876                                 --       True becomes out-of-bounds, hence False
  877                                 -- "succ" is exactly "negate" on bools
  878                 TypeInt{}  -> do
  879                     let xTagInt = reTag TagInt x
  880                     return [essence| &xTagInt + 1 |]
  881                 _          -> bug "predSucc"
  882         -- replacePredSucc [essence| &a .< &b |] = return [essence| &a < &b |]
  883         -- replacePredSucc [essence| &a .<= &b |] = return [essence| &a <= &b |]
  884         replacePredSucc x = return x
  885 
  886     st <- transformBiM replacePredSucc (mStatements m)
  887     return m { mStatements = transformBi (\ _ -> TagInt) st }

  890 updateDeclarations ::
  891     MonadUserError m =>
  892     MonadFail m =>
  893     NameGen m =>
  894     EnumerateDomain m =>
  895     (?typeCheckerMode :: TypeCheckerMode) =>
  896     Model -> m Model
  897 updateDeclarations model = do
  898     let
  899         representations = model |> mInfo |> miRepresentations
  900 
  901         onEachStatement (inStatement, afters) =
  902             case inStatement of
  903                 Declaration (FindOrGiven forg nm _) -> do
  904                     let
  905                         -- the refined domains for the high level declaration
  906                         domains = [ d | (n, d) <- representations, n == nm ]
  907                     nub <$> concatMapM (onEachDomain forg nm) domains
  908                 Declaration (GivenDomainDefnEnum name) -> return
  909                     [ Declaration (FindOrGiven Given (name `mappend` "_EnumSize") (DomainInt TagInt [])) ]
  910                 Declaration (Letting nm x)             -> do
  911                     let usedAfter = nbUses nm afters > 0
  912                     let isRefined = (0 :: Int) == sum
  913                                             [ case y of
  914                                                 Constant (ConstantAbstract AbsLitMatrix{}) -> 0
  915                                                 Constant ConstantAbstract{} -> 1
  916                                                 AbstractLiteral AbsLitMatrix{} -> 0
  917                                                 AbstractLiteral{} -> 1
  918                                                 _ -> 0
  919                                             | y <- universe x ]
  920                     return [inStatement | and [usedAfter, isRefined]]
  921                 Declaration LettingDomainDefnEnum{}    -> return []
  922                 Declaration LettingDomainDefnUnnamed{} -> return []
  923                 SearchOrder orders -> do
  924                     orders' <- forM orders $ \case
  925                         BranchingOn nm -> do
  926                             let domains = [ d | (n, d) <- representations, n == nm ]
  927                             -- last one is the representation of what's in true(?)
  928                             -- put that first!
  929                             let reorder xs =
  930                                     case reverse xs of
  931                                         [] -> []
  932                                         (y:ys) -> y : reverse ys
  933                             outNames <- concatMapM (onEachDomainSearch nm) (reorder domains)
  934                             return $ map BranchingOn $ nub outNames
  935                         Cut{} -> bug "updateDeclarations, Cut shouldn't be here"
  936                     return [ SearchOrder (concat orders') ]
  937                 _ -> return [inStatement]
  938 
  939         onEachDomain forg nm domain =
  940             runExceptT (downD (nm, domain)) >>= \case
  941                 Left err -> bug err
  942                 Right outs -> forM outs $ \ (n, d) -> do
  943                     d' <- transformBiM (trySimplify []) $ forgetRepr d
  944                     return $ Declaration (FindOrGiven forg n d')
  945 
  946         onEachDomainSearch nm domain =
  947             runExceptT (downD (nm, domain)) >>= \case
  948                 Left err -> bug err
  949                 Right outs -> return [ n
  950                                      | (n, _) <- outs
  951                                      ]
  952 
  953     statements <- concatMapM onEachStatement (withAfter (mStatements model))
  954     return model { mStatements = statements }

 1041 topLevelBubbles ::
 1042     MonadFail m =>
 1043     MonadUserError m =>
 1044     NameGen m =>
 1045     (?typeCheckerMode :: TypeCheckerMode) =>
 1046     Model -> m Model
 1047 topLevelBubbles m = do
 1048     let
 1049         onStmt (SuchThat xs) = onExprs xs
 1050         onStmt (Where    xs) = concatMapM onWheres xs
 1051         onStmt (Objective obj (WithLocals h locals)) =
 1052             case locals of
 1053                 AuxiliaryVars          locs -> (           locs  ++ [Objective obj h] ) |> onStmts
 1054                 DefinednessConstraints locs -> ( [SuchThat locs] ++ [Objective obj h] ) |> onStmts
 1055         onStmt (Declaration decl) =
 1056             let
 1057                 f (WithLocals h locs) = tell [locs] >> return h
 1058                 f x = return x
 1059 
 1060                 (decl', locals) = runWriter (transformBiM f decl)
 1061 
 1062                 conv :: InBubble -> [Statement]
 1063                 conv (AuxiliaryVars locs) = locs
 1064                 conv (DefinednessConstraints locs) = [SuchThat locs]
 1065 
 1066                 newStmts :: [Statement]
 1067                 newStmts = concatMap conv locals
 1068             in
 1069                 if null newStmts
 1070                     then return [Declaration decl]
 1071                     else onStmts (newStmts ++ [Declaration decl'])
 1072         onStmt s = return [s]
 1073 
 1074         -- a where that has a bubble at the top-most level will be replaced
 1075         -- with a Comprehension. this is to avoid creating a where with decision variables inside.
 1076         onWheres (WithLocals h (DefinednessConstraints locals)) =
 1077             return $ map (Where . return) (locals ++ [h])
 1078         onWheres (WithLocals h (AuxiliaryVars locals)) = do
 1079             let (localfinds, gens) = mconcat
 1080                     [ case local of
 1081                         Declaration (FindOrGiven LocalFind nm dom) ->
 1082                             ([nm], [Generator (GenDomainNoRepr (Single nm) dom)])
 1083                         SuchThat xs ->
 1084                             ([], map Condition xs)
 1085                         _ -> bug ("topLevelBubbles.onWheres:" <+> pretty local)
 1086                     | local <- locals
 1087                     ]
 1088             let forgetReprsOfLocalFinds (Reference nm _) | nm `elem` localfinds = Reference nm Nothing
 1089                 forgetReprsOfLocalFinds x = descend forgetReprsOfLocalFinds x
 1090             let out = Comprehension h gens
 1091             out' <- resolveNamesX (forgetReprsOfLocalFinds out)
 1092             return [Where [out']]
 1093         onWheres x = return [Where [x]]
 1094 
 1095         onExpr (WithLocals h (AuxiliaryVars          locals)) = (          locals  ++ [SuchThat [h]]) |> onStmts
 1096         onExpr (WithLocals h (DefinednessConstraints locals)) = ([SuchThat locals] ++ [SuchThat [h]]) |> onStmts
 1097         onExpr x = return [SuchThat [x]]
 1098 
 1099         onStmts = concatMapM onStmt
 1100         onExprs = concatMapM onExpr
 1101 
 1102     statements' <- onStmts (mStatements m)
 1103     return m { mStatements = statements' }

 1106 sliceThemMatrices ::
 1107     Monad m =>
 1108     (?typeCheckerMode :: TypeCheckerMode) =>
 1109     Model -> m Model
 1110 sliceThemMatrices model = do
 1111     let
 1112         -- nothing stays with a matrix type
 1113         -- we are doing this top down
 1114         -- when we reach a matrix-typed expression, we know it needs to be sliced
 1115         -- we descend otherwise
 1116         -- we also descend into components of the matrix-typed expression during slicing
 1117         onExpr :: Monad m => Expression -> m Expression
 1118         onExpr p = do
 1119             let computeExistingSlices t =
 1120                     case match opSlicing t of
 1121                         Nothing -> return 0
 1122                         Just (t', _, _) -> (+1) <$> computeExistingSlices t'
 1123             let isIndexedMatrix = do
 1124                     (m, is) <- match opMatrixIndexing p
 1125                     tyM     <- typeOf m
 1126                     nSlices <- computeExistingSlices m
 1127                     return (m, nSlices, is, tyM)
 1128             case isIndexedMatrix of
 1129                 Nothing -> descendM onExpr p
 1130                 Just (m, existingSlices, is, tyM) -> do
 1131                     let nestingLevel (TypeMatrix _ a) = 1 + nestingLevel a
 1132                         nestingLevel (TypeList     a) = 1 + nestingLevel a
 1133                         nestingLevel _ = 0 :: Int
 1134                     -- "is" is the number of existing indices
 1135                     -- "nestingLevel" is the nesting level of the original matrix
 1136                     -- "existingSlices" is the number of existing slices
 1137                     let howMany = nestingLevel tyM - existingSlices - length is
 1138                     let unroll a 0 = a
 1139                         unroll a i = make opSlicing (unroll a (i-1)) Nothing Nothing
