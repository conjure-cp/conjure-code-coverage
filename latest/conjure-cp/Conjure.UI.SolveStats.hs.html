   29 mkSolveStats :: String -> Text -> SolveStats
   30 mkSolveStats raw stdout =
   31   let info = M.fromList [(k, v) | [k, v] <- map (splitOn ":") (lines raw)]
   32       status
   33         | info M.! "SavileRowTimeOut" == "1" = TimeOut
   34         | info M.! "SavileRowClauseOut" == "1" = TimeOut
   35         | info M.! "SolverTimeOut" == "1" = TimeOut
   36         | T.isInfixOf "Savile Row timed out." stdout = TimeOut
   37         | T.isInfixOf "java.lang.OutOfMemoryError" stdout = MemOut
   38         | otherwise = OK
   39       totalTime
   40         | Just srTotalTime <- readMay $ info M.! "SavileRowTotalTime",
   41           Just solverTotalTime <- readMay $ info M.! "SolverTotalTime" =
   42             Just (srTotalTime + solverTotalTime)
   43         | otherwise = Nothing
   44       savilerowInfo = info
   45    in SolveStats {..}
