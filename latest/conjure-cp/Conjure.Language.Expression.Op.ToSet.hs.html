   21 instance FromJSON  x => FromJSON  (OpToSet x) where parseJSON = genericParseJSON jsonOptions

   23 instance (TypeOf x, Pretty x) => TypeOf (OpToSet x) where
   24     typeOf p@(OpToSet _ x) = do
   25         tx <- typeOf x
   26         case tx of
   27             TypeRelation is  -> return (TypeSet (TypeTuple is))
   28             TypeMSet i       -> return (TypeSet i)
   29             TypeFunction i j -> return (TypeSet (TypeTuple [i,j]))
   30             TypeMatrix _ i   -> return (TypeSet i)
   31             TypeList i       -> return (TypeSet i)
   32             _ -> raiseTypeError $ vcat [ pretty p
   33                                        , "The argument has type:" <+> pretty tx
   34                                        ]
