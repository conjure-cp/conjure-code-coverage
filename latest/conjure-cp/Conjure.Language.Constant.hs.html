   44 data Constant
   45     = ConstantBool Bool
   46     | ConstantInt IntTag Integer
   47     | ConstantEnum Name   {- name for the enum domain -}
   48                    [Name] {- values in the enum domain -}
   49                    Name   {- the literal -}
   50     | ConstantField Name Type                               -- the name of a field of Record or Variant and its type
   51     | ConstantAbstract (AbstractLiteral Constant)
   52     | DomainInConstant (Domain () Constant)
   53     | TypedConstant Constant Type
   54     | ConstantUndefined Text Type                           -- never use this for a bool
   55                                                             -- use false instead for them
   56     deriving (Show, Data, Typeable, Generic)
   57 
   58 instance Eq Constant where
   59     a == b = compare a b == EQ
   60 
   61 -- implementing the Eq&Ord instances by hand, because we want to special case the TypedConstant constructor
   62 instance Ord Constant where
   63 
   64     -- do not use type info when comparing
   65     compare (TypedConstant a _) (TypedConstant b _) = compare a b
   66     compare (TypedConstant a _) b = compare a b
   67     compare a (TypedConstant b _) = compare a b
   68 
   69     -- the "usual" comparisons
   70     compare (ConstantBool a) (ConstantBool b) = compare a b
   71     compare (ConstantInt _ a) (ConstantInt _ b) = compare a b
   72     compare (ConstantEnum _ aVals aVal) (ConstantEnum _ bVals bVal) =
   73         compare (elemIndex aVal aVals, aVal) (elemIndex bVal bVals, bVal)
   74     compare (ConstantField a1 a2) (ConstantField b1 b2) = compare (a1,a2) (b1,b2)
   75     compare (ConstantAbstract a) (ConstantAbstract b) = compare a b
   76     compare (DomainInConstant a) (DomainInConstant b) = compare a b
   77     compare (ConstantUndefined a1 a2) (ConstantUndefined b1 b2) = compare (a1,a2) (b1,b2)
   78 
   79     -- if the constructors do not match
   80     compare a b = compare (constrIndex (toConstr a)) (constrIndex (toConstr b))

   87 instance SimpleJSON Constant where
   88     toSimpleJSON c =
   89         case c of
   90             ConstantBool b -> return (toJSON b)
   91             ConstantInt _ i -> return (toJSON i)
   92             ConstantEnum _ _ nm -> return (toJSON (renderNormal nm))
   93             ConstantAbstract lit -> toSimpleJSON lit
   94             TypedConstant c' _ -> toSimpleJSON c'
   95             _ -> noToSimpleJSON c
   96     fromSimpleJSON _ = noFromSimpleJSON

  104 instance TypeOf Constant where
  105     typeOf ConstantBool{}             = return TypeBool
  106     typeOf (ConstantInt t _)          = return (TypeInt t)
  107     typeOf (ConstantEnum defn _ _ )   = return (TypeEnum defn)

  110     typeOf (DomainInConstant dom)     = typeOfDomain dom
  111     typeOf (TypedConstant _ ty)       = return ty
  112     typeOf (ConstantUndefined _ ty)   = return ty

  114 instance DomainSizeOf Constant Integer where
  115     domainSizeOf DomainBool{} = return 2
  116     domainSizeOf (DomainIntE x) = bug ("not implemented, domainSizeOf DomainIntE" <+> pretty (show x))
  117     domainSizeOf (DomainInt _ rs) = domainSizeOfRanges rs
  118     domainSizeOf DomainEnum{} = fail "domainSizeOf: Unknown for given enum."
  119     domainSizeOf (DomainTuple ds) = product <$> mapM domainSizeOf ds
  120     domainSizeOf (DomainMatrix index inner) = intPow <$> domainSizeOf inner <*> domainSizeOf index
  121     domainSizeOf d@(DomainSet _ (SetAttr attrs) inner) =
  122         case attrs of
  123             SizeAttr_None -> do
  124                 innerSize <- domainSizeOf inner
  125                 return (2 `intPow` innerSize)
  126             SizeAttr_Size (ConstantInt _ size) -> do
  127                 innerSize <- domainSizeOf inner
  128                 return (nchoosek (product . enumFromTo 1) innerSize size)
  129             SizeAttr_MinSize{} -> do
  130                 -- TODO: we can do better here
  131                 innerSize <- domainSizeOf inner
  132                 return (2 `intPow` innerSize)
  133             SizeAttr_MaxSize (ConstantInt _ maxSize) -> do
  134                 innerSize <- domainSizeOf inner
  135                 return $ sum [ nchoosek (product . enumFromTo 1) innerSize k | k <- [0 .. maxSize] ]
  136             SizeAttr_MinMaxSize (ConstantInt _ minSize) (ConstantInt _ maxSize) -> do
  137                 innerSize <- domainSizeOf inner
  138                 return $ sum [ nchoosek (product . enumFromTo 1) innerSize k | k <- [minSize .. maxSize] ]
  139             _ -> fail ("domainSizeOf{Constant}" <+> pretty d)
  140     domainSizeOf DomainMSet      {} = bug "not implemented: domainSizeOf DomainMSet"
  141     domainSizeOf DomainFunction  {} = bug "not implemented: domainSizeOf DomainFunction"
  142     domainSizeOf DomainRelation  {} = bug "not implemented: domainSizeOf DomainRelation"
  143     domainSizeOf DomainPartition {} = bug "not implemented: domainSizeOf DomainPartition"
  144     domainSizeOf _                  = bug "not implemented: domainSizeOf"

  165 instance Pretty Constant where
  166 
  167     pretty (TypedConstant (ConstantAbstract (AbsLitMatrix _ [])) ty) | TypeAny `elem` universe ty = "[]"
  168 
  169     -- hack, oh sweet hack!
  170     -- print a domain instead of a type when printing an empty matrix literal.
  171     -- this means we print "int()" instead of "int" inside the index of a matrix type
  172     -- SR expects it this way...
  173     pretty (TypedConstant (ConstantAbstract (AbsLitMatrix _ [])) ty) =
  174         let
  175             pretty' (TypeMatrix index innerNested)
  176                 = "matrix indexed by" <+> prettyList prBrackets "," (map pretty' indices)
  177                                       <+> "of" <+> pretty inner
  178                 where
  179                     (indices,inner) = first (index:) $ collect innerNested
  180                     collect (TypeMatrix i j) = first (i:) $ collect j
  181                     collect x = ([],x)
  182             pretty' TypeInt{} = "int()"
  183             pretty' t = pretty t
  184         in
  185             prParens $ "[] : `" <> pretty' ty <> "`"
  186 
  187     pretty (ConstantBool False)          = "false"
  188     pretty (ConstantBool True )          = "true"
  189     pretty (ConstantInt _ x   )          = pretty x
  190     pretty (ConstantEnum _ _ x)          = pretty x
  191     pretty (ConstantField n _)           = pretty n
  192     pretty (ConstantAbstract x)          = pretty x
  193     pretty (DomainInConstant d)          = "`" <> pretty d <> "`"
  194     pretty (TypedConstant x ty)          = prParens $ pretty x <+> ":" <+> "`" <> pretty ty <> "`"
  195     pretty (ConstantUndefined reason ty) = "undefined" <> prParens (pretty reason <+> ":" <+> "`" <> pretty ty <> "`")

  197 instance ExpressionLike Constant where
  198     fromInt = ConstantInt TagInt
  199     fromIntWithTag i t = ConstantInt t i
  200     intOut _ (ConstantInt _ x) = return x

  205     fromBool = ConstantBool

  232 normaliseConstant :: Constant -> Constant
  233 normaliseConstant x@ConstantBool{} = x
  234 normaliseConstant x@ConstantInt{}  = x
  235 normaliseConstant x@ConstantEnum{} = x
  236 normaliseConstant x@ConstantField{} = x
  237 normaliseConstant (ConstantAbstract x) = ConstantAbstract (normaliseAbsLit normaliseConstant x)
  238 normaliseConstant (DomainInConstant d) = DomainInConstant (normaliseDomain normaliseConstant d)
  239 normaliseConstant (TypedConstant c ty) = TypedConstant (normaliseConstant c) ty
  240 normaliseConstant x@ConstantUndefined{} = x

  242 instance Num Constant where
  243     ConstantInt _ x + ConstantInt _ y = ConstantInt TagInt (x+y)
  244     x + y = bug $ vcat [ "Num Constant (+)", "x:" <+> pretty x, "y:" <+> pretty y ]
  245     ConstantInt _ x - ConstantInt _ y = ConstantInt TagInt (x-y)
  246     x - y = bug $ vcat [ "Num Constant (-)", "x:" <+> pretty x, "y:" <+> pretty y ]
  247     ConstantInt _ x * ConstantInt _ y = ConstantInt TagInt (x*y)
  248     x * y = bug $ vcat [ "Num Constant (*)", "x:" <+> pretty x, "y:" <+> pretty y ]
  249     abs (ConstantInt t x) = ConstantInt t (abs x)
  250     abs x = bug $ vcat [ "Num Constant abs", "x:" <+> pretty x ]
  251     signum (ConstantInt t x) = ConstantInt t (signum x)
  252     signum x = bug $ vcat [ "Num Constant signum", "x:" <+> pretty x ]
  253     fromInteger = ConstantInt TagInt . fromInteger

  256 valuesInIntDomain :: MonadFail m => [Range Constant] -> m [Integer]
  257 valuesInIntDomain ranges =
  258     if isFinite
  259         then return allValues
  260         else fail $ "Expected finite integer ranges, but got:" <++> prettyList id "," ranges
  261 
  262     where
  263 
  264         allRanges :: [Maybe [Integer]]
  265         allRanges =
  266             [ vals
  267             | r <- ranges
  268             , let vals = case r of
  269                     RangeSingle (ConstantInt _ x) -> return [x]
  270                     RangeBounded (ConstantInt _ l) (ConstantInt _ u) -> return [l..u]
  271                     _ -> Nothing
  272             ]
  273 
  274         isFinite :: Bool
  275         isFinite = Nothing `notElem` allRanges
  276 
  277         allValues :: [Integer]
  278         allValues = sortNub $ concat $ catMaybes allRanges

  321 viewConstantFunction  :: MonadFail m => Constant -> m [(Constant, Constant)]
  322 viewConstantFunction  (ConstantAbstract (AbsLitFunction xs)) = return xs
  323 viewConstantFunction  (TypedConstant c _) = viewConstantFunction c
  324 viewConstantFunction  constant = do
  325     let
  326         suggestion = case constant of
  327             ConstantAbstract (AbsLitMatrix (expandDomainReference -> DomainInt _ rs) vals) -> do
  328                 froms <- valuesInIntDomain rs
  329                 return $ Just $ pretty $ AbsLitFunction (zip (map (ConstantInt TagInt) froms) vals)
  330             _ -> return Nothing
  331     suggestion >>= \case
  332         Nothing  -> fail ("Expecting a function, but got:" <++> pretty constant)
  333         Just sug -> fail (vcat [ "Expecting a function, but got:" <++> pretty constant
  334                                , "Maybe you meant:" <++> sug
  335                                ])
