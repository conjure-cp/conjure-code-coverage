   46 rule_Comprehension :: Rule
   47 rule_Comprehension = "function-comprehension{FunctionNDPartialDummy}" `namedRule` theRule where
   48     theRule (Comprehension body gensOrConds) = do
   49         (gocBefore, (pat, func), gocAfter) <- matchFirst gensOrConds $ \ goc -> case goc of
   50             Generator (GenInExpr pat@Single{} expr) -> return (pat, matchDefs [opToSet,opToMSet,opToRelation] expr)
   51             _ -> na "rule_Comprehension"
   52         Function_NDPartialDummy <- representationOf func
   53         DomainFunction _ _ innerDomainFr innerDomainTo <- domainOf func
   54         [values] <- downX1 func
   55         let dummy = [essence| min(`&innerDomainTo`) - 1 |]
   56         let upd val old = lambdaToFunction pat old val
   57         return
   58             ( "Mapping over a function, FunctionNDPartialDummy representation"
   59             , do
   60                 (jPat, j) <- quantifiedVar
   61                 let kRange = case innerDomainFr of
   62                         DomainTuple ts  -> Just $ map fromInt [1 .. genericLength ts]
   63                         DomainRecord rs -> Just $ map (fromName . fst) rs
   64                         _ -> Nothing
   65                     toIndex       = case kRange of
   66                                         Just ks -> [ [essence| &j[&k] |] | k <- ks ]
   67                                         Nothing -> [j]
   68                     valuesIndexed = make opMatrixIndexing values toIndex
   69                     val           = [essence| (&j, &valuesIndexed) |]
   70                 return $ Comprehension (upd val body)
   71                     $  gocBefore
   72                     ++ [ Generator (GenDomainNoRepr jPat (forgetRepr innerDomainFr))
   73                        , Condition [essence| &val[2] != &dummy |]
   74                        ]
   75                     ++ transformBi (upd val) gocAfter
   76             )
   77     theRule _ = na "rule_Comprehension"
