   23 data AbstractLiteral x
   24     = AbsLitTuple [x]
   25     | AbsLitRecord [(Name, x)]
   26     | AbsLitVariant (Maybe [(Name, Domain () x)]) Name x            -- Nothing before name resolution
   27     | AbsLitMatrix (Domain () x) [x]                                -- the domain is the index domain
   28     | AbsLitSet [x]
   29     | AbsLitMSet [x]
   30     | AbsLitFunction [(x, x)]
   31     | AbsLitSequence [x]
   32     | AbsLitRelation [[x]]
   33     | AbsLitPartition [[x]]
   34     deriving (Eq, Ord, Show, Data, Functor, Traversable, Foldable, Typeable, Generic)
   35

   41 instance (SimpleJSON x, Pretty x, ExpressionLike x) => SimpleJSON (AbstractLiteral x) where
   42     toSimpleJSON lit =
   43         case lit of
   44             AbsLitTuple xs -> toSimpleJSON xs
   45             AbsLitRecord xs -> do
   46                 xs' <- forM xs $ \ (nm, x) -> do
   47                     x' <- toSimpleJSON x
   48                     return (stringToText (renderNormal nm), x')
   49                 return $ JSON.Object $ M.fromList xs'
   50             AbsLitMatrix index xs ->
   51                 case index of
   52                     DomainInt _ ranges -> do
   53                         indices <- failToUserError $ rangesInts ranges
   54                         toSimpleJSON (AsDictionary (zip indices xs))
   55                     _ -> toSimpleJSON xs
   56             AbsLitSet xs -> toSimpleJSON xs
   57             AbsLitMSet xs -> toSimpleJSON xs
   58             AbsLitFunction xs -> toSimpleJSON (AsDictionary xs)
   59             AbsLitSequence xs -> toSimpleJSON xs
   60             AbsLitRelation xs -> toSimpleJSON xs
   61             AbsLitPartition xs -> toSimpleJSON xs
   62             _ -> noToSimpleJSON lit
   63     fromSimpleJSON x = noFromSimpleJSON "AbstractLiteral" x

   65 instance (ToFromMiniZinc x, Pretty x, ExpressionLike x) => ToFromMiniZinc (AbstractLiteral x) where
   66     toMiniZinc lit =
   67         case lit of
   68             AbsLitTuple xs -> MZNArray Nothing <$> mapM toMiniZinc xs
   69             AbsLitMatrix (DomainInt _ [RangeSingle r]) xs -> MZNArray (Just $ show $ pretty r <> ".." <> pretty r) <$> mapM toMiniZinc xs
   70             AbsLitMatrix (DomainInt _ [r]) xs -> MZNArray (Just $ show $ pretty r) <$> mapM toMiniZinc xs
   71             AbsLitMatrix _index xs -> MZNArray Nothing <$> mapM toMiniZinc xs
   72             AbsLitSet xs ->
   73                 case xs of
   74                     (x:_) | Just _ <- intOut "toMiniZinc" x -> MZNSet <$> mapM toMiniZinc xs
   75                     _ -> MZNArray Nothing <$> mapM toMiniZinc xs
   76             AbsLitMSet xs -> MZNArray Nothing <$> mapM toMiniZinc xs
   77             AbsLitFunction xs -> MZNArray Nothing <$> mapM (toMiniZinc . snd) xs
   78             AbsLitSequence xs -> MZNArray Nothing <$> mapM toMiniZinc xs
   79             AbsLitRelation xss ->
   80                 MZNArray Nothing <$> forM xss (\ xs ->
   81                     MZNArray Nothing <$> mapM toMiniZinc xs)
   82             AbsLitPartition xss ->
   83                 MZNArray Nothing <$> forM xss (\ xs ->
   84                     MZNArray Nothing <$> mapM toMiniZinc xs)
   85             _ -> noToMiniZinc lit

   87 instance Pretty a => Pretty (AbstractLiteral a) where
   88     pretty (AbsLitTuple xs) = (if length xs < 2 then "tuple" else prEmpty) <+> prettyList prParens "," xs
   89     pretty (AbsLitRecord xs) = "record" <+> prettyList prBraces "," [ pretty n <+> "=" <+> pretty x
   90                                                                     | (n,x) <- xs ]
   91     pretty (AbsLitVariant _ n x) = "variant" <+> prBraces (pretty n <+> "=" <+> pretty x)
   92     pretty (AbsLitMatrix _     []) = "[]"
   93     pretty (AbsLitMatrix index xs) = let f i = prBrackets (i <> ";" <++> pretty index) in prettyList f "," xs
   94     pretty (AbsLitSet       xs ) =                prettyList prBraces "," xs
   95     pretty (AbsLitMSet      xs ) = "mset"      <> prettyList prParens "," xs
   96     pretty (AbsLitFunction  xs ) = "function"  <> prettyListDoc prParens "," [ pretty a <++> "-->" <+> pretty b | (a,b) <- xs ]
   97     pretty (AbsLitSequence  xs ) = "sequence"  <> prettyList prParens "," xs
   98     pretty (AbsLitRelation  xss) = "relation"  <> prettyListDoc prParens "," [ pretty (AbsLitTuple xs)         | xs <- xss   ]
   99     pretty (AbsLitPartition xss) = "partition" <> prettyListDoc prParens "," [ prettyList prBraces "," xs      | xs <- xss   ]

  101 instance (VarSymBreakingDescription x, ExpressionLike x) => VarSymBreakingDescription (AbstractLiteral x) where
  102     varSymBreakingDescription (AbsLitTuple xs) = JSON.Object $ M.fromList
  103         [ ("type", JSON.String "AbsLitTuple")
  104         , ("children", JSON.Array $ V.fromList $ map varSymBreakingDescription xs)
  105         ]
  106     varSymBreakingDescription AbsLitRecord{} = JSON.Object $ M.fromList
  107         [ ("type", JSON.String "AbsLitRecord")
  108         ]
  109     varSymBreakingDescription AbsLitVariant{} = JSON.Object $ M.fromList
  110         [ ("type", JSON.String "AbsLitVariant")
  111         ]
  112     varSymBreakingDescription (AbsLitMatrix _ xs) = JSON.Object $ M.fromList
  113         [ ("type", JSON.String "AbsLitMatrix")
  114         , ("children", JSON.Array $ V.fromList $ map varSymBreakingDescription xs)
  115         ]
  116     varSymBreakingDescription (AbsLitSet xs) = JSON.Object $ M.fromList
  117         [ ("type", JSON.String "AbsLitSet")
  118         , ("children", JSON.Array $ V.fromList $ map varSymBreakingDescription xs)
  119         , ("symmetricChildren", JSON.Bool True)
  120         ]
  121     varSymBreakingDescription (AbsLitMSet xs) = JSON.Object $ M.fromList
  122         [ ("type", JSON.String "AbsLitMSet")
  123         , ("children", JSON.Array $ V.fromList $ map varSymBreakingDescription xs)
  124         , ("symmetricChildren", JSON.Bool True)
  125         ]
  126     varSymBreakingDescription (AbsLitFunction xs) = JSON.Object $ M.fromList
  127         [ ("type", JSON.String "AbsLitFunction")
  128         , ("children", JSON.Array $ V.fromList
  129             [ varSymBreakingDescription (AbsLitTuple [x,y]) | (x,y) <- xs ])
  130         , ("symmetricChildren", JSON.Bool True)
  131         ]
  132     varSymBreakingDescription (AbsLitSequence xs) = JSON.Object $ M.fromList
  133         [ ("type", JSON.String "AbsLitSequence")
  134         , ("children", JSON.Array $ V.fromList
  135             [ varSymBreakingDescription (AbsLitTuple [fromInt i, x]) | (i,x) <- zip allNats xs ])
  136         , ("symmetricChildren", JSON.Bool True)
  137         ]
  138     varSymBreakingDescription (AbsLitRelation xs) = JSON.Object $ M.fromList
  139         [ ("type", JSON.String "AbsLitRelation")
  140         , ("children", JSON.Array $ V.fromList $ map (varSymBreakingDescription . AbsLitTuple) xs)
  141         , ("symmetricChildren", JSON.Bool True)
  142         ]
  143     varSymBreakingDescription (AbsLitPartition xs) = JSON.Object $ M.fromList
  144         [ ("type", JSON.String "AbsLitPartition")
  145         , ("children", JSON.Array $ V.fromList $ map (varSymBreakingDescription . AbsLitSet) xs)
  146         , ("symmetricChildren", JSON.Bool True)
  147         ]

  149 instance (TypeOf a, Pretty a) => TypeOf (AbstractLiteral a) where
  150 
  151     typeOf   (AbsLitTuple        []) = return (TypeTuple [TypeAny])
  152     typeOf   (AbsLitTuple        xs) = TypeTuple    <$> mapM typeOf xs
  153 
  154     typeOf   (AbsLitRecord       xs) = TypeRecord   <$> sequence [ do t <- typeOf x ; return (n,t)
  155                                                                  | (n,x) <- xs ]
  156 
  157     typeOf   (AbsLitVariant Nothing  _ _) = fail "Cannot calculate the type of variant literal."
  158     typeOf   (AbsLitVariant (Just t) _ _) = fmap TypeVariant $ forM t $ \ (n,d) -> do
  159         dt <- typeOfDomain d
  160         return (n, dt)
  161 
  162     typeOf   (AbsLitMatrix _   []  ) = return (TypeMatrix TypeAny TypeAny)
  163     typeOf p@(AbsLitMatrix ind inn ) = TypeMatrix   <$> typeOfDomain ind <*> (homoType (pretty p) =<< mapM typeOf inn)
  164 
  165     typeOf   (AbsLitSet         [] ) = return (TypeSet TypeAny)
  166     typeOf p@(AbsLitSet         xs ) = TypeSet      <$> (homoType (pretty p) =<< mapM typeOf xs)
  167 
  168     typeOf   (AbsLitMSet        [] ) = return (TypeMSet TypeAny)
  169     typeOf p@(AbsLitMSet        xs ) = TypeMSet     <$> (homoType (pretty p) =<< mapM typeOf xs)
  170 
  171     typeOf   (AbsLitFunction    [] ) = return (TypeFunction TypeAny TypeAny)
  172     typeOf p@(AbsLitFunction    xs ) = TypeFunction <$> (homoType (pretty p) =<< mapM (typeOf . fst) xs)
  173                                                     <*> (homoType (pretty p) =<< mapM (typeOf . snd) xs)
  174 
  175     typeOf   (AbsLitSequence    [] ) = return (TypeSequence TypeAny)
  176     typeOf p@(AbsLitSequence    xs ) = TypeSequence <$> (homoType (pretty p) =<< mapM typeOf xs)
  177 
  178     typeOf   (AbsLitRelation    [] ) = return (TypeRelation [TypeAny])
  179     typeOf p@(AbsLitRelation    xss) = do
  180         ty <- homoType (pretty p) =<< mapM (typeOf . AbsLitTuple) xss
  181         case ty of
  182             TypeTuple ts -> return (TypeRelation ts)
  183             _ -> bug "expecting TypeTuple in typeOf"
  184 
  185     typeOf   (AbsLitPartition   [] ) = return (TypePartition TypeAny)
  186     typeOf p@(AbsLitPartition   xss) = TypePartition <$> (homoType (pretty p) =<< mapM typeOf (concat xss))

  189 normaliseAbsLit :: (Ord c, ExpressionLike c) => (c -> c) -> AbstractLiteral c -> AbstractLiteral c
  190 normaliseAbsLit norm (AbsLitTuple     xs ) = AbsLitTuple                           $ map norm xs
  191 normaliseAbsLit norm (AbsLitRecord    xs ) = AbsLitRecord                          $ map (second norm) xs
  192 normaliseAbsLit norm (AbsLitVariant t n x) = AbsLitVariant t n (norm x)
  193 normaliseAbsLit norm (AbsLitMatrix d  xs ) = AbsLitMatrix (normaliseDomain norm d) $ map norm xs
  194 normaliseAbsLit norm (AbsLitSet       xs ) = AbsLitSet                   $ sortNub $ map norm xs
  195 normaliseAbsLit norm (AbsLitMSet      xs ) = AbsLitMSet                  $ sort    $ map norm xs
  196 normaliseAbsLit norm (AbsLitFunction  xs ) = AbsLitFunction              $ sortNub [ (norm x, norm y) | (x, y) <- xs ]
  197 normaliseAbsLit norm (AbsLitSequence  xs ) = AbsLitSequence              $           map norm xs
  198 normaliseAbsLit norm (AbsLitRelation  xss) = AbsLitRelation              $ sortNub $ map (map norm) xss
  199 normaliseAbsLit norm (AbsLitPartition xss) = AbsLitPartition             $ sortNub $ map (sortNub . map norm) xss

  201 emptyCollectionAbsLit :: AbstractLiteral c -> Bool
  202 emptyCollectionAbsLit AbsLitTuple{} = False
  203 emptyCollectionAbsLit AbsLitRecord{} = False
  204 emptyCollectionAbsLit AbsLitVariant{} = False
  205 emptyCollectionAbsLit (AbsLitMatrix _ xs) = null xs
  206 emptyCollectionAbsLit (AbsLitSet xs) = null xs
  207 emptyCollectionAbsLit (AbsLitMSet xs) = null xs
  208 emptyCollectionAbsLit (AbsLitFunction xs) = null xs
  209 emptyCollectionAbsLit (AbsLitSequence xs) = null xs
  210 emptyCollectionAbsLit (AbsLitRelation xs) = null xs
  211 emptyCollectionAbsLit (AbsLitPartition xs) = null xs
