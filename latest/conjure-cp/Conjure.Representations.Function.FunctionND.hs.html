   15 functionND :: forall m . (MonadFail m, NameGen m, ?typeCheckerMode :: TypeCheckerMode) => Representation m
   16 functionND = Representation chck downD structuralCons downC up symmetryOrdering
   17 
   18     where
   19 
   20         chck :: TypeOf_ReprCheck m
   21         chck f (DomainFunction _
   22                     attrs@(FunctionAttr _ PartialityAttr_Total _)
   23                     innerDomainFr@(viewAsDomainTuple -> Just innerDomainFrs)
   24                     innerDomainTo) | all domainCanIndexMatrix innerDomainFrs = do
   25             innerDomainFr' <- f innerDomainFr
   26             innerDomainTo' <- f innerDomainTo
   27             return [ DomainFunction Function_ND attrs fr to
   28                    | fr <- innerDomainFr'
   29                    , to <- innerDomainTo'
   30                    ]
   31         chck _ _ = return []
   32 
   33         nameValues :: Domain HasRepresentation x -> Name -> Name
   34         nameValues = mkOutName Nothing
   35 
   36         downD :: TypeOf_DownD m
   37         downD (name, domain@(DomainFunction Function_ND
   38                     (FunctionAttr _ PartialityAttr_Total _)
   39                     (viewAsDomainTuple -> Just innerDomainFrs)
   40                     innerDomainTo)) | all domainCanIndexMatrix innerDomainFrs = do
   41             let unroll is j = foldr DomainMatrix j is
   42             return $ Just
   43                 [ ( nameValues domain name
   44                   , unroll (map forgetRepr innerDomainFrs) innerDomainTo
   45                   )
   46                 ]
   47         downD _ = na "{downD} FunctionND"
   48 
   49         structuralCons :: TypeOf_Structural m
   50         structuralCons f downX1
   51             (DomainFunction Function_ND
   52                 (FunctionAttr sizeAttr PartialityAttr_Total jectivityAttr)
   53                 innerDomainFr@(viewAsDomainTuple -> Just innerDomainFrs)
   54                 innerDomainTo) | all domainCanIndexMatrix innerDomainFrs = do
   55 
   56             let
   57                 kRange = case innerDomainFr of
   58                         DomainTuple ts  -> map fromInt [1 .. genericLength ts]
   59                         DomainRecord rs -> map (fromName . fst) rs
   60                         _ -> bug $ vcat [ "FunctionNDPartial.structuralCons"
   61                                         , "innerDomainFr:" <+> pretty innerDomainFr
   62                                         ]
   63                 toIndex x = [ [essence| &x[&k] |] | k <- kRange ]
   64                 index x m = make opMatrixIndexing m (toIndex x)
   65 
   66             let
   67                 injectiveCons :: Expression -> m [Expression]
   68                 injectiveCons values = do
   69                     tyTo <- typeOfDomain innerDomainTo
   70                     let canAllDiff = case tyTo of
   71                             TypeBool{} -> True
   72                             TypeInt{}  -> True
   73                             TypeEnum{} -> True
   74                             _          -> False
   75                     if canAllDiff
   76                         then do
   77                             (iPat, i) <- quantifiedVar
   78                             let valuesIndexedI = index i values
   79                             return $ return $ -- list
   80                                 [essence|
   81                                     allDiff([ &valuesIndexedI
   82                                             | &iPat : &innerDomainFr
   83                                             ])
   84                                 |]
   85                         else do
   86                             (iPat, i) <- quantifiedVar
   87                             (jPat, j) <- quantifiedVar
   88                             let valuesIndexedI = index i values
   89                             let valuesIndexedJ = index j values
   90                             return $ return $ -- list
   91                                 [essence|
   92                                     forAll &iPat, &jPat : &innerDomainFr .
   93                                         &i .< &j -> &valuesIndexedI != &valuesIndexedJ
   94                                 |]
   95 
   96                 surjectiveCons :: Expression -> m [Expression]
   97                 surjectiveCons values = do
   98                     (iPat, i) <- quantifiedVar
   99                     (jPat, j) <- quantifiedVar
  100                     let valuesIndexedJ = index j values
  101                     return $ return $ -- list
  102                         [essence|
  103                             forAll &iPat : &innerDomainTo .
  104                                 exists &jPat : &innerDomainFr .
  105                                     &valuesIndexedJ = &i
  106                         |]
  107 
  108                 jectivityCons :: Expression -> m [Expression]
  109                 jectivityCons values = case jectivityAttr of
  110                     JectivityAttr_None       -> return []
  111                     JectivityAttr_Injective  -> injectiveCons  values
  112                     JectivityAttr_Surjective -> surjectiveCons values
  113                     JectivityAttr_Bijective  -> (++) <$> injectiveCons  values
  114                                                      <*> surjectiveCons values
  115 
  116                 cardinality :: m Expression
  117                 cardinality = do
  118                     (iPat, _) <- quantifiedVar
  119                     return [essence| sum &iPat : &innerDomainFr . 1 |]
  120 
  121             let innerStructuralCons values = do
  122                     (iPat, i) <- quantifiedVarOverDomain (forgetRepr innerDomainFr)
  123                     let valuesIndexedI = index i values
  124                     let activeZone b = [essence| forAll &iPat : &innerDomainFr . &b |]
  125 
  126                     -- preparing structural constraints for the inner guys
  127                     innerStructuralConsGen <- f innerDomainTo
  128 
  129                     let inLoop = valuesIndexedI
  130                     outs <- innerStructuralConsGen inLoop
  131                     return (map activeZone outs)
  132 
  133             return $ \ func -> do
  134                 refs <- downX1 func
  135                 case refs of
  136                     [values] ->
  137                         concat <$> sequence
  138                             [ jectivityCons values
  139                             , mkSizeCons sizeAttr <$> cardinality
  140                             , innerStructuralCons values
  141                             ]
  142                     _ -> na "{structuralCons} FunctionND"
  143 
  144         structuralCons _ _ _ = na "{structuralCons} FunctionND"
  145 
  146         downC :: TypeOf_DownC m
  147         downC ( name
  148               , domain@(DomainFunction Function_ND
  149                     (FunctionAttr _ PartialityAttr_Total _)
  150                     innerDomainFr@(viewAsDomainTuple -> Just innerDomainFrs)
  151                     innerDomainTo)
  152               , value@(ConstantAbstract (AbsLitFunction vals))
  153               ) | all domainCanIndexMatrix innerDomainFrs
  154                 , Just (_mk, inspect) <- mkLensAsDomainTuple innerDomainFr = do
  155             let
  156                 check :: [Constant] -> Maybe Constant
  157                 check indices = listToMaybe [ v
  158                                             | (inspect -> Just k, v) <- vals
  159                                             , k == indices
  160                                             ]
  161 
  162             let
  163                 unrollD :: [Domain () Constant] -> Domain r Constant -> Domain r Constant
  164                 unrollD is j = foldr DomainMatrix j is
  165 
  166             let
  167                 unrollC :: MonadFail m
  168                         => [Domain () Constant]
  169                         -> [Constant]               -- indices
  170                         -> m Constant
  171                 unrollC [i] prevIndices = do
  172                     domVals <- domainValues i
  173                     let active val = check $ prevIndices ++ [val]
  174 
  175                     missing <- concatForM domVals $ \ val ->
  176                         case active val of
  177                             Nothing -> return [ConstantAbstract $ AbsLitTuple $ prevIndices ++ [val]]
  178                             Just {} -> return []
  179 
  180                     unless (null missing) $
  181                         fail $ vcat [ "Some points are undefined on a total function:" <++> prettyList id "," missing
  182                                     , "    Function:" <+> pretty name
  183                                     , "    Domain:" <++> pretty domain
  184                                     , "    Value :" <++> pretty value
  185                                     ]
  186 
  187                     return $ ConstantAbstract $ AbsLitMatrix i
  188                                 [ fromMaybe (bug $ "FunctionND downC" <+> pretty val) (active val)
  189                                 | val <- domVals ]
  190                 unrollC (i:is) prevIndices = do
  191                     domVals <- domainValues i
  192                     matrixVals <- forM domVals $ \ val ->
  193                         unrollC is (prevIndices ++ [val])
  194                     return $ ConstantAbstract $ AbsLitMatrix i matrixVals
  195                 unrollC is prevIndices = fail $ vcat [ "FunctionND.up.unrollC"
  196                                                      , "    is         :" <+> vcat (map pretty is)
  197                                                      , "    prevIndices:" <+> pretty (show prevIndices)
  198                                                      ]
  199 
  200             outValues <- unrollC (map forgetRepr innerDomainFrs) []
  201             return $ Just
  202                 [ ( nameValues domain name
  203                   , unrollD (map forgetRepr innerDomainFrs) innerDomainTo
  204                   , outValues
  205                   )
  206                 ]
  207 
  208         downC _ = na "{downC} FunctionND"
  209 
  210         up :: TypeOf_Up m
  211         up ctxt (name, domain@(DomainFunction Function_ND
  212                                 (FunctionAttr _ PartialityAttr_Total _)
  213                                 innerDomainFr@(viewAsDomainTuple -> Just innerDomainFrs) _))
  214 
  215             | Just (mk, _inspect) <- mkLensAsDomainTuple innerDomainFr =
  216 
  217             case lookup (nameValues domain name) ctxt of
  218                 Just valuesMatrix -> do
  219                     let
  220                         allIndices :: (MonadFail m, Pretty r) => [Domain r Constant] -> m [[Constant]]
  221                         allIndices = fmap sequence . mapM domainValues
  222 
  223                         index :: MonadFail m => Constant -> [Constant] -> m Constant
  224                         index m [] = return m
  225                         index (ConstantAbstract (AbsLitMatrix indexDomain vals)) (i:is) = do
  226                             froms <- domainValues indexDomain
  227                             case lookup i (zip froms vals) of
  228                                 Nothing -> fail "Value not found. FunctionND.up.index"
  229                                 Just v  -> index v is
  230                         index m is = bug ("FunctionND.up.index" <+> pretty m <+> pretty (show is))
  231 
  232                     indices  <- allIndices innerDomainFrs
  233                     vals     <- forM indices $ \ these -> do
  234                         value <- index valuesMatrix these
  235                         return (mk these, value)
  236                     return ( name
  237                            , ConstantAbstract $ AbsLitFunction vals
  238                            )
  239                 Nothing -> fail $ vcat $
  240                     [ "(in FunctionND up)"
  241                     , "No value for:" <+> pretty (nameValues domain name)
  242                     , "When working on:" <+> pretty name
  243                     , "With domain:" <+> pretty domain
  244                     ] ++
  245                     ("Bindings in context:" : prettyContext ctxt)
  246         up _ _ = na "{up} FunctionND"
  247 
  248         symmetryOrdering :: TypeOf_SymmetryOrdering m
  249         symmetryOrdering innerSO downX1 inp domain = do
  250             [inner] <- downX1 inp
  251             Just [(_, innerDomain)] <- downD ("SO", domain)
  252             innerSO downX1 inner innerDomain

  254 viewAsDomainTuple :: Domain r x -> Maybe [Domain r x]
  255 viewAsDomainTuple (DomainTuple doms) = Just doms
  256 viewAsDomainTuple (DomainRecord doms) = Just (doms |> sortBy (comparing fst) |> map snd)
  257 viewAsDomainTuple _ = Nothing
  258 
  259 
  260 mkLensAsDomainTuple :: Domain r x -> Maybe ( [Constant] -> Constant             -- how to make a literal
  261                                            , Constant -> Maybe [Constant]       -- how to inspect a literal
  262                                            )
  263 mkLensAsDomainTuple (DomainTuple _) =
  264     Just
  265         ( \ vals -> ConstantAbstract (AbsLitTuple vals)
  266         , \ val -> case val of
  267                 ConstantAbstract (AbsLitTuple vals) -> Just vals
  268                 _ -> Nothing
  269         )
  270 mkLensAsDomainTuple (DomainRecord doms) =
  271     let names = doms |> sortBy (comparing fst) |> map fst
  272     in  Just
  273         ( \ vals -> ConstantAbstract (AbsLitRecord (zip names vals))
  274         , \ val -> case val of
  275                 ConstantAbstract (AbsLitRecord vals) -> Just (vals |> sortBy (comparing fst) |> map snd)
  276                 _ -> Nothing
  277         )
  278 mkLensAsDomainTuple _ = Nothing
