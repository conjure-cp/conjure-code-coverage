   24 instance DomainSizeOf Expression Expression where
   25     domainSizeOf (DomainReference _ (Just d)) = domainSizeOf d
   26     domainSizeOf DomainBool = return 2
   27     domainSizeOf (DomainInt _ [] ) = fail "domainSizeOf infinite integer domain"
   28     domainSizeOf (DomainInt _ [r]) = domainSizeOfRange r
   29     domainSizeOf (DomainInt _ rs ) = make opSum . fromList <$> mapM domainSizeOfRange rs
   30     domainSizeOf (DomainEnum n Nothing _) = return $
   31         let n' = n `mappend` "_EnumSize"
   32         in  Reference n' (Just (DeclHasRepr Given n' (DomainInt TagInt [])))
   33     domainSizeOf (DomainUnnamed _ x) = return x
   34     domainSizeOf (DomainTuple []) = return 1
   35     domainSizeOf (DomainTuple xs) = make opProduct . fromList <$> mapM domainSizeOf xs
   36     domainSizeOf (DomainRecord xs) = make opProduct . fromList <$> mapM (domainSizeOf . snd) xs
   37     domainSizeOf (DomainVariant xs) = make opSum . fromList <$> mapM (domainSizeOf . snd) xs
   38     domainSizeOf (DomainMatrix index inner) = make opPow <$> domainSizeOf inner <*> domainSizeOf index
   39     domainSizeOf (DomainSet _ (SetAttr sizeAttr) inner) = do
   40         innerSize <- domainSizeOf inner
   41         case sizeAttr of
   42             SizeAttr_None           -> return (make opPow 2 innerSize)
   43             SizeAttr_Size size      -> return (nchoosek (make opFactorial) innerSize size)
   44             SizeAttr_MinSize _      -> return (make opPow 2 innerSize)              -- TODO: can be better
   45             SizeAttr_MaxSize _      -> return (make opPow 2 innerSize)              -- TODO: can be better
   46             SizeAttr_MinMaxSize _ _ -> return (make opPow 2 innerSize)              -- TODO: can be better
   47     domainSizeOf (DomainMSet _ attrs inner) = do
   48         innerSize <- domainSizeOf inner
   49         let
   50             getMaxSize = case attrs of
   51                 MSetAttr (SizeAttr_Size x) _ -> return x
   52                 MSetAttr (SizeAttr_MaxSize x) _ -> return x
   53                 MSetAttr (SizeAttr_MinMaxSize _ x) _ -> return x
   54                 MSetAttr _ (OccurAttr_MaxOccur x) -> return (x * innerSize)
   55                 MSetAttr _ (OccurAttr_MinMaxOccur _ x) -> return (x * innerSize)
   56                 _ -> fail ("domainSizeOf.getMaxSize, mset not supported. attributes:" <+> pretty attrs)
   57             getMaxOccur = case attrs of
   58                 MSetAttr _ (OccurAttr_MaxOccur x) -> return x
   59                 MSetAttr _ (OccurAttr_MinMaxOccur _ x) -> return x
   60                 MSetAttr (SizeAttr_Size x) _ -> return (make opMin $ fromList [x, innerSize])
   61                 MSetAttr (SizeAttr_MaxSize x) _ -> return (make opMin $ fromList [x, innerSize])
   62                 MSetAttr (SizeAttr_MinMaxSize _ x) _ -> return (make opMin $ fromList [x, innerSize])
   63                 _ -> fail ("domainSizeOf.getMaxSize, mset not supported. attributes:" <+> pretty attrs)
   64         maxSize  <- getMaxSize
   65         maxOccur <- getMaxOccur
   66         return (make opPow maxOccur maxSize)
   67     domainSizeOf d@(DomainSequence _ (SequenceAttr sizeAttr jectivityAttr) innerTo) = do
   68         size <- case sizeAttr of
   69             SizeAttr_None           -> fail ("Infinite domain:" <+> pretty d)
   70             SizeAttr_Size s         -> return s
   71             SizeAttr_MinSize _      -> fail ("Infinite domain:" <+> pretty d)
   72             SizeAttr_MaxSize s      -> return s
   73             SizeAttr_MinMaxSize _ s -> return s
   74         domainSizeOf $ DomainFunction def (FunctionAttr sizeAttr PartialityAttr_Partial jectivityAttr)
   75             (DomainInt TagInt [RangeBounded 1 size]) innerTo
   76     domainSizeOf (DomainFunction _ (FunctionAttr sizeAttr _ _) innerFr innerTo) =
   77         domainSizeOf $ DomainRelation def (RelationAttr sizeAttr def) [innerFr, innerTo]
   78     domainSizeOf (DomainRelation _ (RelationAttr sizeAttr _binRelAttr) inners) =
   79         domainSizeOf (DomainSet def (SetAttr sizeAttr) (DomainTuple inners))
   80     domainSizeOf (DomainPartition _ a inner) =
   81         domainSizeOf $ DomainSet def (SetAttr (partsNum  a))
   82                       $ DomainSet def (SetAttr (partsSize a)) inner
   83     domainSizeOf d = bug ("not implemented: domainSizeOf:" <+> vcat [pretty d, pretty (show d)])
   84

   86 domainSizeOfRange :: (Op a :< a, ExpressionLike a, Pretty a, MonadFail m, Num a, Eq a) => Range a -> m a
   87 domainSizeOfRange RangeSingle{} = return 1
   88 domainSizeOfRange (RangeBounded 1 u) = return u
   89 domainSizeOfRange (RangeBounded l u) = return $ make opSum $ fromList [1, make opMinus u l]
   90 domainSizeOfRange r = fail ("domainSizeOf infinite range:" <+> pretty r)

   93 getMaxNumberOfElementsInContainer :: Domain () Expression -> Expression
   94 getMaxNumberOfElementsInContainer domain@(DomainSet _ (SetAttr sizeAttr) inner) =
   95     case (getMaxFrom_SizeAttr sizeAttr, domainSizeOf inner) of
   96         (Just n, _) -> n
   97         (_, Just n) -> n
   98         _           -> bug $ "getMaxNumberOfElementsInContainer, DomainSet:" <+> pretty domain
   99 getMaxNumberOfElementsInContainer domain@(DomainMSet _ (MSetAttr sizeAttr occurAttr) inner) =
  100     case (getMaxFrom_SizeAttr sizeAttr, getMaxFrom_OccurAttr occurAttr, domainSizeOf inner) of
  101         (Just n, _     , _     ) -> n
  102         (_     , Just o, Just n) -> [essence| &o * &n |]
  103         _                        -> bug $ "getMaxNumberOfElementsInContainer, DomainMSet:" <+> pretty domain
  104 getMaxNumberOfElementsInContainer domain@(DomainSequence _ (SequenceAttr sizeAttr _) _) =
  105     case getMaxFrom_SizeAttr sizeAttr of
  106         Just n -> n
  107         _      -> bug $ "getMaxNumberOfElementsInContainer, DomainSequence:" <+> pretty domain
  108 getMaxNumberOfElementsInContainer domain = bug $ "getMaxNumberOfElementsInContainer:" <+> pretty domain
