   20 type DomainX x = Domain HasRepresentation x
   21 type DomainC = Domain HasRepresentation Constant
   22 
   23 -- | This data type represents a representation selection rule.
   24 -- The 3 functions -- rDownD, rDownC, and rUp -- all work one level at a time.
   25 -- The maybe for rDownD and rDownC is Nothing when representation doesn't change anything.
   26 -- Like for primitives.
   27 -- * rDownD is for refining a class level domain.
   28 -- * rDownC is for refining constants, together with an instance level domain.
   29 -- * rUp is for translating low level constants upwards
   30 --   It takes in a instance level domain for the high level object.
   31 -- * rCheck is for calculating all representation options.
   32 --   It take a function to be used as a "checker" for inner domains, if any.
   33 data Representation (m :: * -> *) = Representation
   34     { rCheck            :: TypeOf_ReprCheck        m
   35     , rDownD            :: TypeOf_DownD            m
   36     , rStructural       :: TypeOf_Structural       m
   37     , rDownC            :: TypeOf_DownC            m
   38     , rUp               :: TypeOf_Up               m
   39     , rSymmetryOrdering :: TypeOf_SymmetryOrdering m
   40     }
   41 
   42 type TypeOf_ReprCheck (m :: * -> *) =
   43        forall x . (Data x, Pretty x, ExpressionLike x)
   44     => (Domain () x -> m [DomainX x])               -- other checkers for inner domains
   45     -> Domain () x                                  -- this domain
   46     -> m [DomainX x]                                -- with all repr options
   47 
   48 type TypeOf_DownD (m :: * -> *) =
   49                  (Name, DomainX Expression)
   50     -> m (Maybe [(Name, DomainX Expression)])
   51 
   52 type TypeOf_SymmetryOrdering (m :: * -> *) =
   53        ((Expression -> m [Expression]) -> Expression -> DomainX Expression -> m Expression) -- inner S.O.
   54     -> (Expression -> m [Expression])               -- general downX1
   55     -> Expression                                   -- this as an expression
   56     -> DomainX Expression                           -- name and domain
   57     -> m Expression                                 -- output, of type [int]
   58 
   59 type TypeOf_Structural (m :: * -> *) =
   60        (DomainX Expression -> m (Expression -> m [Expression]))
   61                                                     -- other structural constraints for inner domains
   62     -> (Expression -> m [Expression])               -- general downX1
   63     -> DomainX Expression                           -- this domain
   64     -> m (      Expression                          -- the original variable, before refinement
   65           -> m [Expression]                         -- structural constraints
   66          )
   67 
   68 type TypeOf_DownC (m :: * -> *) =
   69                  (Name, DomainC, Constant)          -- the input name, domain and constant
   70     -> m (Maybe [(Name, DomainC, Constant)])        -- the outputs names, domains, and constants
   71 
   72 type TypeOf_Up (m :: * -> *) =
   73         [(Name, Constant)] ->                       -- all known constants, representing a solution at the low level
   74         (Name, DomainC) ->                          -- the name and domain we are working on
   75         m (Name, Constant)                          -- the output constant, at the high level
   76

   78 type DispatchFunction m x =
   79         Data x =>
   80         Pretty x =>
   81         Domain HasRepresentation x ->
   82         Representation m
   83 
   84 type ReprOptionsFunction m r x =
   85         Data x =>
   86         Pretty x =>
   87         ExpressionLike x =>
   88         Domain () x ->
   89         m [Domain HasRepresentation x]

   92 rDownToX ::
   93     Monad m =>
   94     Representation m ->                             -- for a given representation
   95     FindOrGiven ->                                  -- and a declaration: forg
   96     Name ->                                         --                  : name
   97     Domain HasRepresentation Expression ->          --                  : domain
   98     m [Expression]                                  -- expressions referring to the representation
   99 rDownToX repr forg name domain = do
  100     pairs <- rDownD repr (name, domain)
  101     return [ Reference n (Just (DeclHasRepr forg n d))
  102            | (n,d) <- concat pairs
  103            ]

  105 mkOutName :: Maybe Name -> Domain HasRepresentation x -> Name -> Name
  106 mkOutName Nothing       domain origName = mconcat [origName, "_", Name (reprTreeEncoded domain)]
  107 mkOutName (Just suffix) domain origName = mconcat [origName, "_", Name (reprTreeEncoded domain), "_", suffix]
