   15 relationAsSet
   16     :: forall m . (MonadFail m, NameGen m)
   17     => (forall x . DispatchFunction m x)
   18     -> (forall r x . ReprOptionsFunction m r x)
   19     -> Bool
   20     -> Representation m
   21 relationAsSet dispatch reprOptions useLevels = Representation chck downD structuralCons downC up symmetryOrdering
   22 
   23     where
   24 
   25         chck :: TypeOf_ReprCheck m
   26         chck _ dom1@(DomainRelation _ attrs _) = do
   27             dom2 <- outDomain_ dom1
   28             dom3 <- reprOptions dom2
   29             return [ DomainRelation (Relation_AsSet r) attrs innerDomains
   30                    | DomainSet r _ (DomainTuple innerDomains) <- dom3
   31                    -- special hack: do not use Set_ExplicitVarSizeWithFlags when --representation-levels=yes
   32                    , if useLevels
   33                        then r /= Set_ExplicitVarSizeWithFlags
   34                        else True
   35                    ]
   36         chck _ _ = return []
   37 
   38         outName :: Domain HasRepresentation x -> Name -> Name
   39         outName = mkOutName Nothing
   40 
   41         outDomain_ :: Pretty x => Domain () x -> m (Domain () x)
   42         outDomain_ (DomainRelation () (RelationAttr sizeAttr _binRelAttrs) innerDomains) =
   43             return (DomainSet () (SetAttr sizeAttr) (DomainTuple innerDomains))
   44         outDomain_ domain = na $ vcat [ "{outDomain_} RelationAsSet"
   45                                       , "domain:" <+> pretty domain
   46                                       ]
   47 
   48         outDomain :: Pretty x => Domain HasRepresentation x -> m (Domain HasRepresentation x)
   49         outDomain (DomainRelation (Relation_AsSet repr) (RelationAttr sizeAttr _binRelAttrs) innerDomains) =
   50             return (DomainSet repr (SetAttr sizeAttr) (DomainTuple innerDomains))
   51         outDomain domain = na $ vcat [ "{outDomain} RelationAsSet"
   52                                      , "domain:" <+> pretty domain
   53                                      ]
   54 
   55         downD :: TypeOf_DownD m
   56         downD (name, inDom) = do
   57             outDom <- outDomain inDom
   58             return $ Just [ ( outName inDom name , outDom ) ]
   59 
   60         structuralCons :: TypeOf_Structural m
   61         structuralCons f downX1 inDom = do
   62             let
   63                 innerStructuralCons rel = do
   64                     outDom                 <- outDomain inDom
   65                     innerStructuralConsGen <- f outDom
   66                     innerStructuralConsGen rel
   67 
   68             return $ \ rel -> do
   69                 refs <- downX1 rel
   70                 case refs of
   71                     [set] -> do
   72                         binRelCons <- case inDom of
   73                             DomainRelation Relation_AsSet{} (RelationAttr _ binRelAttrs) [innerDomain1, innerDomain2]
   74                                 | binRelAttrs == def
   75                                     -> return []
   76                                 | forgetRepr innerDomain1 == forgetRepr innerDomain2
   77                                     -> mkBinRelCons binRelAttrs innerDomain1 rel
   78                                 | otherwise
   79                                     -> bug $ vcat [ "Binary relation between different domains. (RelationAsSet)"
   80                                                   , "innerDomain1:" <+> pretty innerDomain1
   81                                                   , "innerDomain2:" <+> pretty innerDomain2
   82                                                   ]
   83                             DomainRelation Relation_AsSet{} (RelationAttr _ binRelAttrs) innerDomains
   84                                 | length innerDomains /= 2 && binRelAttrs /= def
   85                                     -> bug "Non-binary relation has binary relation attributes."
   86                             _ -> return []
   87                         concat <$> sequence
   88                             [ innerStructuralCons set
   89                             , return binRelCons
   90                             ]
   91                     _ -> na $ vcat [ "{structuralCons} RelationAsSet"
   92                                    , pretty inDom
   93                                    ]
   94 
   95         downC :: TypeOf_DownC m
   96         downC ( name
   97               , inDom
   98               , ConstantAbstract (AbsLitRelation vals)
   99               ) = do
  100             outDom <- outDomain inDom
  101             rDownC
  102                 (dispatch outDom)
  103                 ( outName inDom name
  104                 , outDom
  105                 , ConstantAbstract $ AbsLitSet $ map (ConstantAbstract . AbsLitTuple) vals
  106                 )
  107         downC (name, domain, constant) = na $ vcat [ "{downC} RelationAsSet"
  108                                                    , "name:" <+> pretty name
  109                                                    , "domain:" <+> pretty domain
  110                                                    , "constant:" <+> pretty constant
  111                                                    ]
  112 
  113         up :: TypeOf_Up m
  114         up ctxt (name, domain@(DomainRelation Relation_AsSet{} _ _)) =
  115             case lookup (outName domain name) ctxt of
  116                 Just (ConstantAbstract (AbsLitSet tuples)) -> do
  117                     vals <- mapM viewConstantTuple tuples
  118                     return (name, ConstantAbstract (AbsLitRelation vals))
  119                 Nothing -> fail $ vcat $
  120                     [ "(in RelationAsSet up)"
  121                     , "No value for:" <+> pretty (outName domain name)
  122                     , "When working on:" <+> pretty name
  123                     , "With domain:" <+> pretty domain
  124                     ] ++
  125                     ("Bindings in context:" : prettyContext ctxt)
  126                 Just constant -> fail $ vcat $
  127                     [ "Incompatible value for:" <+> pretty (outName domain name)
  128                     , "When working on:" <+> pretty name
  129                     , "With domain:" <+> pretty domain
  130                     , "Expected a set value, but got:" <++> pretty constant
  131                     ] ++
  132                     ("Bindings in context:" : prettyContext ctxt)
  133         up _ (name, domain) = na $ vcat [ "{up} RelationAsSet"
  134                                         , "name:" <+> pretty name
  135                                         , "domain:" <+> pretty domain
  136                                         ]
  137 
  138         symmetryOrdering :: TypeOf_SymmetryOrdering m
  139         symmetryOrdering innerSO downX1 inp domain = do
  140             [inner] <- downX1 inp
  141             Just [(_, innerDomain)] <- downD ("SO", domain)
  142             innerSO downX1 inner innerDomain
