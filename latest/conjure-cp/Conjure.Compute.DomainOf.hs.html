   20     -- | calculate the domain of `a`
   21     domainOf ::
   22         MonadFail m =>
   23         NameGen m =>
   24         (?typeCheckerMode :: TypeCheckerMode) =>
   25         a -> m Dom
   26 
   27     -- | calculate the index domains of `a`
   28     --   the index is the index of a matrix.
   29     --   returns [] for non-matrix inputs.
   30     --   has a default implementation in terms of domainOf, so doesn't need to be implemented specifically.
   31     --   but sometimes it is better to implement this directly.
   32     indexDomainsOf ::
   33         MonadFail m =>
   34         NameGen m =>
   35         Pretty a =>
   36         (?typeCheckerMode :: TypeCheckerMode) =>
   37         a -> m [Dom]
   38     indexDomainsOf = defIndexDomainsOf
   39

   41 domainOfR ::
   42     DomainOf a =>
   43     RepresentationOf a =>
   44     MonadFail m =>
   45     NameGen m =>
   46     (?typeCheckerMode :: TypeCheckerMode) =>
   47     a -> m (Domain HasRepresentation Expression)
   48 domainOfR inp = do
   49     dom <- domainOf inp
   50     rTree <- representationTreeOf inp
   51     applyReprTree dom rTree
   52

   54 defIndexDomainsOf ::
   55     MonadFail m =>
   56     NameGen m =>
   57     DomainOf a =>
   58     (?typeCheckerMode :: TypeCheckerMode) =>
   59     a -> m [Dom]
   60 defIndexDomainsOf x = do
   61     dom <- domainOf x
   62     let
   63         collect (DomainMatrix index inner) = index : collect inner
   64         collect _ = []
   65     return (collect dom)

   67 instance DomainOf ReferenceTo where
   68     domainOf (Alias x) = domainOf x
   69     domainOf (InComprehension (GenDomainNoRepr Single{} dom)) = return dom
   70     domainOf (InComprehension (GenDomainHasRepr _ dom)) = return (forgetRepr dom)
   71     domainOf (InComprehension (GenInExpr Single{} x)) = domainOf x >>= innerDomainOf
   72     domainOf x@InComprehension{} = fail $ vcat [ "domainOf-ReferenceTo-InComprehension", pretty x, pretty (show x) ]
   73     domainOf (DeclNoRepr  _ _ dom _) = return dom
   74     domainOf (DeclHasRepr _ _ dom  ) = return (forgetRepr dom)
   75     domainOf RecordField{}  = fail "domainOf-ReferenceTo-RecordField"
   76     domainOf VariantField{} = fail "domainOf-ReferenceTo-VariantField"

   88     indexDomainsOf (Typed lit ty) | emptyCollectionX lit =
   89         let
   90             tyToDom (TypeMatrix (TypeInt nm) t) = DomainInt nm [RangeBounded 1 0] : tyToDom t
   91             tyToDom _ = []
   92         in
   93             return (tyToDom ty)

  252     domainOf i@(ConstantInt t _)        = return $ DomainInt t [RangeSingle (Constant i)]
  253     domainOf (ConstantEnum defn _ _ )   = return (DomainEnum defn Nothing Nothing)

  269 instance DomainOf (AbstractLiteral Expression) where
  270 
  271     domainOf (AbsLitTuple        xs) = DomainTuple  <$> mapM domainOf xs
  272 
  273     domainOf (AbsLitRecord       xs) = DomainRecord <$> sequence [ do t <- domainOf x ; return (n,t)
  274                                                                  | (n,x) <- xs ]
  275 
  276     domainOf (AbsLitVariant Nothing  _ _) = fail "Cannot calculate the domain of variant literal."
  277     domainOf (AbsLitVariant (Just t) _ _) = return (DomainVariant t)
  278 
  279     domainOf (AbsLitMatrix ind inn ) = DomainMatrix ind <$> (domainUnions =<< mapM domainOf inn)
  280 
  281     domainOf (AbsLitSet         [] ) = return $ DomainSet def attr (DomainAny "domainOf-AbsLitSet-[]" TypeAny)
  282         where attr = SetAttr (SizeAttr_Size 0)
  283     domainOf (AbsLitSet         xs ) = DomainSet def attr <$> (domainUnions =<< mapM domainOf xs)
  284         where attr = SetAttr (SizeAttr_MaxSize $ fromInt $ genericLength xs)
  285 
  286     domainOf (AbsLitMSet        [] ) = return $ DomainMSet def attr (DomainAny "domainOf-AbsLitMSet-[]" TypeAny)
  287         where attr = MSetAttr (SizeAttr_Size 0) OccurAttr_None
  288     domainOf (AbsLitMSet        xs ) = DomainMSet def attr <$> (domainUnions =<< mapM domainOf xs)
  289         where attr = MSetAttr (SizeAttr_MaxSize $ fromInt $ genericLength xs) OccurAttr_None
  290 
  291     domainOf (AbsLitFunction    [] ) = return $ DomainFunction def attr
  292                                         (DomainAny "domainOf-AbsLitFunction-[]-1" TypeAny)
  293                                         (DomainAny "domainOf-AbsLitFunction-[]-2" TypeAny)
  294         where attr = FunctionAttr (SizeAttr_Size 0) def def
  295     domainOf (AbsLitFunction    xs ) = DomainFunction def attr
  296                                                 <$> (domainUnions =<< mapM (domainOf . fst) xs)
  297                                                 <*> (domainUnions =<< mapM (domainOf . snd) xs)
  298         where attr = FunctionAttr (SizeAttr_MaxSize $ fromInt $ genericLength xs) def def
  299 
  300     domainOf (AbsLitSequence    [] ) = return $ DomainSequence def attr
  301                                         (DomainAny "domainOf-AbsLitSequence-[]" TypeAny)
  302         where attr = SequenceAttr (SizeAttr_Size 0) def
  303     domainOf (AbsLitSequence    xs ) = DomainSequence def attr
  304                                                 <$> (domainUnions =<< mapM domainOf xs)
  305         where attr = SequenceAttr (SizeAttr_MaxSize (fromInt $ genericLength xs)) def
  306 
  307     domainOf (AbsLitRelation    [] ) = return $ DomainRelation def attr []
  308         where attr = RelationAttr (SizeAttr_Size 0) def
  309     domainOf (AbsLitRelation    xss) = do
  310         ty <- domainUnions =<< mapM (domainOf . AbsLitTuple) xss
  311         case ty of
  312             DomainTuple ts -> return (DomainRelation def attr ts)
  313             _ -> bug "expecting DomainTuple in domainOf"
  314         where attr = RelationAttr (SizeAttr_MaxSize (fromInt $ genericLength xss)) def
  315 
  316     domainOf (AbsLitPartition   [] ) = return $ DomainPartition def attr
  317                                         (DomainAny "domainOf-AbsLitPartition-[]" TypeAny)
  318         where attr = PartitionAttr (SizeAttr_Size 0) (SizeAttr_Size 0) False
  319     domainOf (AbsLitPartition   xss) = DomainPartition def attr <$> (domainUnions =<< mapM domainOf (concat xss))
  320         where attr = PartitionAttr (SizeAttr_MaxSize (fromInt $ genericLength xss))
  321                                    (SizeAttr_MaxSize (fromInt $ maximum [genericLength xs | xs <- xss]))
  322                                    False
  323 
  324     indexDomainsOf (AbsLitMatrix ind inn) = (ind :) <$> (mapM domainUnions =<< mapM indexDomainsOf inn)
  325     indexDomainsOf _ = return []

  359 instance DomainOf x => DomainOf (OpDiv x) where
  360     domainOf (OpDiv x y) = do
  361         xDom :: Dom <- domainOf x
  362         yDom :: Dom <- domainOf y
  363         (iPat, i) <- quantifiedVar
  364         (jPat, j) <- quantifiedVar
  365         let vals = [essence| [ &i / &j
  366                              | &iPat : &xDom
  367                              , &jPat : &yDom
  368                              ] |]
  369         let low  = [essence| min(&vals) |]
  370         let upp  = [essence| max(&vals) |]
  371         return (DomainInt TagInt [RangeBounded low upp] :: Dom)

  424 instance (Pretty x, TypeOf x, ExpressionLike x, DomainOf x) => DomainOf (OpIndexing x) where
  425     domainOf (OpIndexing m i) = do
  426         iType <- typeOf i
  427         case iType of
  428             TypeBool{} -> return ()
  429             TypeInt{} -> return ()
  430             _ -> fail "domainOf, OpIndexing, not a bool or int index"
  431         mDom <- domainOf m
  432         case mDom of
  433             DomainMatrix _ inner -> return inner
  434             DomainTuple inners -> do
  435                 iInt <- intOut "domainOf OpIndexing" i
  436                 return $ atNote "domainOf" inners (fromInteger (iInt-1))
  437             _ -> fail "domainOf, OpIndexing, not a matrix or tuple"
  438 
  439     indexDomainsOf p@(OpIndexing m i) = do
  440         iType <- typeOf i
  441         case iType of
  442             TypeBool{} -> return ()
  443             TypeInt{} -> return ()
  444             _ -> fail "domainOf, OpIndexing, not a bool or int index"
  445         is <- indexDomainsOf m
  446         case is of
  447             [] -> fail ("indexDomainsOf{OpIndexing}, not a matrix domain:" <++> pretty p)
  448             (_:is') -> return is'

  471 instance (Pretty x, TypeOf x, ExpressionLike x, DomainOf x, Domain () x :< x) => DomainOf (OpMax x) where
  472     domainOf (OpMax x)
  473         | Just xs <- listOut x
  474         , not (null xs) = do
  475         doms <- mapM domainOf xs
  476         let lows = fromList [ [essence| min(`&d`) |] | d <- doms ]
  477         let low  = [essence| max(&lows) |]
  478         let upps = fromList [ [essence| max(`&d`) |] | d <- doms ]
  479         let upp  = [essence| max(&upps) |]
  480         TypeInt t <- typeOfDomain (head doms)
  481         return (DomainInt t [RangeBounded low upp] :: Dom)
  482     domainOf op = mkDomainAny ("OpMax:" <++> pretty op) <$> typeOf op
