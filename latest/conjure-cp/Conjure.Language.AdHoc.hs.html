   48 data MiniZincData = MZNBool Bool
   49                   | MZNInt Integer
   50                   | MZNArray (Maybe String) [MiniZincData] -- index if any, then data
   51                   | MZNSet [MiniZincData]
   52                   | MZNNamed [(Name, MiniZincData)]
   53     deriving (Eq, Ord, Show)
   54 
   55 instance Pretty MiniZincData where
   56     pretty (MZNBool x) = pretty x
   57     pretty (MZNInt x) = pretty x
   58     pretty (MZNArray index xs) =
   59         let
   60             nestedPretty (MZNArray _ ys) = prettyList id "," ys
   61             nestedPretty y = pretty y
   62 
   63             fillNothingIndices (MZNArray Nothing ys) = MZNArray (Just $ "1.." ++ show (length ys)) (map fillNothingIndices ys)
   64             fillNothingIndices (MZNArray (Just index2) ys) = MZNArray (Just index2) (map fillNothingIndices ys)
   65             fillNothingIndices m@MZNBool{} = m
   66             fillNothingIndices m@MZNInt{} = m
   67             fillNothingIndices (MZNSet ys) = MZNSet (map fillNothingIndices ys)
   68             fillNothingIndices (MZNNamed ys) = MZNNamed [(n, fillNothingIndices y) | (n, y) <- ys]
   69 
   70             calcIndices (MZNArray index2 []) = [index2]
   71             calcIndices (MZNArray index2 (y:_)) = index2 : calcIndices y
   72             calcIndices _ = []
   73 
   74             indices = calcIndices $ fillNothingIndices $ MZNArray index xs
   75             depth = length indices
   76 
   77             args = [pretty i | Just i <- indices] ++ [prettyList prBrackets "," (map nestedPretty xs)]
   78         in
   79             "array" <> pretty depth <> "d" <> prettyList prParens "," args
   80     pretty (MZNSet xs) = prettyList prBraces "," (map pretty xs)
   81     pretty (MZNNamed xs) = vcat [pretty n <+> "=" <+> pretty x <> ";" | (n,x) <- xs]

   99 class SimpleJSON a where
  100     toSimpleJSON :: MonadUserError m => a -> m JSON.Value
  101     fromSimpleJSON :: MonadUserError m => Type -> JSON.Value -> m a
  102 
  103 instance SimpleJSON Integer where
  104     toSimpleJSON = return . toJSON
  105     fromSimpleJSON t x =
  106         case x of
  107             JSON.Number y ->
  108                 case floatingOrInteger y of
  109                     Right z -> return z
  110                     Left (d :: Double) -> noFromSimpleJSON "Integer" t d
  111             JSON.String text ->
  112                 case readMay (textToString text) of
  113                     Just z -> return z
  114                     Nothing -> noFromSimpleJSON "Integer" t text
  115             _ -> noFromSimpleJSON "Integer" t x

  117 data AsDictionary a b = AsDictionary [(a,b)]
  118 
  119 instance (Pretty x, SimpleJSON x, SimpleJSON y) => SimpleJSON (AsDictionary x y) where
  120     toSimpleJSON (AsDictionary xs) = do
  121         (ys, asList) <- fmap unzip $ forM xs $ \ (a,b) -> do
  122             let aStr = stringToText $ renderNormal $ pretty a
  123             aJSON <- toSimpleJSON a
  124             bJSON <- toSimpleJSON b
  125             let abPair = JSON.Array $ V.fromList [aJSON, bJSON]
  126             case aJSON of
  127                 JSON.Bool{}   -> return (Just (aStr, bJSON), abPair)
  128                 JSON.Number{} -> return (Just (aStr, bJSON), abPair)
  129                 JSON.String{} -> return (Just (aStr, bJSON), abPair)
  130                 _             -> return (Nothing           , abPair)
  131         let zs = catMaybes ys
  132         if length ys == length zs
  133             -- all were suitable as keys, great
  134             then return $ JSON.Object $ M.fromList zs
  135             else return $ JSON.Array $ V.fromList asList
  136     fromSimpleJSON = noFromSimpleJSON "AsDictionary"
  137 
  138 instance SimpleJSON x => SimpleJSON [x] where
  139     toSimpleJSON xs = do
  140         ys <- mapM toSimpleJSON xs
  141         return $ JSON.Array $ V.fromList ys
  142     fromSimpleJSON = noFromSimpleJSON "list"
  143 
  144 instance (SimpleJSON x, SimpleJSON y) => SimpleJSON (x,y) where
  145     toSimpleJSON (x,y) = do
  146         x' <- toSimpleJSON x
  147         y' <- toSimpleJSON y
  148         return $ JSON.Array $ V.fromList [x', y']
  149     fromSimpleJSON = noFromSimpleJSON "pair"
