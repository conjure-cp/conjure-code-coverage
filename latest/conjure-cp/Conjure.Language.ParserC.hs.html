   29 parseModel :: Parser Model
   30 parseModel = inCompleteFile $ do
   31     let
   32         pLanguage :: Parser LanguageVersion
   33         pLanguage = do
   34             lexeme L_language
   35             pos1 <- getPosition
   36             l <- identifierText
   37             -- ESSENCE' is accepted, just for convenience
   38             unless (l `elem` ["Essence", "ESSENCE", "ESSENCE'"]) $ do
   39                 setPosition pos1
   40                 fail $ "language name has to be Essence, but given:" <+> pretty l
   41             pos2 <- getPosition
   42             is <- sepBy1 integer dot
   43             unless (is >= [1]) $ do
   44                 setPosition pos2
   45                 fail $ "language version expected to be at least 1.0, but given:" <+>
   46                             pretty (intercalate "." (map show is))
   47             return (LanguageVersion (Name l) (map fromInteger is))
   48     l  <- optional pLanguage
   49     xs <- many parseTopLevels
   50     return Model
   51         { mLanguage = fromMaybe def l
   52         , mStatements = concat xs
   53         , mInfo = def
   54         }

   61 parseTopLevels :: Parser [Statement]
   62 parseTopLevels = do
   63     let one = do
   64                 lexeme L_letting
   65                 i <- parseName
   66                 lexeme L_be
   67                 msum
   68                     [ do
   69                         lexeme L_new
   70                         lexeme L_type
   71                         msum
   72                             [ do
   73                                 lexeme L_of
   74                                 lexeme $ LIdentifier "size"
   75                                 j <- parseExpr
   76                                 return $ Declaration (LettingDomainDefnUnnamed i j)
   77                             , do
   78                                 lexeme L_enum
   79                                 ys <- braces (commaSeparated parseName) <|> return []
   80                                 modify (\ st -> st { enumDomains = [i] ++ enumDomains st } )
   81                                 return $ Declaration (LettingDomainDefnEnum i ys)
   82                             ]
   83                     , do
   84                         lexeme L_domain
   85                         j <- parseDomain
   86                         return $ Declaration (Letting i (Domain j))
   87                     , do
   88                         j <- parseExpr
   89                         return $ Declaration (Letting i j)
   90                     ]
   91                 <?> "letting statement"
   92     some one

   94 parseRange :: Parser a -> Parser (Range a)
   95 parseRange p = msum [try pRange, pSingle] <?> "range"
   96     where
   97         pRange = do
   98             fr <- optional p
   99             dotdot
  100             to <- optional p
  101             return $ case (fr,to) of
  102                 (Nothing, Nothing) -> RangeOpen
  103                 (Just x , Nothing) -> RangeLowerBounded x
  104                 (Nothing, Just y ) -> RangeUpperBounded y
  105                 (Just x , Just y ) -> RangeBounded x y
  106         pSingle = do
  107             x <- p
  108             return (RangeSingle x)

  110 parseDomain :: Parser (Domain () Expression)
  111 parseDomain = (forgetRepr <$> parseDomainWithRepr) <?> "domain"
  112 
  113 parseDomainWithRepr :: Parser (Domain HasRepresentation Expression)
  114 parseDomainWithRepr = pDomainAtom
  115 
  116     where
  117 
  118         pDomainAtom = msum
  119             [ pBool, try pIntFromExpr, pInt, try pEnum, try pReference
  120             , pMatrix, try pTupleWithout, pTupleWith
  121             , pRecord, pVariant
  122             , pSet
  123             , pMSet
  124             , try pFunction', pFunction
  125             , pSequence
  126             , pRelation
  127             , pPartition
  128             , DomainMetaVar <$> parseMetaVariable, parens parseDomainWithRepr
  129             ]
  130 
  131         pBool = do
  132             lexeme L_bool
  133             -- parse and discard, compatibility with SR
  134             _ <- optional $ parens $ commaSeparated0 $ parseRange parseExpr
  135             return DomainBool
  136 
  137         pIntFromExpr = do
  138             lexeme L_int
  139             x <- parens parseExpr
  140             case (let ?typeCheckerMode = StronglyTyped in typeOf x) of
  141                 Just (TypeInt TagInt) -> return $ DomainInt TagInt [RangeSingle x]
  142                 _ -> return $ DomainIntE x
  143 
  144         pInt = do
  145             lexeme L_int
  146             mxs <- optional $ parens $ commaSeparated0 $ parseRange parseExpr
  147             let xs = fromMaybe [] mxs
  148             return $ DomainInt TagInt xs
  149 
  150         pReference = do
  151             r  <- identifierText
  152             return $ DomainReference (Name r) Nothing
  153 
  154         pEnum = do
  155             r  <- identifierText
  156             xs <- optional $ parens $ commaSeparated0 $ parseRange parseExpr
  157             st <- get
  158             guard (Name r `elem` enumDomains st)
  159             return $ DomainEnum (Name r) xs Nothing
  160 
  161         pMatrix = do
  162             lexeme L_matrix
  163             lexeme L_indexed
  164             lexeme L_by
  165             xs <- brackets (commaSeparated parseDomain)
  166             lexeme L_of
  167             y  <- parseDomainWithRepr
  168             return $ foldr DomainMatrix y xs
  169 
  170         pTupleWith = do
  171             lexeme L_tuple
  172             xs <- parens $ commaSeparated0 parseDomainWithRepr
  173             return $ DomainTuple xs
  174 
  175         pTupleWithout = do
  176             xs <- parens $ countSepAtLeast 2 parseDomainWithRepr comma
  177             return $ DomainTuple xs
  178 
  179         pRecord = do
  180             lexeme L_record
  181             let one = do n <- parseName
  182                          lexeme L_Colon
  183                          d <- parseDomainWithRepr
  184                          return (n,d)
  185             xs <- braces $ commaSeparated0 one
  186             return $ DomainRecord xs
  187 
  188         pVariant = do
  189             lexeme L_variant
  190             let one = do n <- parseName
  191                          lexeme L_Colon
  192                          d <- parseDomainWithRepr
  193                          return (n,d)
  194             xs <- braces $ commaSeparated0 one
  195             return $ DomainVariant xs
  196 
  197         pSet = do
  198             lexeme L_set
  199             x <- parseSetAttr
  200             y <- lexeme L_of >> parseDomainWithRepr
  201             return $ DomainSet NoRepresentation x y
  202 
  203         pMSet = do
  204             lexeme L_mset
  205             x <- parseMSetAttr
  206             y <- lexeme L_of >> parseDomainWithRepr
  207             return $ DomainMSet NoRepresentation x y
  208 
  209         pFunction' = do
  210             lexeme L_function
  211             (y,z) <- arrowedPair parseDomainWithRepr
  212             return $ DomainFunction NoRepresentation def y z
  213 
  214         pFunction = do
  215             lexeme L_function
  216             x <- parseFunctionAttr
  217             (y,z) <- arrowedPair parseDomainWithRepr
  218             return $ DomainFunction NoRepresentation x y z
  219 
  220         pSequence = do
  221             lexeme L_sequence
  222             x <- parseSequenceAttr
  223             y <- lexeme L_of >> parseDomainWithRepr
  224             return $ DomainSequence NoRepresentation x y
  225 
  226         pRelation = do
  227             lexeme L_relation
  228             pos <- getPosition
  229             x  <- parseRelationAttr
  230             lexeme L_of
  231             ys <- parens (parseDomainWithRepr `sepBy` lexeme L_Times)
  232             let RelationAttr _ (BinaryRelationAttrs binAttrs) = x
  233             when (length ys /= 2 && not (S.null binAttrs)) $ do
  234                 setPosition pos
  235                 fail $ "Only binary relations can have these attributes:" <+>
  236                             prettyList id "," (S.toList binAttrs)
  237             return $ DomainRelation NoRepresentation x ys
  238 
  239         pPartition = do
  240             lexeme L_partition
  241             x <- parsePartitionAttr
  242             lexeme L_from
  243             y <- parseDomainWithRepr
  244             return $ DomainPartition NoRepresentation x y

  246 parseAttributes :: Parser (DomainAttributes Expression)
  247 parseAttributes = do
  248     xs <- parens (commaSeparated0 parseAttribute) <|> return []
  249     return $ DomainAttributes xs
  250     where
  251         parseAttribute = msum [parseDontCare, try parseNameValue, parseDAName]
  252         parseNameValue = DANameValue <$> (Name <$> identifierText) <*> parseExpr
  253         parseDAName = DAName <$> (Name <$> identifierText)
  254         parseDontCare = do dotdot ; return DADotDot

  428 checkExtraAttributes :: SourcePos -> Doc -> [DomainAttribute a] -> [Name] -> Parser ()
  429 checkExtraAttributes pos ty attrs supported = do
  430     let extras = mapMaybe f attrs
  431     unless (null extras) $ do
  432         setPosition pos
  433         fail $ vcat [ "Unsupported attributes for" <+> ty <> ":" <+> prettyList id "," extras
  434                     , "Only these are supported:" <+> prettyList id "," supported
  435                     ]
  436     where
  437         f (DANameValue nm _) | nm `notElem` supported = Just nm
  438         f (DAName      nm  ) | nm `notElem` supported = Just nm
  439         f _ = Nothing
  440 
  441 filterAttrName :: Ord a => [Name] -> [DomainAttribute a] -> [DomainAttribute a]
  442 filterAttrName keep = sort . filter f
  443     where
  444         f (DANameValue nm _) | nm `elem` keep = True
  445         f (DAName      nm  ) | nm `elem` keep = True
  446         f _ = False
  447 
  448 filterSizey :: Ord a => [DomainAttribute a] -> [DomainAttribute a]
  449 filterSizey = filterAttrName ["size", "minSize", "maxSize"]

  505 parseLiteral :: Parser Expression
  506 parseLiteral = label "value" (do p <- pCore ; p)
  507 
  508     where
  509 
  510         -- convert x to a constant if possible
  511         -- might save us from evaluating it again and again later
  512         mkAbstractLiteral x =
  513             case e2c (AbstractLiteral x) of
  514                 Nothing -> AbstractLiteral x
  515                 Just c  -> Constant c
  516 
  517         pCore :: Parser (Parser Expression)
  518         pCore = satisfyL $ \case
  519                 L_false       -> Just $ return $ Constant $ ConstantBool False
  520                 L_true        -> Just $ return $ Constant $ ConstantBool True
  521                 LIntLiteral i -> Just $ return $ Constant $ ConstantInt TagInt (fromInteger i)
  522                 L_OpenBracket -> Just pMatrix
  523                 L_tuple       -> Just pTupleWith
  524                 L_OpenParen   -> Just pTupleWithout
  525                 L_record      -> Just pRecord
  526                 L_variant     -> Just pVariant
  527                 L_OpenCurly   -> Just pSet
  528                 L_mset        -> Just pMSet
  529                 L_function    -> Just pFunction
  530                 L_sequence    -> Just pSequence
  531                 L_relation    -> Just pRelation
  532                 L_partition   -> Just pPartition
  533                 L_Minus       -> Just $ do
  534                     p <- pCore
  535                     res <- p
  536                     return (negate res)
  537                 _ -> Nothing
  538 
  539         pMatrix = mkAbstractLiteral <$> do
  540             -- lexeme L_OpenBracket
  541             xs <- commaSeparated0 parseExpr
  542             msum
  543                 [ do
  544                     let r = mkDomainIntB 1 (fromInt (genericLength xs))
  545                     lexeme L_CloseBracket
  546                     return (AbsLitMatrix r xs)
  547                 , do
  548                     lexeme L_SemiColon
  549                     r <- parseDomain
  550                     lexeme L_CloseBracket
  551                     return (AbsLitMatrix r xs)
  552                 ]
  553 
  554         pTupleWith = mkAbstractLiteral <$> do
  555             -- lexeme L_tuple
  556             xs <- parens $ commaSeparated0 parseExpr
  557             return (AbsLitTuple xs)
  558 
  559         pTupleWithout = mkAbstractLiteral <$> do
  560             -- xs <- parens $ countSepAtLeast 2 parseExpr comma
  561             xs <- countSepAtLeast 2 parseExpr comma
  562             lexeme L_CloseParen
  563             return (AbsLitTuple xs)
  564 
  565         pRecord = mkAbstractLiteral <$> do
  566             -- lexeme L_record
  567             let one = do n <- parseName
  568                          lexeme L_Eq
  569                          x <- parseExpr
  570                          return (n,x)
  571             xs <- braces $ commaSeparated0 one
  572             return $ AbsLitRecord xs
  573 
  574         pVariant = mkAbstractLiteral <$> do
  575             -- lexeme L_variant
  576             let one = do n <- parseName
  577                          lexeme L_Eq
  578                          x <- parseExpr
  579                          return (n,x)
  580             (n,x) <- braces one
  581             return $ AbsLitVariant Nothing n x
  582 
  583         pSet = mkAbstractLiteral <$> do
  584             -- xs <- braces (commaSeparated0 parseExpr)
  585             xs <- commaSeparated0 parseExpr
  586             lexeme L_CloseCurly
  587             return (AbsLitSet xs)
  588 
  589         pMSet = mkAbstractLiteral <$> do
  590             -- lexeme L_mset
  591             xs <- parens (commaSeparated0 parseExpr)
  592             return (AbsLitMSet xs)
  593 
  594         pFunction = mkAbstractLiteral <$> do
  595             -- lexeme L_function
  596             xs <- parens (commaSeparated0 inner)
  597             return (AbsLitFunction xs)
  598             where
  599                 inner = arrowedPair parseExpr
  600 
  601         pSequence = mkAbstractLiteral <$> do
  602             -- lexeme L_sequence
  603             xs <- parens (commaSeparated0 parseExpr)
  604             return (AbsLitSequence xs)
  605 
  606         pRelation = mkAbstractLiteral <$> do
  607             -- lexeme L_relation
  608             xs <- parens (commaSeparated0 (pTupleWith <|> pTupleWithout))
  609             xsFiltered <- forM xs $ \case
  610                 Constant (ConstantAbstract (AbsLitTuple is)) -> return (map Constant is)
  611                 AbstractLiteral (AbsLitTuple is) -> return is
  612                 x -> fail ("Cannot parse as part of relation literal:" <+> vcat [pretty x, pretty (show x)])
  613             return (AbsLitRelation xsFiltered)
  614 
  615         pPartition = mkAbstractLiteral <$> do
  616             -- lexeme L_partition
  617             xs <- parens (commaSeparated0 inner)
  618             return (AbsLitPartition xs)
  619             where
  620                 inner = braces (commaSeparated0 parseExpr)

  623 identifierText :: Parser T.Text
  624 identifierText = do

  626     return i
  627     where isIdentifier LIdentifier {} = True
  628           isIdentifier _ = False

  630 satisfyT :: (Lexeme -> Bool) -> Parser Lexeme
  631 satisfyT predicate = token nextPos testTok
  632     where
  633         testTok :: LexemePos -> Either [Message] Lexeme
  634         testTok (LexemePos tok _ _) =
  635             -- trace ("satisfyT: " ++ show pos ++ "\t" ++ show tok) $
  636             if predicate tok
  637                 then Right tok
  638                 else Left [Unexpected (showToken tok)]
  639 
  640         nextPos :: Int -> SourcePos -> LexemePos -> SourcePos
  641         nextPos _ _ (LexemePos _ _ pos) = pos
  642 
  643 satisfyL :: forall a . (Lexeme -> Maybe a) -> Parser a
  644 satisfyL predicate = token nextPos testTok
  645     where
  646         testTok :: LexemePos -> Either [Message] a
  647         testTok (LexemePos tok _ _) =
  648             -- trace ("satisfyL: " ++ show pos ++ "\t" ++ show tok) $
  649             case predicate tok of
  650                 Nothing  -> Left [Unexpected (showToken tok)]
  651                 Just res -> Right res
  652 
  653         nextPos :: Int -> SourcePos -> LexemePos -> SourcePos
  654         nextPos _ _ (LexemePos _ _ pos) = pos
  655 
  656 integer :: Parser Integer
  657 integer = do
  658     LIntLiteral i <- satisfyT isInt
  659     return i
  660     where isInt LIntLiteral {} = True
  661           isInt _ = False

  681 -- parses a specified number of elements separated by the given separator
  682 countSep :: Int -> Parser a -> Parser sep -> Parser [a]
  683 countSep 1 p _ = (:[]) <$> p
  684 countSep i p separator | i > 1 = (:) <$> (p <* separator) <*> countSep (i-1) p separator
  685 countSep _ _ _ = return []
  686 
  687 -- parses at least a given number of elements separated by the given separator
  688 countSepAtLeast :: Int -> Parser a -> Parser sep -> Parser [a]
  689 countSepAtLeast i p separator = (++) <$> countSep i p separator <*> many (separator *> p)
