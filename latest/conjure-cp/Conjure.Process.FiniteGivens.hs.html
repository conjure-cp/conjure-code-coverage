   20 -- | givens should have finite domains. except ints.
   21 --   this transformation introduces extra given ints to make them finite.
   22 --   the values for the extra givens will be computed during translate-solution
   23 finiteGivens ::
   24     NameGen m =>
   25     MonadLog m =>
   26     MonadUserError m =>
   27     EnumerateDomain m =>
   28     (?typeCheckerMode :: TypeCheckerMode) =>
   29     Model -> m Model
   30 finiteGivens m = flip evalStateT 1 $ do
   31     statements <- forM (mStatements m) $ \ st ->
   32         case st of
   33             Declaration (FindOrGiven Given name domain) -> do
   34                 (domain', extras, _) <- mkFinite domain
   35                 return $ [ Declaration $ FindOrGiven Given e (DomainInt TagInt []) | e <- extras ]
   36                       ++ [ Declaration $ FindOrGiven Given name domain'                   ]
   37             _ -> return [st]
   38     namegenst <- exportNameGenState
   39     return m { mStatements = concat statements
   40              , mInfo = (mInfo m) { miNameGenState = namegenst
   41                                  , miNbExtraGivens = maybe 0 (\ n -> n - 1 ) (lookup "fin" namegenst)
   42                                  }
   43              }

   46 finiteGivensParam ::
   47     NameGen m =>
   48     MonadLog m =>
   49     MonadUserError m =>
   50     EnumerateDomain m =>
   51     (?typeCheckerMode :: TypeCheckerMode) =>
   52     Model ->                                -- eprime
   53     Model ->                                -- essence-param
   54     [(Name, Constant)] ->                   -- some additional lettings
   55     m (Model, [Name])                       -- essence-param
   56 finiteGivensParam eprimeModel essenceParam additionalLettings = flip evalStateT 1 $ do
   57     let essenceGivenNames = eprimeModel |> mInfo |> miGivens
   58     let essenceGivens     = eprimeModel |> mInfo |> miOriginalDomains
   59     let essenceLettings   = extractLettings essenceParam
   60                          ++ eprimeModel |> mInfo |> miLettings
   61                          ++ [ (nm, Constant c) | (nm, c) <- additionalLettings ]
   62     let nbExtraGivens     = eprimeModel |> mInfo |> miNbExtraGivens
   63     let expectedExtras    = [ MachineName "fin" extraGiven []
   64                             | extraGiven <- [1..nbExtraGivens]
   65                             ]
   66     extras <- fmap concat $ forM essenceGivenNames $ \ name -> do
   67         logDebugVerbose $ "finiteGivensParam name" <+> pretty name
   68         case (lookup name essenceGivens, lookup name essenceLettings) of
   69             (Nothing, _) -> bug $ "Not found:" <+> pretty name
   70             (_, Nothing) -> return []
   71             (Just domain', Just expr) -> do
   72                 logDebugVerbose $ "finiteGivensParam domain' " <+> pretty domain'
   73                 domain  <- failToUserError $ fmap Constant <$> instantiateDomain essenceLettings domain'
   74                 logDebugVerbose $ "finiteGivensParam domain  " <+> pretty domain
   75                 logDebugVerbose $ "finiteGivensParam expr    " <+> pretty expr
   76                 constant <- failToUserError $ instantiateExpression essenceLettings expr
   77                 logDebugVerbose $ "finiteGivensParam constant" <+> pretty constant
   78                 (_, _, f) <- mkFinite domain
   79                 outs <- f constant
   80                 logDebugVerbose $ "finiteGivensParam outs    " <+> vcat (map pretty outs)
   81                 return outs
   82     logDebugVerbose $ "finiteGivensParam extras  " <+> vcat (map (pretty . show) extras)
   83     return
   84         ( essenceParam
   85             { mStatements = [ Declaration (Letting name (Constant value))
   86                             | name <- expectedExtras
   87                             -- we are storing the number of "extra givens" in the model info.
   88                             -- also, defaulting their values to 0 if they do not come out of
   89                             -- the usual finiteGivens process.
   90                             -- the idea is: if they don't come out from that,
   91                             -- they must be a part of an emply collection, hence 0.
   92                             , let value = fromMaybe 0 (lookup name extras)
   93                             ]
   94                          ++ mStatements essenceParam
   95             }
   96         , expectedExtras
   97         )

  100 -- | given a domain, add it additional attributes to make it _smaller_
  101 --   for example, this means adding a size attribute at the outer-most level
  102 --   and adding a maxSize attribute at the inner levels.
  103 mkFinite ::
  104     MonadState Int m =>
  105     NameGen m =>
  106     MonadLog m =>
  107     MonadUserError m =>
  108     EnumerateDomain m =>
  109     (?typeCheckerMode :: TypeCheckerMode) =>
  110     Domain () Expression ->
  111     m ( Domain () Expression                    -- "finite" domain
  112       , [Name]                                  -- extra givens
  113       , Constant -> m [(Name, Constant)]        -- value calculator for the extra givens
  114       )                                         -- input is a list of values for the domain
  115 mkFinite d@DomainTuple{}     = mkFiniteOutermost d
  116 mkFinite d@DomainRecord{}    = mkFiniteOutermost d
  117 mkFinite d@DomainVariant{}   = mkFiniteOutermost d
  118 mkFinite d@DomainMatrix{}    = mkFiniteOutermost d
  119 mkFinite d@DomainSet{}       = mkFiniteOutermost d
  120 mkFinite d@DomainMSet{}      = mkFiniteOutermost d
  121 mkFinite d@DomainSequence{}  = mkFiniteOutermost d
  122 mkFinite d@DomainFunction{}  = mkFiniteOutermost d
  123 mkFinite d@DomainRelation{}  = mkFiniteOutermost d
  124 mkFinite d@DomainPartition{} = mkFiniteOutermost d
  125 mkFinite d = return (d, [], const (return []))

  128 mkFiniteOutermost ::
  129     MonadState Int m =>
  130     NameGen m =>
  131     MonadLog m =>
  132     MonadUserError m =>
  133     EnumerateDomain m =>
  134     (?typeCheckerMode :: TypeCheckerMode) =>
  135     Domain () Expression ->
  136     m ( Domain () Expression
  137       , [Name]
  138       , Constant -> m [(Name, Constant)]
  139       )
  140 mkFiniteOutermost (DomainTuple inners) = do
  141     mids <- mapM mkFiniteInner inners
  142     return
  143         ( DomainTuple (map fst3 mids)
  144         , concatMap snd3 mids
  145         , \ constant -> do
  146                 logDebug $ "mkFiniteOutermost DomainTuple" <+> pretty constant
  147                 xs <- failToUserError $ viewConstantTuple constant
  148                 let innerFs = map thd3 mids
  149                 innerValues <- sequence [ innerF [x] | (innerF, x) <- zip innerFs xs ]
  150                 return (concat innerValues)
  151         )
  152 mkFiniteOutermost (DomainRecord (sortOn fst -> inners)) = do
  153     mids <- mapM (mkFiniteInner . snd) inners
  154     return
  155         ( DomainRecord (zip (map fst inners) (map fst3 mids))
  156         , concatMap snd3 mids
  157         , \ constant -> do
  158                 logDebug $ "mkFiniteOutermost DomainRecord" <+> pretty constant
  159                 xs' <- failToUserError $ viewConstantRecord constant
  160                 let
  161                     xs :: [Constant]
  162                     xs = map snd $ sortOn fst xs'
  163                 let innerFs = map thd3 mids
  164                 innerValues <- sequence [ innerF [x] | (innerF, x) <- zip innerFs xs ]
  165                 return (concat innerValues)
  166         )
  167 mkFiniteOutermost (DomainVariant inners) = do
  168     mids <- mapM (mkFiniteInner . snd) inners
  169     return
  170         ( DomainVariant (zip (map fst inners) (map fst3 mids))
  171         , concatMap snd3 mids
  172         , \ constant -> do
  173                 logDebug $ "mkFiniteOutermost DomainVariant" <+> pretty constant
  174                 xs' <- failToUserError $ viewConstantVariant constant
  175                 xs :: [Constant] <- sequence
  176                     [ case xs' of
  177                         (_, nm', c') | nm == nm' -> return c'
  178                         _ -> failToUserError $ instantiateDomain [] d >>= zeroVal
  179                     | (nm, d) <- inners ]
  180                 let innerFs = map thd3 mids
  181                 innerValues <- sequence [ innerF [x] | (innerF, x) <- zip innerFs xs ]
  182                 return (concat innerValues)
  183         )
  184 mkFiniteOutermost (DomainMatrix index inner) = do
  185     (inner', innerExtras, innerF) <- mkFiniteInner inner
  186     return
  187         ( DomainMatrix index inner'
  188         , innerExtras
  189         , \ constant -> do
  190                 logDebug $ "mkFiniteOutermost DomainMatrix" <+> pretty constant
  191                 (_, matr) <- failToUserError $ viewConstantMatrix constant
  192                 innerValues <- innerF matr
  193                 return innerValues
  194         )

  267 mkFiniteOutermost (DomainFunction () attr@(FunctionAttr SizeAttr_Size{} _ _) innerFr innerTo) = do
  268     (innerFr', innerFrExtras, innerFrF) <- mkFiniteInner innerFr
  269     (innerTo', innerToExtras, innerToF) <- mkFiniteInner innerTo
  270     return
  271         ( DomainFunction () attr innerFr' innerTo'
  272         , innerFrExtras ++ innerToExtras
  273         , \ constant -> do
  274                 logDebug $ "mkFiniteOutermost DomainFunction" <+-> pretty constant
  275                 function <- failToUserError $ viewConstantFunction constant
  276                 innerFrValues <- innerFrF (map fst function)
  277                 innerToValues <- innerToF (map snd function)
  278                 return $ innerFrValues ++ innerToValues
  279         )
  280 mkFiniteOutermost (DomainFunction () (FunctionAttr _ partialityAttr jectivityAttr) innerFr innerTo) = do
  281     s <- nextName "fin"
  282     (innerFr', innerFrExtras, innerFrF) <- mkFiniteInner innerFr
  283     (innerTo', innerToExtras, innerToF) <- mkFiniteInner innerTo
  284     return
  285         ( DomainFunction ()
  286                 (FunctionAttr (SizeAttr_Size (fromName s)) partialityAttr jectivityAttr)
  287                 innerFr' innerTo'
  288         , s : innerFrExtras ++ innerToExtras
  289         , \ constant -> do
  290                 logDebug $ "mkFiniteOutermost DomainFunction" <+-> pretty constant
  291                 function <- failToUserError $ viewConstantFunction constant
  292                 let functionSize = genericLength function
  293                 innerFrValues <- innerFrF (map fst function)
  294                 innerToValues <- innerToF (map snd function)
  295                 return $ innerFrValues ++ innerToValues ++ [(s, ConstantInt TagInt functionSize)]
  296         )
  297 mkFiniteOutermost (DomainRelation () attr@(RelationAttr SizeAttr_Size{} _) inners) = do
  298     (inners', innersExtras, innersF) <- unzip3 <$> mapM mkFiniteInner inners
  299     return
  300         ( DomainRelation () attr inners'
  301         , concat innersExtras
  302         , \ constant -> do
  303                 logDebug $ "mkFiniteOutermost DomainRelation" <+> pretty constant
  304                 relation <- failToUserError $ viewConstantRelation constant
  305                 innersValues <- zipWithM ($) innersF (transpose relation)
  306                 return (concat innersValues)
  307         )
  308 mkFiniteOutermost (DomainRelation () (RelationAttr _ binRelAttr) inners) = do
  309     s <- nextName "fin"
  310     (inners', innersExtras, innersF) <- unzip3 <$> mapM mkFiniteInner inners
  311     return
  312         ( DomainRelation ()
  313                 (RelationAttr (SizeAttr_Size (fromName s)) binRelAttr)
  314                 inners'
  315         , s : concat innersExtras
  316         , \ constant -> do
  317                 logDebug $ "mkFiniteOutermost DomainRelation" <+> pretty constant
  318                 relation <- failToUserError $ viewConstantRelation constant
  319                 let relationSize = genericLength relation
  320                 innersValues <- zipWithM ($) innersF (transpose relation)
  321                 return $ concat innersValues ++ [(s, ConstantInt TagInt relationSize)]
  322         )
  323 mkFiniteOutermost (DomainPartition () attr@(PartitionAttr SizeAttr_Size{} SizeAttr_Size{} _) inner) = do
  324     (inner', innerExtras, innerF) <- mkFiniteInner inner
  325     return
  326         ( DomainPartition () attr inner'
  327         , innerExtras
  328         , \ constant -> do
  329                 logDebug $ "mkFiniteOutermost DomainPartition" <+> pretty constant
  330                 parts <- failToUserError $ viewConstantPartition constant
  331                 innerValues <- mapM innerF parts
  332                 return (concat innerValues)

  334 mkFiniteOutermost (DomainPartition () (PartitionAttr _ _ isRegularAttr) inner) = do
  335     numPartsFin  <- nextName "fin"
  336     partsSizeFin <- nextName "fin"
  337     (inner', innerExtras, innerF) <- mkFiniteInner inner
  338     return
  339         ( DomainPartition ()
  340             (PartitionAttr (SizeAttr_Size (fromName numPartsFin))
  341                            (SizeAttr_MaxSize (fromName partsSizeFin))
  342                            isRegularAttr)
  343             inner'
  344         , numPartsFin:partsSizeFin:innerExtras
  345         , \ constant -> do
  346                 logDebug $ "mkFiniteOutermost DomainPartition" <+> pretty constant
  347                 parts <- failToUserError $ viewConstantPartition constant
  348                 let numPartsVal = genericLength parts
  349                 let partsSizeVal = maximum0 $ map genericLength parts
  350                 innerValues <- mapM innerF parts
  351                 return $ concat innerValues ++ [ (numPartsFin, ConstantInt TagInt numPartsVal)
  352                                                , (partsSizeFin, ConstantInt TagInt partsSizeVal)
  353                                                ]
  354         )

  358 mkFiniteInner ::
  359     MonadState Int m =>
  360     NameGen m =>
  361     MonadLog m =>
  362     MonadUserError m =>
  363     EnumerateDomain m =>
  364     (?typeCheckerMode :: TypeCheckerMode) =>
  365     Domain () Expression ->
  366     m ( Domain () Expression
  367       , [Name]
  368       , [Constant] -> m [(Name, Constant)]
  369       )
  370 mkFiniteInner (DomainInt t []) = do
  371     fr <- nextName "fin"
  372     to <- nextName "fin"
  373     return
  374         ( DomainInt t [RangeBounded (fromName fr) (fromName to)]
  375         , [fr, to]
  376         , \ constants -> do
  377                 logDebug $ "mkFiniteInner DomainInt 1" <+-> fsep (map pretty constants)
  378                 ints <- failToUserError $ mapM viewConstantInt constants
  379                 return [ (fr, ConstantInt t (minimum0 ints))
  380                        , (to, ConstantInt t (maximum0 ints))
  381                        ]
  382         )
  383 mkFiniteInner (DomainInt t [RangeLowerBounded low]) = do
  384     new <- nextName "fin"
  385     return
  386         ( DomainInt t [RangeBounded low (fromName new)]
  387         , [new]
  388         , \ constants -> do
  389                 logDebug $ "mkFiniteInner DomainInt 2" <+-> fsep (map pretty constants)
  390                 ints <- failToUserError $ mapM viewConstantInt constants
  391                 return [ (new, ConstantInt t (maximum0 ints)) ]
  392         )
  393 mkFiniteInner (DomainInt t [RangeUpperBounded upp]) = do
  394     new <- nextName "fin"
  395     return
  396         ( DomainInt t [RangeBounded (fromName new) upp]
  397         , [new]

  540                 return $ innerFrValues ++ innerToValues
  541         )
  542 mkFiniteInner (DomainFunction () (FunctionAttr _ partialityAttr jectivityAttr) innerFr innerTo) = do
  543     s <- nextName "fin"
  544     (innerFr', innerFrExtras, innerFrF) <- mkFiniteInner innerFr
  545     (innerTo', innerToExtras, innerToF) <- mkFiniteInner innerTo
  546     return
  547         ( DomainFunction ()
  548                 (FunctionAttr (SizeAttr_MaxSize (fromName s)) partialityAttr jectivityAttr)
  549                 innerFr' innerTo'
  550         , s : innerFrExtras ++ innerToExtras
  551         , \ constants -> do
  552                 logDebug $ "mkFiniteInner DomainFunction" <+-> vcat (map pretty constants)
  553                 functions <- failToUserError $ mapM viewConstantFunction constants
  554                 let functionMaxSize = maximum0 $ map genericLength functions
  555                 innerFrValues <- innerFrF (map fst (concat functions))
  556                 innerToValues <- innerToF (map snd (concat functions))
  557                 return $ innerFrValues ++ innerToValues ++ [(s, ConstantInt TagInt functionMaxSize)]
  558         )
  559 mkFiniteInner (DomainRelation () attr@(RelationAttr SizeAttr_Size{} _) inners) = do
  560     (inners', innersExtras, innersF) <- unzip3 <$> mapM mkFiniteInner inners
  561     return
  562         ( DomainRelation () attr inners'
  563         , concat innersExtras
  564         , \ constants -> do
  565                 logDebug $ "mkFiniteInner DomainRelation" <+-> vcat (map pretty constants)
  566                 relations <- failToUserError $ mapM viewConstantRelation constants
  567                 innersValues <- zipWithM ($) innersF (transpose $ concat relations)
  568                 return $ concat innersValues
  569         )
  570 mkFiniteInner (DomainRelation () (RelationAttr _ binRelAttr) inners) = do
  571     s <- nextName "fin"
  572     (inners', innersExtras, innersF) <- unzip3 <$> mapM mkFiniteInner inners
  573     return
  574         ( DomainRelation ()
  575                 (RelationAttr (SizeAttr_MaxSize (fromName s)) binRelAttr)
  576                 inners'
  577         , s : concat innersExtras
  578         , \ constants -> do
  579                 logDebug $ "mkFiniteInner DomainRelation" <+> vcat (map pretty constants)
  580                 relations <- failToUserError $ mapM viewConstantRelation constants
  581                 let relationMaxSize = maximum0 $ map genericLength relations
  582                 innersValues <- zipWithM ($) innersF (transpose $ concat relations)
  583                 return $ concat innersValues ++ [(s, ConstantInt TagInt relationMaxSize)]
  584         )
  585 mkFiniteInner (DomainPartition () attr@(PartitionAttr SizeAttr_Size{} SizeAttr_Size{} _) inner) = do
  586     (inner', innerExtras, innerF) <- mkFiniteInner inner
  587     return
  588         ( DomainPartition () attr inner'
  589         , innerExtras
  590         , \ constants -> do
  591                 logDebug $ "mkFiniteInner DomainPartition" <+> vcat (map pretty constants)
  592                 parts <- failToUserError $ mapM viewConstantPartition constants
  593                 innersValues <- mapM innerF (concat parts)
  594                 return $ concat innersValues
  595         )
  596 mkFiniteInner (DomainPartition () (PartitionAttr _ _ isRegularAttr) inner) = do
  597     numPartsFin  <- nextName "fin"
  598     partsSizeFin <- nextName "fin"
  599     (inner', innerExtras, innerF) <- mkFiniteInner inner
  600     return
  601         ( DomainPartition ()
  602             (PartitionAttr (SizeAttr_MaxSize (fromName numPartsFin))
  603                            (SizeAttr_MaxSize (fromName partsSizeFin))
  604                            isRegularAttr)
  605             inner'
  606         , numPartsFin:partsSizeFin:innerExtras
  607         , \ constants -> do
  608                 logDebug $ "mkFiniteInner DomainPartition" <+> vcat (map pretty constants)
  609                 parts <- failToUserError $ mapM viewConstantPartition constants
  610                 let numPartsVal = maximum0 $ map genericLength parts
  611                 let partsSizeVal = maximum0 $ map genericLength parts
  612                 innerValues <- mapM innerF (concat parts)
  613                 return $ concat innerValues ++ [ (numPartsFin, ConstantInt TagInt numPartsVal)
  614                                                , (partsSizeFin, ConstantInt TagInt partsSizeVal)
  615                                                ]
  616         )
  617 mkFiniteInner d = return (d, [], const (return []))
