   23 instance (TypeOf x, Pretty x, Domain () x :< x) => TypeOf (OpRestrict x) where
   24     typeOf p@(OpRestrict f domX) = do
   25         dom :: Domain () x   <- project domX
   26         TypeFunction from to <- typeOf f
   27         from'                <- typeOfDomain dom
   28         if typesUnify [from, from']
   29             then return (TypeFunction (mostDefined [from', from]) to)
   30             else raiseTypeError p
