   25 instance (TypeOf x, Pretty x, ExpressionLike x) => TypeOf (OpAnd x) where
   26     typeOf p@(OpAnd x) = do
   27         ty <- typeOf x
   28         case ty of
   29             TypeList TypeAny -> return TypeBool
   30             TypeList TypeBool -> return TypeBool
   31             TypeMatrix _ TypeAny -> return TypeBool
   32             TypeMatrix _ TypeBool -> return TypeBool
   33             TypeSet TypeBool -> return TypeBool
   34             TypeMSet TypeBool -> return TypeBool
   35             TypeAny -> return TypeBool
   36             _ -> raiseTypeError $ vcat [ pretty p
   37                                        , "The argument has type:" <+> pretty ty
   38                                        ]

   40 instance (OpAnd x :< x) => SimplifyOp OpAnd x where
   41     simplifyOp (OpAnd x)
   42         | Just xs <- listOut x
   43         , let filtered = filter (/= fromBool False) xs
   44         , length filtered /= length xs      -- there were false's
   45         = return $ fromBool False
   46     simplifyOp (OpAnd x)
   47         | Just xs <- listOut x
   48         , let filtered = filter (/= fromBool True) xs
   49         , length filtered /= length xs      -- there were true's
   50         = return $ inject $ OpAnd $ fromList filtered
   51     simplifyOp _ = na "simplifyOp{OpAnd}"
