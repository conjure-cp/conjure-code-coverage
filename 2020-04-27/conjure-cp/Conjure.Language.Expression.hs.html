   52 data Statement
   53     = Declaration Declaration
   54     | SearchOrder [SearchOrder]
   55     | SearchHeuristic Name
   56     | Where [Expression]
   57     | Objective Objective Expression
   58     | SuchThat [Expression]
   59     deriving (Eq, Ord, Show, Data, Typeable, Generic)
   60 
   61 instance Serialize Statement
   62 instance Hashable  Statement
   63 instance ToJSON    Statement where toJSON = genericToJSON jsonOptions
   64 instance FromJSON  Statement where parseJSON = genericParseJSON jsonOptions
   65 
   66 instance SimpleJSON Statement where
   67     toSimpleJSON st =
   68         case st of
   69             Declaration d -> toSimpleJSON d
   70             _ -> noToSimpleJSON st
   71     fromSimpleJSON _ = noFromSimpleJSON
   72 
   73 instance Pretty Statement where
   74     pretty (Declaration x) = pretty x
   75     pretty (SearchOrder nms) = "branching on" <++> prettyList prBrackets "," nms
   76     pretty (SearchHeuristic nm) = "heuristic" <+> pretty nm
   77     pretty (Where xs) = "where" <++> vcat (punctuate "," $ map pretty xs)
   78     pretty (Objective obj x) = pretty obj <++> pretty x
   79     pretty (SuchThat xs) = "such that" <++> vcat (punctuate "," $ map pretty xs)
   80 
   81 instance VarSymBreakingDescription Statement where
   82     varSymBreakingDescription (Declaration x) = JSON.Object $ M.fromList
   83         [ ("type", JSON.String "Declaration")
   84         , ("children", varSymBreakingDescription x)
   85         ]
   86     varSymBreakingDescription SearchOrder{} = JSON.Null
   87     varSymBreakingDescription SearchHeuristic{} = JSON.Null
   88     varSymBreakingDescription (Where xs) = JSON.Object $ M.fromList
   89         [ ("type", JSON.String "Where")
   90         , ("symmetricChildren", JSON.Bool True)
   91         , ("children", JSON.Array $ V.fromList $ map varSymBreakingDescription xs)
   92         ]
   93     varSymBreakingDescription (Objective obj x) = JSON.Object $ M.fromList
   94         [ ("type", JSON.String $ "Objective-" `mappend` stringToText (show obj))
   95         , ("children", varSymBreakingDescription x)
   96         ]
   97     varSymBreakingDescription (SuchThat xs) = JSON.Object $ M.fromList
   98         [ ("type", JSON.String "SuchThat")
   99         , ("symmetricChildren", JSON.Bool True)
  100         , ("children", JSON.Array $ V.fromList $ map varSymBreakingDescription xs)
  101         ]

  167 instance Pretty Declaration where
  168     pretty (FindOrGiven forg nm d) = hang (pretty forg <+> pretty nm <>  ":" ) 8 (pretty d)
  169     pretty (Letting nm (Domain x)) = hang ("letting" <+> pretty nm <+> "be domain") 8 (pretty x)
  170     pretty (Letting nm x) =
  171         let
  172             extract :: Constant -> Maybe [Constant]
  173             extract (viewConstantMatrix   -> Just (_, rows)) = Just rows
  174             extract (viewConstantTuple    -> Just rows     ) = Just rows
  175             extract (viewConstantSet      -> Just rows     ) = Just rows
  176             extract (viewConstantMSet     -> Just rows     ) = Just rows
  177             extract (viewConstantFunction -> Just rows     ) = Just (map snd rows)
  178             extract (viewConstantSequence -> Just rows     ) = Just rows
  179             extract _ = Nothing
  180 
  181             isPrim :: Constant -> Maybe Prim
  182             isPrim (ConstantBool val) = Just (Left val)
  183             isPrim (ConstantInt _ val) = Just (Right (Left val))
  184             isPrim val@ConstantEnum{} = Just (Right (Right val))
  185             isPrim _ = Nothing
  186 
  187             isPrim1DT :: Constant -> Maybe [Prim]
  188             -- isPrim1DT p@(viewConstantMatrix   -> Just{}) = isPrim1D p
  189             -- isPrim1DT p@(viewConstantTuple    -> Just{}) = isPrim1D p
  190             -- isPrim1DT p@(viewConstantSet      -> Just{}) = isPrim1D p
  191             -- isPrim1DT p@(viewConstantMSet     -> Just{}) = isPrim1D p
  192             -- isPrim1DT p@(viewConstantFunction -> Just{}) = isPrim1D p
  193             -- isPrim1DT p@(viewConstantSequence -> Just{}) = isPrim1D p
  194             isPrim1DT _ = Nothing
  195 
  196             isPrim1D :: Constant -> Maybe [Prim]
  197             isPrim1D (extract -> Just cells) = mapM isPrim cells
  198             isPrim1D _ = Nothing
  199 
  200             isPrim2D :: Constant -> Maybe [[Prim]]
  201             isPrim2D (extract -> Just rows) = mapM isPrim1D rows
  202             isPrim2D (viewConstantRelation  -> Just table) = mapM (mapM isPrim) table
  203             isPrim2D (viewConstantPartition -> Just table) = mapM (mapM isPrim) table
  204             isPrim2D _ = Nothing
  205 
  206             isPrim3D :: Constant -> Maybe [[[Prim]]]
  207             isPrim3D (extract -> Just table) = mapM isPrim2D table
  208             isPrim3D _ = Nothing
  209 
  210             showPrim :: Int -> Prim -> String
  211             showPrim _ (Left True)  = "T"
  212             showPrim _ (Left False) = "_"
  213             showPrim n (Right (Left  i)) = padLeft n ' ' (show i)
  214             showPrim n (Right (Right i)) = padRight n ' ' (show (pretty i))
  215 
  216             maxIntWidth :: Data a => a -> Int
  217             maxIntWidth primTable =
  218                 maximum (0 : [ length (show i)          | i <- universeBi primTable :: [Integer] ]
  219                           ++ [ length (show (pretty i)) | i@ConstantEnum{} <- universeBi primTable ])
  220 
  221             comment1D :: Int ->  [Prim] -> String
  222             comment1D width primTable =
  223                 unlines
  224                     [ "$ Visualisation for " ++ show (pretty nm)
  225                     , "$ " ++ unwords [ showPrim width cell | cell <- primTable ]
  226                     ]
  227 
  228             comment2D :: Int ->  [[Prim]] -> String
  229             comment2D width primTable =
  230                 unlines
  231                     $ ( "$ Visualisation for " ++ show (pretty nm))
  232                     : [ "$ " ++ unwords [ showPrim width cell | cell <- row ]
  233                       | row <- primTable ]
  234 
  235             comment3D :: Int -> [[[Prim]]] -> String
  236             comment3D width primTable =
  237                 unlines
  238                     $ ( "$ Visualisation for " ++ show (pretty nm))
  239                     : concat [ [ "$ " ++ unwords [ showPrim width cell | cell <- row ]
  240                                   | row <- table
  241                                ] ++ ["$ "]
  242                              | table <- primTable ]
  243 
  244             modifierX =
  245                 case x of
  246                     Constant c -> modifierC c
  247                     _          -> id
  248 
  249             modifierC c
  250                 | Just primTable <- isPrim1DT c
  251                 , not (null primTable) = \ s ->
  252                     vcat [s, pretty (comment1D (maxIntWidth primTable) primTable)]
  253             modifierC c
  254                 | Just primTable <- isPrim2D c
  255                 , not (null (concat primTable)) = \ s ->
  256                     vcat [s, pretty (comment2D (maxIntWidth primTable) primTable)]
  257             modifierC c
  258                 | Just primTable <- isPrim3D c
  259                 , not (null (concat (concat primTable))) = \ s ->
  260                     vcat [s, pretty (comment3D (maxIntWidth primTable) primTable)]
  261             modifierC _ = id
  262 
  263         in
  264             modifierX $ hang ("letting" <+> pretty nm <+> "be") 8 (pretty x)
  265     pretty (GivenDomainDefnEnum name) =
  266         hang ("given"   <+> pretty name) 8 "new type enum"
  267     pretty (LettingDomainDefnEnum name values) =
  268         hang ("letting" <+> pretty name <+> "be new type enum") 8
  269              (prettyList prBraces "," values)
  270     pretty (LettingDomainDefnUnnamed name size) =
  271         hang ("letting" <+> pretty name <+> "be new type of size") 8 (pretty size)

  323 data Expression
  324     = Constant Constant
  325     | AbstractLiteral (AbstractLiteral Expression)
  326     | Domain (Domain () Expression)
  327     | Reference Name (Maybe ReferenceTo)
  328     | WithLocals Expression InBubble
  329     | Comprehension Expression [GeneratorOrCondition]
  330     | Typed Expression Type
  331     | Op (Op Expression)
  332     | ExpressionMetaVar String
  333     deriving (Eq, Ord, Show, Data, Typeable, Generic)
  334 
  335 instance Serialize Expression
  336 instance Hashable  Expression
  337 instance ToJSON    Expression where toJSON = genericToJSON jsonOptions
  338 instance FromJSON  Expression where parseJSON = genericParseJSON jsonOptions
  339 
  340 instance SimpleJSON Expression where
  341     toSimpleJSON x =
  342         case x of
  343             Constant c -> toSimpleJSON c
  344             AbstractLiteral lit -> toSimpleJSON lit
  345             Typed y _ -> toSimpleJSON y
  346             Op (MkOpMinus (OpMinus a b)) -> do
  347                 a' <- toSimpleJSON a
  348                 b' <- toSimpleJSON b
  349                 case (a', b') of
  350                     (JSON.Number a'', JSON.Number b'') -> return (JSON.Number (a'' - b''))
  351                     _ -> noToSimpleJSON x
  352             _ -> noToSimpleJSON x
  353     fromSimpleJSON _ = noFromSimpleJSON

  356 viewIndexed :: Expression -> (Expression, [Doc])
  357 viewIndexed (Op (MkOpIndexing (OpIndexing m i  ))) =
  358     let this = pretty i
  359     in  second (++ [this]) (viewIndexed m)
  360 viewIndexed (Op (MkOpSlicing  (OpSlicing  m a b))) =
  361     let this = maybe prEmpty pretty a <> ".." <> maybe prEmpty pretty b
  362     in  second (++ [this]) (viewIndexed m)
  363 viewIndexed m = (m, [])

  424 instance TypeOf Expression where
  425     typeOf (Constant x) = typeOf x
  426     typeOf (AbstractLiteral x) = typeOf x
  427     typeOf (Domain x) = fail ("Expected an expression, but got a domain:" <++> pretty x)
  428     typeOf (Reference nm Nothing) = fail ("Type error, identifier not bound:" <+> pretty nm)
  429     typeOf (Reference nm (Just refTo)) =
  430         case refTo of
  431             Alias x -> typeOf x
  432             InComprehension gen ->
  433                 let
  434                     lu pat ty = maybe
  435                         (bug $ vcat ["Type error, InComprehension:", pretty nm, pretty pat, pretty ty])
  436                         return
  437                         (lu' pat ty)
  438                     lu' (Single nm') ty | nm == nm' = Just ty
  439                     lu' (AbsPatTuple  pats) (TypeTuple    tys) = zipWith lu' pats tys   |> catMaybes |> listToMaybe
  440                     lu' (AbsPatMatrix pats) (TypeMatrix _ ty ) = [lu' p ty | p <- pats] |> catMaybes |> listToMaybe
  441                     lu' (AbsPatSet    pats) (TypeSet      ty ) = [lu' p ty | p <- pats] |> catMaybes |> listToMaybe
  442                     lu' _ _ = Nothing
  443                 in
  444                     case gen of
  445                         GenDomainNoRepr  pat domain -> typeOfDomain domain >>= lu pat
  446                         GenDomainHasRepr pat domain -> typeOfDomain domain >>= lu (Single pat)
  447                         GenInExpr        pat expr   ->
  448                             case project expr of
  449                                 Just (d :: Domain () Expression) -> fail $ vcat
  450                                     [ "Expected an expression, but got a domain:" <++> pretty d
  451                                     , "In the generator of a comprehension or a quantified expression"
  452                                     , "Consider using" <+> pretty pat <+> ":" <+> pretty expr
  453                                     ]
  454                                 Nothing -> do
  455                                     tyExpr <- typeOf expr
  456                                     case innerTypeOf tyExpr of
  457                                         Just tyExprInner -> lu pat tyExprInner
  458                                         Nothing -> fail $ vcat
  459                                             [ "Type error in the generator of a comprehension or a quantified expression"
  460                                             , "Consider using" <+> pretty pat <+> ":" <+> pretty expr
  461                                             ]
  462             DeclNoRepr  _ _ dom _ -> typeOfDomain dom
  463             DeclHasRepr _ _ dom   -> typeOfDomain dom
  464             RecordField _ ty      -> return ty
  465             VariantField _ ty     -> return ty
  466     typeOf p@(WithLocals h (DefinednessConstraints cs)) = do
  467         forM_ cs $ \ c -> do
  468             ty <- typeOf c
  469             unless (typeUnify TypeBool ty) $ fail $ vcat
  470                     [ "Local constraint is not boolean."
  471                     , "Condition:" <+> pretty c
  472                     , "In:" <+> pretty p
  473                     ]
  474         typeOf h
  475     typeOf p@(WithLocals h (AuxiliaryVars stmts)) = do
  476         forM_ stmts $ \ stmt ->
  477             case stmt of
  478                 Declaration{} -> return ()                  -- TODO: what other checks make sense?
  479                 SuchThat xs -> forM_ xs $ \ x -> do
  480                     ty <- typeOf x
  481                     case ty of
  482                         TypeBool{} -> return ()
  483                         _ -> fail $ vcat
  484                             [ "Inside a bubble, in a 'such that' statement:" <++> pretty x
  485                             , "Expected type `bool`, but got:" <++> pretty ty
  486                             ]
  487                 _ -> fail $ vcat
  488                     [ "Unexpected statement inside a bubble."
  489                     , "Expected type `find` or `such that`, but got:" <++> pretty stmt
  490                     , "The complete expression:" <+> pretty p
  491                     ]
  492         typeOf h
  493     typeOf p@(Comprehension x gensOrConds) = do
  494         forM_ gensOrConds $ \ goc -> case goc of
  495             Generator{} -> return ()                    -- TODO: do this properly
  496             Condition c -> do
  497                 ty <- typeOf c
  498                 unless (typeUnify TypeBool ty) $ fail $ vcat
  499                     [ "Condition is not boolean."
  500                     , "Condition:" <+> pretty c
  501                     , "In:" <+> pretty p
  502                     ]
  503             ComprehensionLetting{} -> return ()
  504         TypeList <$> typeOf x
  505     typeOf (Typed _ ty) = return ty
  506     typeOf (Op op) = typeOf op
  507     typeOf x@ExpressionMetaVar{} = bug ("typeOf:" <+> pretty x)

  509 instance RepresentationOf Expression where
  510     representationTreeOf (Reference _ (Just (DeclHasRepr _ _ dom))) = return (reprTree dom)
  511     representationTreeOf (Op (MkOpIndexing (OpIndexing m i))) = do
  512         iType <- typeOf i
  513         case iType of
  514             TypeBool{} -> return ()
  515             TypeInt{} -> return ()
  516             _ -> fail "representationOf, OpIndexing, not a bool or int index"
  517         mTree <- representationTreeOf m
  518         case mTree of
  519             Tree _ [r] -> return r
  520             _ -> fail "domainOf, OpIndexing, not a matrix"
  521     representationTreeOf _ = fail "doesn't seem to have a representation"

  566 instance Num Expression where
  567     x + y = Op $ MkOpSum     $ OpSum     $ fromList [x,y]
  568     x - y = Op $ MkOpMinus   $ OpMinus x y
  569     x * y = Op $ MkOpProduct $ OpProduct $ fromList [x,y]
  570     abs x = Op $ MkOpTwoBars $ OpTwoBars x
  571     signum _ = bug "signum {Expression}"
  572     fromInteger = fromInt . fromInteger
  573 
  574 instance Integral Expression where
  575     divMod a b = ( Op $ MkOpDiv $ OpDiv a b
  576                  , Op $ MkOpMod $ OpMod a b )
  577     quotRem = divMod
  578     toInteger = bug "toInteger {Expression}"
  579 
  580 instance Real Expression where
  581     toRational = bug "toRational {Expression}"
  582 
  583 instance Enum Expression where
  584     fromEnum = bug "fromEnum {Expression}"
  585     toEnum = fromInt . fromIntegral
  586     succ a = a + 1
  587     pred a = a - 1
  588     enumFrom x = x : enumFrom (succ x)
  589     enumFromThen x n = x : enumFromThen (x+n) n
  590     enumFromTo _x _y = bug "enumFromTo {Expression}"
  591     enumFromThenTo _x _n _y = bug "enumFromThenTo {Expression}"

  614 -- | This is only for when you know the Expression you have is actually a Constant, but
  615 --   is refusing to believe that it is one.
  616 --   Remind it where it comes from!
  617 --   (Srsly: Can be useful after parsing a solution file, for example.)
  618 e2c :: MonadFail m => Expression -> m Constant
  619 e2c (Constant c) = return c
  620 e2c (AbstractLiteral c) = ConstantAbstract <$> mapM e2c c
  621 e2c (Op (MkOpNegate (OpNegate (Constant (ConstantInt t x))))) = return $ ConstantInt t $ negate x
  622 e2c x = fail ("e2c, not a constant:" <+> pretty x)

  624 -- | generate a fresh name for a quantified variable.
  625 --   fst: the pattern to be used inside a generator
  626 --   snd: the expression to be used everywhere else
  627 quantifiedVar :: NameGen m => m (AbstractPattern, Expression)
  628 quantifiedVar = do
  629     nm <- nextName "q"
  630     let pat = Single nm
  631         ref = Reference nm Nothing
  632     return (pat, ref)

  634 -- | like `quantifiedVar`, but already name-resolved as a quantified variable over the given domain
  635 quantifiedVarOverDomain :: NameGen m => Domain () Expression -> m (AbstractPattern, Expression)
  636 quantifiedVarOverDomain domain = do
  637     nm <- nextName "q"
  638     let pat = Single nm
  639         ref = Reference nm (Just (InComprehension (GenDomainNoRepr (Single nm) domain)))
  640     return (pat, ref)

  642 -- | generate a fresh name for an auxiliary variable.
  643 --   fst: the name to be used when declaring the variable
  644 --   snd: the expression to be used everywhere else
  645 auxiliaryVar :: NameGen m => m (Name, Expression)
  646 auxiliaryVar = do
  647     -- Savile Row has a bug which is triggered when there are variables with names of the form aux*
  648     nm <- nextName "conjure_aux"
  649     let ref = Reference nm Nothing
  650     return (nm, ref)

  653 -- | pattern, template, argument, result
  654 lambdaToFunction :: AbstractPattern -> Expression -> Expression -> Expression
  655 lambdaToFunction (Single nm) body = \ p ->
  656     let
  657         replacer :: Expression -> Expression
  658         replacer (Reference n _) | n == nm = p
  659         replacer x = x
  660     in
  661         transform replacer body
  662 lambdaToFunction (AbsPatTuple ts) body = \ p ->
  663     let
  664         unroll :: [AbstractPattern] -> [Expression] -> Expression -> Expression
  665         unroll [] [] b = b
  666         unroll (pat:pats) (val:vals) b = unroll pats vals (lambdaToFunction pat b val)
  667         unroll _ _ _ = bug "lambdaToFunction, AbsPatTuple, unroll"
  668 
  669         ps :: [Expression]
  670         ps = case p of
  671             Constant (ConstantAbstract (AbsLitTuple xs)) -> map Constant xs
  672             AbstractLiteral (AbsLitTuple xs) -> xs
  673             _ -> [ Op (MkOpIndexing (OpIndexing p i))
  674                  | i' <- [ 1 .. genericLength ts ]
  675                  , let i = fromInt i'
  676                  ]
  677     in
  678         unroll ts ps body
  679 lambdaToFunction (AbsPatMatrix ts) body = \ p ->
  680     let
  681         unroll :: [AbstractPattern] -> [Expression] -> Expression -> Expression
  682         unroll [] [] b = b
  683         unroll (pat:pats) (val:vals) b = unroll pats vals (lambdaToFunction pat b val)
  684         unroll _ _ _ = bug "lambdaToFunction, AbsPatMatrix, unroll"
  685 
  686         ps :: [Expression]
  687         ps = case p of
  688             Constant (ConstantAbstract (AbsLitMatrix _ xs)) -> map Constant xs
  689             AbstractLiteral (AbsLitMatrix _ xs) -> xs
  690             _ -> bug $ "lambdaToFunction, AbsPatMatrix" <++> vcat [pretty p, pretty (show p)]
  691     in
  692         unroll ts ps body
  693 lambdaToFunction (AbsPatSet ts) body = \ p ->
  694     let
  695         unroll :: [AbstractPattern] -> [Expression] -> Expression -> Expression
  696         unroll [] [] b = b
  697         unroll (pat:pats) (val:vals) b = unroll pats vals (lambdaToFunction pat b val)
  698         unroll _ _ _ = bug "lambdaToFunction, AbsPatSet, unroll"
  699 
  700         ps :: [Expression]
  701         ps = case p of
  702             Constant (ConstantAbstract (AbsLitSet xs)) -> map Constant xs
  703             AbstractLiteral (AbsLitSet xs) -> xs
  704             _ -> bug $ "lambdaToFunction, AbsPatSet" <++> vcat [pretty p, pretty (show p)]
  705     in
  706         unroll ts ps body
  707 lambdaToFunction p@AbstractPatternMetaVar{} _ = bug $ "Unsupported AbstractPattern, got" <+> pretty (show p)

  714 data ReferenceTo
  715     = Alias           Expression
  716     | InComprehension Generator
  717     | DeclNoRepr      FindOrGiven Name (Domain () Expression)
  718                       Region -- the region of this reference
  719                              -- references with the same region identifier will get the same representation
  720     | DeclHasRepr     FindOrGiven Name (Domain HasRepresentation Expression)
  721     | RecordField     Name Type         -- the type of the field with this name
  722     | VariantField    Name Type         -- the type of the variant with this name
  723     deriving (Eq, Ord, Show, Data, Typeable, Generic)

  728 instance FromJSON  ReferenceTo where parseJSON = genericParseJSON jsonOptions

  730 instance Pretty ReferenceTo where
  731     pretty (Alias x) = "Alias" <+> prParens (pretty x)
  732     pretty (InComprehension gen) = "InComprehension" <+> prParens (pretty gen)
  733     pretty (DeclNoRepr  forg nm dom _) = "DeclNoRepr" <+> prParens (pretty forg <+> pretty nm <> ":" <+> pretty dom)
  734     pretty (DeclHasRepr forg nm dom  ) = "DeclHasRepr" <+> prParens (pretty forg <+> pretty nm <> ":" <+> pretty dom)
  735     pretty (RecordField  nm ty) = "RecordField"  <+> prParens (pretty nm <+> ":" <+> pretty ty)
  736     pretty (VariantField nm ty) = "VariantField" <+> prParens (pretty nm <+> ":" <+> pretty ty)

  753 data AbstractPattern
  754     = Single Name
  755     | AbsPatTuple [AbstractPattern]
  756     | AbsPatMatrix
  757             -- (Domain () a)          -- TODO: Should there be a domain here?
  758             [AbstractPattern]
  759     | AbsPatSet [AbstractPattern]
  760     -- | AbsPatMSet [a]
  761     -- | AbsPatFunction [(a, a)]
  762     -- | AbsPatRelation [[a]]
  763     -- | AbsPatPartition [[a]]
  764     -- TODO: Consider introducing the above as abstract patterns...
  765     | AbstractPatternMetaVar String
  766     deriving (Eq, Ord, Show, Data, Typeable, Generic)

  768 instance Serialize AbstractPattern
  769 instance Hashable  AbstractPattern
  770 instance ToJSON    AbstractPattern where toJSON = genericToJSON jsonOptions
  771 instance FromJSON  AbstractPattern where parseJSON = genericParseJSON jsonOptions
  772 
  773 instance Pretty AbstractPattern where
  774     pretty (Single       nm) = pretty nm
  775     pretty (AbsPatTuple  xs) = (if length xs <= 1 then "tuple" else prEmpty) <>
  776                                prettyList prParens   "," xs
  777     pretty (AbsPatMatrix xs) = prettyList prBrackets "," xs
  778     pretty (AbsPatSet    xs) = prettyList prBraces "," xs
  779     pretty (AbstractPatternMetaVar s) = "&" <> pretty s

  802 patternToExpr :: AbstractPattern -> Expression
  803 patternToExpr (Single nm) = Reference nm Nothing
  804 patternToExpr (AbsPatTuple  ts) = AbstractLiteral $ AbsLitTuple  $ map patternToExpr ts
  805 patternToExpr (AbsPatMatrix ts) = AbstractLiteral $ AbsLitMatrix
  806                                     (DomainInt TagInt [RangeBounded 1 (fromInt (genericLength ts))])
  807                                                                  $ map patternToExpr ts

  846 data Generator
  847      = GenDomainNoRepr  AbstractPattern (Domain () Expression)
  848      | GenDomainHasRepr Name            (Domain HasRepresentation Expression)
  849      | GenInExpr        AbstractPattern Expression
  850     deriving (Eq, Ord, Show, Data, Typeable, Generic)
  851 
  852 instance Pretty Generator where
  853     pretty (GenDomainNoRepr  pat x) = pretty pat <+> ":"  <+> pretty x
  854     pretty (GenDomainHasRepr pat x) = pretty pat <+> ":"  <+> pretty x
  855     pretty (GenInExpr        pat x) = pretty pat <+> "<-" <+> pretty x
  856 
  857 instance VarSymBreakingDescription Generator where
  858     varSymBreakingDescription (GenDomainNoRepr  pat x) = JSON.Object $ M.fromList
  859         [ ("type", JSON.String "GenDomainNoRepr")
  860         , ("pattern", varSymBreakingDescription pat)
  861         , ("generator", varSymBreakingDescription x)
  862         ]
  863     varSymBreakingDescription (GenDomainHasRepr pat x) = JSON.Object $ M.fromList
  864         [ ("type", JSON.String "GenDomainHasRepr")
  865         , ("pattern", toJSON pat)
  866         , ("generator", varSymBreakingDescription x)
  867         ]
  868     varSymBreakingDescription (GenInExpr        pat x) = JSON.Object $ M.fromList
  869         [ ("type", JSON.String "GenInExpr")
  870         , ("pattern", varSymBreakingDescription pat)
  871         , ("generator", varSymBreakingDescription x)
  872         ]
  873 
  874 instance Serialize Generator
  875 instance Hashable  Generator
  876 instance ToJSON    Generator where toJSON = genericToJSON jsonOptions
  877 instance FromJSON  Generator where parseJSON = genericParseJSON jsonOptions
  878 
  879 generatorPat :: Generator -> AbstractPattern
  880 generatorPat (GenDomainNoRepr  pat _) = pat
  881 generatorPat (GenDomainHasRepr pat _) = Single pat
  882 generatorPat (GenInExpr        pat _) = pat
