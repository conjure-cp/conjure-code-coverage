  248 -- the fst component: generate a list yielding the elements of the input list in order
  249 -- the snd component: is all those elements except the fst.
  250 withRest :: [a] -> [(a,[a])]
  251 withRest [] = []
  252 withRest (x:xs) = (x,xs) : map (second (x:)) (withRest xs)
  253 
  254 -- generate a list yielding the elements of the input list in order in the fst component.
  255 -- the snd component is all those elements to the right of fst.
  256 withAfter :: [a] -> [(a,[a])]
  257 withAfter [] = []
  258 withAfter (x:xs) = (x,xs) : withAfter xs

  260 -- generate a list yielding the elements of the input list in order in the fst component.
  261 -- the snd component is all those elements to the left of fst.
  262 withBefore :: [a] -> [(a,[a])]
  263 withBefore = reverse . withAfter . reverse

  438 -- | "failCheaply: premature optimisation at its finest." - Oz
  439 --   If you have a (MonadFail m => m a) action at hand which doesn't require anything else from the monad m,
  440 --   it can be run in any monad that implements MonadFail.
  441 --   Running it in a monad like IO will be a little bit more expensive though.
  442 --   Why not run it in Either and raise the error in the outer monad instead?

  444 failCheaply :: MonadFail m2 => (forall m . MonadFail m => m a) -> m2 a
  445 failCheaply m = either fail return m

  448 allContexts :: Data b => Zipper a b -> [Zipper a b]
  449 allContexts z0 = concatMap subtreeOf (allSiblings z0)
  450     where
  451         -- the input has to be the left most
  452         allSiblings :: Zipper a b -> [Zipper a b]
  453         allSiblings z = z : maybe [] allSiblings (right z)
  454 
  455         subtreeOf :: Data b => Zipper a b -> [Zipper a b]
  456         subtreeOf z = z : maybe [] allContexts (down z)

  523 runLoggerPipeIO :: MonadIO m => LogLevel -> Pipes.Producer (Either (LogLevel, Doc) a) m r -> m r

  525     where
  526         each (Left (lvl, msg)) =
  527             when (lvl <= l) $ do
  528                 let txt = Pr.renderStyle (Pr.style { Pr.lineLength = 200 }) msg
  529                 when ("[" `isPrefixOf` txt) $ do
  530                     liftIO clearScreen
