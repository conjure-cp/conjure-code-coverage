   16 data LSPConfig = LSPConfig {}
   17 
   18 startServer :: LSPConfig -> IO ()
   19 startServer _ = do
   20     _ <- runServer $ conjureLanguageServer
   21     return ()
   22 
   23 
   24 
   25 conjureLanguageServer :: ServerDefinition ()
   26 conjureLanguageServer =
   27     ServerDefinition
   28         { onConfigurationChange = const $ pure $ Right ()
   29         , doInitialize = \env _req -> pure $ Right env
   30         , staticHandlers = handlers
   31         , interpretHandler = \env -> Iso (runLspT env) liftIO
   32         , options = lspOptions
   33         , defaultConfig = def
   34         }
   35 
   36 handlers :: Handlers (LspM ())
   37 handlers =
   38     mconcat $ unhandled ++
   39         [ fileHandlers
   40         , handleInitialized
   41         , hoverHandler
   42         , docSymbolHandler
   43         , semanticTokensHandler
   44         , formatHandler
   45         ]
   46 
   47 
   48 syncOptions :: J.TextDocumentSyncOptions
   49 syncOptions = J.TextDocumentSyncOptions
   50     (Just True)
   51     (Just J.TdSyncIncremental)
   52     (Just False)
   53     (Just False)
   54     (Just $ J.InR $ J.SaveOptions $ Just False)
   55   
   56 
   57 lspOptions :: Options
   58 lspOptions = defaultOptions
   59   { textDocumentSync = Just syncOptions
   60   , executeCommandCommands = Just ["lsp-hello-command"]
   61   }
